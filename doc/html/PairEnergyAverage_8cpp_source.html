00118          if (simulation().domain().isMaster()) {
00119             DMatrix<double> pair = simulation().pairEnergies();
00120             for (int i = 0; i < simulation().nAtomType(); ++i){
00121                for (int j = 0; j < simulation().nAtomType(); ++j){
00122                   pair(i,j) = 0.5*( pair(i,j)+pair(j,i) );
00123                   pair(j,i) = pair(i,j);

00126             accumulator_->sample(pair(pairs_[0],pairs_[1]));
