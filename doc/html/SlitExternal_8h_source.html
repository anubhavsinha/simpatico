00200    * Calculate repulsive potential energy for particle of type i.
00201    *
00202    * 7.905694 = 25/sqrt(10), which is the negative of the minimum of the
00203    * unshifted 9-3 potential.

00205    inline double SlitExternal::energy(double d, int type) const
00206    {
00207       if (d < cutoff_) {
00208          double r3i;
00209          r3i = d * d * d;
00210          if (r3i < 0.129615*sigmaCb_) r3i = 0.129615 * sigmaCb_;
00211          r3i = sigmaCb_ / r3i;
00212          return coeff_ * (r3i*(r3i*r3i - 7.5) + 7.905694);
00213       } else {
00214          return 0.0;
