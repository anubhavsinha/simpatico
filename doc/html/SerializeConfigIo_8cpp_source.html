00059    int SerializeConfigIo::loadGroups(Serializable::IArchive& ar,
00060                                      GroupDistributor<N>& distributor) 
00061    {
00062       int nGroup = 0;  // Total number of groups in archive
00063       if (domain().isMaster()) {  
00064          ar >> nGroup;
00065          Group<N>* groupPtr;
00066          distributor.setup();
00067          for (int i = 0; i < nGroup; ++i) {
00068             groupPtr = distributor.newPtr();
00069             ar >> *groupPtr;
00070             distributor.add();
00071          }
00072          // Send any groups not sent previously.
00073          distributor.send();
00074       } else { // If I am not the master processor
00075          // Receive all groups into BondStorage
00076          distributor.receive();
00077       }
00078       return nGroup; // Valid only on master
00079    }

00111 
00112          #ifdef UTIL_MPI
00113          //Initialize the send buffer.
00114          atomDistributor().setup();
00115          #endif
00116

00118          Vector  r;
00119          Atom* atomPtr;

00122          int  typeId;

00126             atomPtr = atomDistributor().newAtomPtr();

00147             ar >> atomPtr->velocity();

00150             atomDistributor().addAtom();

00152
