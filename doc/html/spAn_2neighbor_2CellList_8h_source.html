00366    /*
00367    * Return value of offset for cell a distance x along axis i shifted to the range
00368    * 0 <= x <= grid_.Dimension(i)
00369    */
00370    inline int CellList::calculateAxisOffset(int cellId, int i, int x) const
00371    {
00372       // mod the shift so it's within the grid dimension
00373       x %= grid_.dimension(i);
00374 
00375       // calculate cell's position within the grid
00376       int p[3]; int j;
00377       for (j = 0; j < Dimension -1; ++j) {
00378          p[j] = cellId/gridOffsets_[j];
00379          cellId -= p[j]*gridOffsets_[j];
00380       }
00381       p[j] = cellId;
00382 
00383       if (p[i] + x >= grid_.dimension(i)) return (x - grid_.dimension(i))*gridOffsets_[i];
00384       if (p[i] + x < 0) return (grid_.dimension(i) + x)*gridOffsets_[i];
00385       return x*gridOffsets_[i];
00386    }

00393    inline int CellList::cellIndexFromPosition(const Vector& position) const
00394    {
00395       IntVector r;
00396       for (int i = 0; i < Dimension; ++i) {
00397          if (position[i] <= lower_[i]) {
00398             return -1;
00399          }
00400          if (position[i] >= upper_[i]) {
00401             return -1;
00402          }
00403          r[i] = int( (position[i] - lower_[i]) / cellLengths_[i] );
