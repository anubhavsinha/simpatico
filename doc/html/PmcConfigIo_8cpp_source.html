00059 
00060 
00061       // Read number of molecules of each species.
00062       Species  *speciesPtr;
00063       int       iSpec, nMolecule, nAtomSum;
00064       nAtomSum = 0;
00065       for (iSpec=0; iSpec < simulation().nSpecies(); ++iSpec) {
00066          speciesPtr = &simulation().species(iSpec);
00067          in >> Label("nMolecule") >> nMolecule;
00068          nMoleculeSpecies.push_back(nMolecule);
00069          nAtomSum += nMolecule*speciesPtr->nAtom();
00070       }
00071       if (nAtomSum != nAtom) {
00072          UTIL_THROW("Input nParticle != Atom Sum over Species");
00073       }
00074       Log::file() << "nAtomSum =" << nAtomSum;
00075 
00076       // Read atomic positions
00077       Molecule              *molPtr;
00078       Molecule::AtomIterator atomIter;
00079       int          atomType; // dummy, read and discarded
00080       for (iSpec=0; iSpec < simulation().nSpecies(); ++iSpec) {
00081          speciesPtr = &simulation().species(iSpec);
00082          nMolecule = nMoleculeSpecies[iSpec];
00083          for (int iMol=0; iMol < nMolecule; ++iMol) {
00084             molPtr = &(speciesPtr->reservoir().pop());
00085             system().addMolecule(*molPtr);
