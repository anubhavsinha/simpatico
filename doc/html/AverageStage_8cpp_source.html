83  void AverageStage::sample(double value)
84  {
85 
86  // Increment global accumulators
87  sum_ += value;
88  sumSq_ += (value*value);
89  ++nSample_;
90 
91  // Increment block accumulators
92  blockSum_ += value;
93  ++nBlockSample_;
94 
95  if (nBlockSample_ == blockFactor_) {
96 
97  if (!childPtr_) {
98  long nextStageInterval = stageInterval_*blockFactor_;
99  int nextStageId = stageId_ + 1;
100  childPtr_ = new AverageStage(nextStageInterval, nextStageId, rootPtr_, blockFactor_);
101  rootPtr_->registerDescendant(childPtr_);
102  }
103 
104  // Add block average to child ensemble
105  childPtr_->sample( blockSum_ / double(blockFactor_) );
106 
107  // Reset block accumulators
108  blockSum_ = 0.0;
109  nBlockSample_ = 0;

111  }
