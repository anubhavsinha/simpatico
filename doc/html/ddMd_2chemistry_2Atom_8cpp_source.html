105  buffer.unpack<Vector>(velocity());
106  #ifdef DDMD_MOLECULES

109  unsigned int ui;

111  plan().setFlags(ui);
112 
113  Mask& m = mask();
114  m.clear();
115  int size;
116  buffer.unpack<int>(size);
117  for (int j = 0; j < size; ++j) {
118  buffer.unpack<int>(i);
119  m.append(i);
120  }

122  buffer.decrementRecvSize();
