00154    void IntraBondTensorAutoCorr<SystemType>::sample(long iStep)

00157          Boundary& boundary = system().boundary();

00159          Molecule::ConstBondIterator  bondIter;
00160          Tensor  t;

00163          int  i, j;
00164 
00165          // Validate nMolecule_ (must remain constant).
00166          if (nMolecule_ != system().nMolecule(speciesId_)) {
00167             UTIL_THROW("Number of molecules has changed.");
00168          }
00169 
00170          // Loop over molecules
00171          i = 0;
00172          system().begin(speciesId_, molIter); 
00173          for ( ; molIter.notEnd(); ++ molIter) {
00174             data_[i].zero();
00175 
00176             // Loop over bonds
00177             molIter->begin(bondIter);
00178             for ( ; bondIter.notEnd(); ++bondIter) {
00179                boundary.distanceSq(bondIter->atom(0).position(), 
00180                                    bondIter->atom(1).position(), dr);
00181                u.versor(dr);
00182                t.dyad(u, u);
00183                data_[i] += t;
00184             }
00185 
00186             // Remove trace
00187             trace  = data_[i].trace()/double(Dimension);
00188             for (j=0; j < Dimension; ++j) {
00189                data_[i](j, j) -= trace;
00190             }
00191             ++i;
00192          }
00193 
00194          accumulator_.sample(data_);
