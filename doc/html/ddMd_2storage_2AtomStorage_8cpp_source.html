00403    double AtomStorage::maxSqDisplacement()

00592    bool AtomStorage::isValid() const
00593    {
00594       if (nAtom() + atomReservoir_.size() != atomCapacity_) {
00595          UTIL_THROW("nAtom + reservoir size != local capacity"); 
00596       }
00597       if (nGhost() + ghostReservoir_.size() != ghostCapacity_) {
00598          UTIL_THROW("nGhost + reservoir size != ghost capacity"); 
00599       }
00600       if (nGhost() != map_.nGhost()) {
00601          UTIL_THROW("Inconsistent values of nGhost");
00602       }
00603 
00604       // Test validity of AtomMap.
00605       map_.isValid();
00606 
00607       // Iterate over, count and find local atoms on this processor.
00608       ConstAtomIterator localIter;
00609       Atom* ptr;
00610       int j = 0;
00611       for (begin(localIter); localIter.notEnd(); ++localIter) {
00612          ++j;
00613          if (localIter->isGhost()) {
00614             UTIL_THROW("Atom in atomSet is marked isGhost");
00615          }
00616          ptr = map_.find(localIter->id());
00617          if (ptr == 0) {
00618             UTIL_THROW("Unable to find local atom returned by iterator"); 
00619          }
00620          if (ptr != localIter.get()) {
00621             UTIL_THROW("Inconsistent find(localIter->id()"); 
00622          }
00623       }
00624       if (j != nAtom()) {
00625          UTIL_THROW("Number counted by localIterator != nAtom()"); 
00626       }
00627 
00628       // Iterate over, count and find ghost atoms
00629       ConstGhostIterator ghostIter;
00630       j = 0;
00631       for (begin(ghostIter); ghostIter.notEnd(); ++ghostIter) {
00632          ++j;
00633          if (!ghostIter->isGhost()) {
00634             UTIL_THROW("Atom in ghostSet is not marked isGhost");
00635          }
00636          ptr = map_.find(ghostIter->id());
00637          if (ptr == 0) {
00638             UTIL_THROW("find(ghostIter->id()) == 0"); 
00639          }
00640          // We do NOT test if ptr == ghostIter.get(), because it is possible
00641          // to have multiple atoms with the same id on one processor. One to 
00642          // one correspondence of ids and pointers is guaranteed only for 
00643          // local atoms.
00644       }
00645       #if 0
00646       if (j != nGhost()) {
00647          UTIL_THROW("Number counted by ghostIterator != nGhost()");
