113  /*
114  * Check consistency of data.
115  */
116  bool OrthorhombicBoundary::isValid()
117  {
118  double dot;
119  double twoPi = 2.0*Constants::Pi;
120  int i, j;
121 
122  OrthoRegion::isValid();
123  for (i = 0; i < Dimension; ++i) {
124  if (!feq(minima_[i], 0.0)) {
125  UTIL_THROW("minima_[i] != 0");
126  }
127  if (!feq(lengths_[i]*invLengths_[i], 1.0)) {
128  UTIL_THROW("invLengths_[i]*lengths_[i] != 1.0");
129  }
130  if (!feq(lengths_[i], bravaisBasisVectors_[i][i])) {
131  UTIL_THROW("bravaisBasisVectors_[i][i] != lengths_[i]");
132  }
133  for (j = 0; j < Dimension; ++j) {
134  dot = bravaisBasisVectors_[i].dot(reciprocalBasisVectors_[j]);
135  if (i == j) {
136  if (!feq(dot, twoPi)) {
137  UTIL_THROW("a[i].b[i] != twoPi");
138  }
139  } else {
140  if (!feq(dot, 0.0)) {
141  UTIL_THROW("a[i].b[j] != 0 for i != j");
142  }
143  if (!feq(bravaisBasisVectors_[i][j], 0.0)) {
144  UTIL_THROW("Nonzero off-diagonal element of bravaisBasisVectors_");
145  }
146  if (!feq(reciprocalBasisVectors_[i][j], 0.0)) {
147  UTIL_THROW("Nonzero off-diagonal element of reciprocalBasisVectors_");
148  }
149  }
150  }
151  }
152  return true;
153  }
