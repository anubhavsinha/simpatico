00078                   const char* sectionLabel, int nGroup,
00079                   GroupDistributor<N>& distributor) 
00080    {
00081       if (domain().isMaster()) {  
00082          file >> Label(sectionLabel);
00083          Group<N>* groupPtr;
00084          int i, j, k;
00085          distributor.setup();
00086          for (i = 0; i < nGroup; ++i) {
00087             groupPtr = distributor.newPtr();
00088             file >> k;
00089             groupPtr->setId(k-1);
00090             file >> k;
00091             groupPtr->setTypeId(k-1);
00092             for (j = 0; j < N; ++j) {
00093                file >> k;
00094                groupPtr->setAtomId(j, k-1);
00095             }
00096             distributor.add();
00097          }
00098          // Send any groups not sent previously.
00099          distributor.send();
00100       } else { // If I am not the master processor
00101          // Receive all groups into BondStorage
00102          distributor.receive();
00103       }
00104    }
