00462    void MdSimulation::simulate(int endStep, bool isContinuation)
00463    {
00464       Timer timer;
00465 
00466       if (isContinuation) {
00467          Log::file() << "Continuing simulation from iStep = " 
00468                      << iStep_ << std::endl;
00469       } else {
00470          iStep_ = 0;
00471          diagnosticManager().setup();
00472          system_.mdIntegrator().setup();
00473       }
00474       int beginStep = iStep_;
00475       int nStep = endStep - beginStep;
00476 
00477       // Main loop 
00478       Log::file() << std::endl;
00479       timer.start();
00480       for ( ; iStep_ < endStep; ++iStep_) {
00481 
00482          // Sample scheduled diagnostics
00483          if (Diagnostic::baseInterval > 0) {
00484             if (iStep_ % Diagnostic::baseInterval == 0) {
00485                system().shiftAtoms();
00486                writeRestart(writeRestartFileName_);
00487                diagnosticManager().sample(iStep_);
00488             }
00489          }
00490 
00491          #ifdef INTER_NOPAIR
00492          else {
00493             // When the pair potential is disabled, require that
00494             // Diagnostic::baseInterval != 0 to guarantee periodic 
00495             // shifting of atomic positions into primary cell.
00496             UTIL_THROW("Diagnostic::baseInterval == 0 with no pair potential");
00497          }
00498          #endif
00499 
00500          // Take one MD step with the MdIntegrator
00501          system_.mdIntegrator().step();
00502 
00503       }
00504       timer.stop();
00505       double time  = timer.time();
00506       double rstep = double(nStep);
00507 
00508       // Shift final atomic positions 
00509       system().shiftAtoms();
00510 
00511       // Final restart / diagnostics
00512       if (Diagnostic::baseInterval > 0) {
00513          if (iStep_ % Diagnostic::baseInterval == 0) {
00514             writeRestart(writeRestartFileName_);
00515             diagnosticManager().sample(iStep_);
00516          }
00517       }
00518 
00519       // Final diagnostic output
00520       diagnosticManager().output();
00521 
00522       // Output timing results
00523       Log::file() << std::endl;
00524       Log::file() << "Time Statistics" << std::endl;
00525       Log::file() << "endStep              " << endStep << std::endl;
00526       Log::file() << "nStep                " << nStep   << std::endl;
00527       Log::file() << "run time             " 
00528                   << time << " sec" << std::endl;
00529       Log::file() << "time / nStep         " 
00530                   << time / rstep << " sec" << std::endl;

00532                   <<  time / (rstep*double(system().nAtom())) 
00533                   << " sec" << std::endl;

00535       Log::file() << std::endl;
00536 
00537       #ifndef INTER_NOPAIR
