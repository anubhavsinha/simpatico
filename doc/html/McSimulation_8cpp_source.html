00605    void McSimulation::simulate(int endStep, bool isContinuation)
00606    {
00607       if (!isInitialized_) {
00608          UTIL_THROW("McSimulation not initialized");
00609       }
00610 
00611       if (isContinuation) {
00612          Log::file() << "Restarting from iStep = " 
00613                      << iStep_ << std::endl;
00614       } else {
00615          iStep_ = 0;
00616          diagnosticManager().setup();
00617          mcMoveManagerPtr_->setup();
00618       }
00619       int beginStep = iStep_;
00620       int nStep = endStep - beginStep;
00621       Log::file() << std::endl;
00622 
00623       // Main Monte Carlo loop
00624       Timer timer;
00625       timer.start();
00626       for ( ; iStep_ < endStep; ++iStep_) {
00627 
00628          // Diagnostics and restart outut
00629          if (Diagnostic::baseInterval > 0) {
00630             if (iStep_ % Diagnostic::baseInterval == 0) {
00631                writeRestart(writeRestartFileName_);
00632                diagnosticManager().sample(iStep_);
00633             }
00634          }
00635 
00636          // Choose and attempt an McMove
00637          mcMoveManagerPtr_->chooseMove().move();
00638 
00639          #ifdef MCMD_PERTURB
00640          #ifdef UTIL_MPI
00641          // Attempt replica move, if any.
00642          if (system().hasPerturbation()) {
00643             if (system().hasReplicaMove()) {
00644                if (system().replicaMove().isAtInterval(iStep_)) {
00645                   bool success = system().replicaMove().move();
00646                   #ifndef INTER_NOPAIR
00647                   if (success) {
00648                      system().pairPotential().buildCellList();
00649                   }
00650                   #endif
00651                }
00652             }
00653          }
00654          #endif
00655          #endif
00656 
00657       }
00658       timer.stop();
00659       double time = timer.time();
00660 
00661       // Final diagnostics / restart 
00662       assert(iStep_ == endStep);
00663       if (Diagnostic::baseInterval > 0) {
00664          if (iStep_ % Diagnostic::baseInterval == 0) {
00665             writeRestart(writeRestartFileName_);
00666             diagnosticManager().sample(iStep_);
00667          }
00668       }
00669 
00670       // Output results of all diagnostics to output files
00671       diagnosticManager().output();
00672 
00673       // Output results of move statistics to files
00674       mcMoveManagerPtr_->output();
00675 
00676       // Output time for the run
00677       Log::file() << std::endl;
00678       Log::file() << "endStep       " << endStep << std::endl;
00679       Log::file() << "nStep         " << nStep << std::endl;
00680       Log::file() << "run time      " << time
00681                   << " sec" << std::endl;
00682       double rStep = double(nStep);
00683       Log::file() << "time / nStep  " <<  time / rStep
00684                   << " sec" << std::endl;
00685       Log::file() << std::endl;
00686 
00687       // Print McMove acceptance statistics
00688       long attempt;
00689       long accept;
00690       using namespace std;
00691       Log::file() << "Move Statistics:" << endl << endl;
