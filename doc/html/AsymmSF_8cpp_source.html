100  void AsymmSF::sample(long iStep)
101  {
102  if (isAtInterval(iStep)) {
103 
104  Vector position;
105  std::complex<double> expFactor;
106  double product;
107  AtomIterator atomIter;
108  int i, j, typeId;
109 
110  makeWaveVectors();
111 
112  // Set all Fourier modes to zero
113  for (i = 0; i < nWave_; ++i) {
114  for (j = 0; j < nMode_; ++j) {
115  fourierModes_(i, j) = std::complex<double>(0.0, 0.0);
116  }
117  }
118 
119  simulation().atomStorage().begin(atomIter);
120  for ( ; atomIter.notEnd(); ++atomIter) {
121  position = atomIter->position();
122  typeId = atomIter->typeId();
123 
124  // Loop over wavevectors
125  for (i = 0; i < nWave_; ++i) {
126 
127  product = position.dot(waveVectors_[i]);
128  expFactor = exp( product*Constants::Im );
129  for (j = 0; j < nMode_; ++j) {
130  fourierModes_(i, j) += modes_(j, typeId)*expFactor;
131  }
132  }
133  }
134 
135  for (i = 0; i < nWave_; ++i) {
136  for (j = 0; j < nMode_; ++j) {
137  totalFourierModes_(i, j) = std::complex<double>(0.0, 0.0);
138  }
139  }
140 
141  #ifdef UTIL_MPI
142  // Loop over wavevectors
143  for (int i = 0; i < nWave_; ++i) {
144  for (int j = 0; j < nMode_; ++j) {
145  //Sum values from all processors.
146  simulation().domain().communicator().Reduce(&fourierModes_(i, j), &totalFourierModes_(i, j), 1,
147  MPI::DOUBLE_COMPLEX, MPI::SUM, 0);
148  }
149  }
150  #else
151  for (int i = 0; i < nWave_; ++i) {
152  for (int j = 0; j < nMode_; ++j) {
153  totalFourierModes_(i, j) = fourierModes_(i, j);
154  }
155  }
156  #endif
157 
158  if (simulation().domain().isMaster()) {
159 
160  // Increment structure factors
161  double volume = simulation().boundary().volume();
162  double norm;
163  for (j = 0; j < nMode_; ++j) {
164  double maxValue = 0.0;
165  IntVector maxIntVector;
166  double maxQ;
167  for (i = 1; i < nWave_; ++i) {
168  norm = std::norm(totalFourierModes_(i, j));
169  if ( double(norm/volume) >= maxValue ) {
170  maxValue = double(norm/volume);
171  maxIntVector = waveIntVectors_[i];
172  maxQ = waveVectors_[i].abs();
173  }
174  structureFactors_(i, j) += norm/volume;
175  }
176  maximumValue_[j].insert(maximumValue_[j].end(), 1, maxValue);
177  maximumWaveIntVector_[j].insert(maximumWaveIntVector_[j].end(), 1, maxIntVector);
178  maximumQ_[j].insert(maximumQ_[j].end(), 1, maxQ);
179  }
180 
181  }
182 
183  ++nSample_;
184  }

195  int i, j;
196 
197  // Calculate wavevectors

200  for (j = 0; j < Dimension; ++j) {
