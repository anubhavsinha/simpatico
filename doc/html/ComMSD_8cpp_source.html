00157    */
00158    void ComMSD::sample(long iStep) 
00159    { 
00160       if (!isAtInterval(iStep)) return;
00161 
00162       // Confirm that nMolecule has remained constant, and nMolecule > 0.
00163       if (nMolecule_ <= 0) {
00164          UTIL_THROW("nMolecule <= 0");
00165       }
00166       if (nMolecule_ != system().nMolecule(speciesId_)) {
00167          UTIL_THROW("Number of molecules has changed.");
00168       }
00169 
00170       Vector     r;
00171       IntVector  shift;
00172       Vector     lengths = system().boundary().lengths();
00173       int        i, j;
00174       for (i = 0; i < nMolecule_; ++i) {
00175 
00176          r = system().molecule(speciesId_, i).atom(0).position();
00177          system().boundary().shift(r);
00178 
00179          // Compare current r to previous position, oldPositions_[i]
00180          system().boundary().distanceSq(r, oldPositions_[i], shift);
00181 
00182          // If atom 0 crossed a boundary, increment its shift vector
00183          shifts_[i] += shift;
00184 
00185          // Store current position in box for comparison to next one
00186          oldPositions_[i] = r;
00187  
00188          // Add the other particle's positions to C.O.M.
00189          moleculeCom(i, r);
00190 
00191          // Reconstruct true position
00192          for (j = 0; j < Dimension; ++j) {
00193             truePositions_[i][j] = r[j] + double(shifts_[i][j])*lengths[j];
00194          }
00195  
00196       }
00197       accumulator_.sample(truePositions_);
00198 
00199    }

00231    void ComMSD::moleculeCom(int iMol, Vector &R0)
00232    {

00234 
00235       // Assuming a linear chain (including Ring molecules).
00236       for (int i = 1; i < nAtom_-1; ++i) {
00237          r1 = system().molecule(speciesId_, iMol).atom(i-1).position();
00238          r2 = system().molecule(speciesId_, iMol).atom(i).position();
00239          system().boundary().distanceSq(r2, r1, dr);
00240          R  += dr;
00241          R0 += R;
00242       }
00243 
00244       // Do the average.
00245       R0 /= double(nAtom_);
