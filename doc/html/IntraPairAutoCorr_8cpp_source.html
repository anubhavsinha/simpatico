00147    void IntraPairAutoCorr::sample(long iStep)

00153 
00154          // Validate nMolecules_
00155          if (nMolecule_ <= 0) {
00156             UTIL_THROW("nMolecule <= 0");
00157          }
00158          if (nMolecule_ != system().nMolecule(speciesId_)) {
00159             UTIL_THROW("Number of molecules has changed");
00160          }
00161 
00162          // Loop over molecules in species
00163          for (i = 0; i < nMolecule_; ++i) {
00164             moleculePtr = &system().molecule(speciesId_, i);
00165 
00166             // Build separation between atoms atom1Id and atom2Id 
00167             // by adding separations between consecutive atoms.
00168             data_[i].zero();
00169             for (j = atom1Id_ + 1; j <= atom2Id_; ++j) {
00170                r1 = moleculePtr->atom(j-1).position();
00171                r2 = moleculePtr->atom(j).position();
00172                system().boundary().distanceSq(r1, r2, dR);
00173                data_[i] += dR;
00174             }
00175 
00176          }
00177          accumulator_.sample(data_);
00178 
00179       } // if isAtInterval
