46  {

48  // Read and set dimensions of simulation Boundary.
49  in >> Label("BOUNDARY");
50  in >> boundary() ;
51 
52  // Get molecules and read atomic positions
53  Label speciesLabel("species");
54  Label nMoleculeLabel("nMolecule");
55  Label moleculeLabel("molecule");
56  Species* speciesPtr;
57  Molecule* molPtr;
58  Molecule::AtomIterator atomIter;
59  int iSpecies, iSpeciesIn, nMolecule, iMol, iMolIn;
60 
61  in >> Label("MOLECULES");
62  for (iSpecies = 0; iSpecies < simulation().nSpecies(); ++iSpecies) {
63  speciesPtr = &simulation().species(iSpecies);
64  in >> Label("species") >> iSpeciesIn;
65  if (iSpeciesIn != iSpecies) {
66  UTIL_THROW("Error: iSpeciesIn != iSpecies");
67  }
68  in >> Label("nMolecule") >> nMolecule;
69  for (iMol = 0; iMol < nMolecule; ++iMol) {
70  in >> moleculeLabel >> iMolIn;
71  molPtr = &(simulation().getMolecule(iSpecies));
72  system().addMolecule(*molPtr);
73 
74  if (iMolIn != iMol) {
75  UTIL_THROW("Error: iMolIn != iMol");
76  }
77  if (molPtr != &system().molecule(iSpecies, iMol)) {
78  UTIL_THROW("Molecule index error");
79  }
80 
81  // If mutable, read molecule state id
82  if (speciesPtr->isMutable()) {
83  speciesPtr->mutator().readMoleculeState(in, *molPtr);
84  }
85 
86  // Read positions, shift into primary cell
87  for (molPtr->begin(atomIter); atomIter.notEnd(); ++atomIter) {
88  readAtom(in, *atomIter);
89  }
90 
91  }
92  }

94  #ifdef INTER_TETHER
95  { // Scope block for variables
96 
97  // Read Tethers
98  Vector anchor;
99  Atom* atomPtr;
100  int iTether, nTether, iAtom;
101  in >> Label("TETHERS");
102  in >> Label("nTether") >> nTether;
103  for (iTether = 0; iTether < nTether; ++iTether) {
104  in >> iSpecies >> iMol >> iAtom >> anchor;
105  if (iSpecies >= simulation().nSpecies()) {
106  UTIL_THROW("Invalid species index for tether");
107  }
108  if (iMol >= system().nMolecule(iSpecies)) {
109  UTIL_THROW("Invalid molecule index for tether");
110  }
111  molPtr = &system().molecule(iSpecies, iMol);
112  if (iAtom >= molPtr->nAtom()) {
113  Log::file() << iAtom << " " << molPtr->nAtom() << std::endl;
114  UTIL_THROW("Invalid atom index for tether");
115  }
116  atomPtr = &molPtr->atom(iAtom);
117  system().tetherMaster().addTether(*atomPtr, anchor);
118  }

120  }

204  out << endl;
