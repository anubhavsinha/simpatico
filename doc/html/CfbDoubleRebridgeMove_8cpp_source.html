00286    /*
00287    * Scanning chains to find all possible pairs of rebriding sites satisfying
00288    * the distance criterion, then choose one randomly to commit the double
00289    * rebridging move.
00290    *
00291    *    probability = 1 / (number of pairs satisfying the distance criterion)
00292    *
00293    *    rebriding criterion:
00294    *
00295    *          distance(i1,j2) < bridgeLength
00296    *       && distance(i2,j1) < bridgeLength
00297    *
00298    *                   i1   i2
00299    *           -->------o-o-o----   i-chain
00300    *                     \ /
00301    *                      X
00302    *                     / \
00303    *           -->------o-o-o----   j-chain
00304    *                   j1   j2
00305    *
00306    *
00307    */
00308    bool CfbDoubleRebridgeMove::forwardScan(int sign, int &iMol,
00309                             int &jMol, int &beginId, double &prob)
00310    {
00311       Molecule *iPtr, *jPtr;
00312       Vector   iPos1, jPos1, iPos2, jPos2;
00313       int      nMol, nAtom, maxPair;
00314       int      signRegrow, nPairs, k, choice;
00315       int      *molBuf, *atomBuf;
00316       double   lengthSq, bridgeLengthSq;  
00317       bool     found;
00318 
00319       // Number of molecules
00320       nMol = system().nMolecule(speciesId_);
00321 
00322       // Randomly choose one molecule
00323       iPtr = &(system().randomMolecule(speciesId_));
00324       iMol = system().moleculeId(*iPtr);
00325 
00326       // Number of atoms per molecule
00327       nAtom = iPtr->nAtom();
00328 
00329       // Require that chain length - 2 >= nRegrow_
00330       if (nRegrow_ > nAtom - 2) {
00331          UTIL_THROW("nRegrow_  >= chain length");
00332       }
00333 
00334       // Allocate arrays holding candidate atom pairs
00335       maxPair = (nMol - 1) * (nAtom - 1 - nRegrow_);
00336       molBuf  = new int[maxPair];  
00337       atomBuf = new int[maxPair];  
00338       
00339       // Identify the direction of scaning
00340       beginId = (sign == -1) ? (nAtom-1) : 0;
00341 
00342       // Auxiliary variables
00343       signRegrow = sign * (nRegrow_ + 1);
00344       bridgeLengthSq = bridgeLength_ * bridgeLength_;
00345 
00346       // Nullify counters
00347       nPairs = 0; 
00348       found  = false;
00349 
00350       // Loop over j molecules to scan possible pairs
00351       for (jMol = 0; jMol < nMol; ++jMol) {
00352 
00353          jPtr = &(system().molecule(speciesId_, jMol));
00354 
00355          if (jMol != iMol) {
00356 
00357             for (k = 0; k <  nAtom - 1 - nRegrow_; ++k) {  
00358 
00359                // Check if i1 to j2 bridging is possible
00360                iPos1 = iPtr->atom(beginId + k*sign).position();
00361                jPos2 = jPtr->atom(beginId + signRegrow + k*sign).position();
00362                lengthSq = boundary().distanceSq(iPos1, jPos2);
00363 
00364                if (lengthSq < bridgeLengthSq) {
00365                   // Check if j2 to i1 bridging is possible          
00366                   jPos1 = jPtr->atom(beginId + k*sign).position();
00367                   iPos2 = iPtr->atom(beginId + signRegrow + k*sign).position();
00368                   lengthSq = boundary().distanceSq(iPos2, jPos1);
00369                   
00370                   if (lengthSq < bridgeLengthSq) {
00371                      molBuf[nPairs]  = jMol;
00372                      atomBuf[nPairs] = beginId + k*sign;
00373                      nPairs += 1;
00374                   }  
00375                }
00376 
00377             } // loop over atom positions
00378          } // if jMol != iMol
00379       } // loop over j molecules
00380       
00381       if (nPairs > 0) found = true;
00382       
00383       if (found) {
00384          // iMol is not changed after the random selection
00385          choice  = random().uniformInt(0, nPairs);  
00386          jMol    = molBuf[choice];
00387          beginId = atomBuf[choice] + sign; // return the the active atom ID
00388          prob    = 1.0 / double(nPairs);
00389       } else {
00390          iMol    = -1;  // invalid value
00391          jMol    = -1;  // invalid value
00392          beginId = -1;  // invalid value
00393          prob    = 0.0;
00394       }   
00395       
00396       // Deallocate memory and quit
00397       delete [] molBuf;
00398       delete [] atomBuf;
00399       return found;
00400
