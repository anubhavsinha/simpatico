00162 
00163    
00164       T_target_ = simulation().energyEnsemble().temperature();
00165       factor = exp(-dtHalf*(xi_ + xiDot_*dtHalf));
00166 
00167       // 1st half of velocity Verlet.
00168       atomStorage().begin(atomIter);
00169       for ( ; atomIter.notEnd(); ++atomIter) {
00170          atomIter->velocity() *= factor;
