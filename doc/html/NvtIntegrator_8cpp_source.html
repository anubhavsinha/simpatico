114  * Setup parameters before beginning of run.
115  */
116  void NvtIntegrator::setup()
117  {
118 
119  // Initialize state and clear statistics on first usage.
120  if (!isSetup()) {
121  clear();
122  setIsSetup();
123  }
124 
125  // Exchange atoms, build pair list, compute forces.
126  setupAtoms();
127 
128  // Calculate prefactors for acceleration
129  double dtHalf = 0.5*dt_;
130  double mass;
131  int nAtomType = prefactors_.capacity();
132  for (int i = 0; i < nAtomType; ++i) {
133  mass = simulation().atomType(i).mass();
134  prefactors_[i] = dtHalf/mass;
135  }
136 
137  // Initialize nAtom_, xiDot_
138  simulation().computeKineticEnergy();
139  #ifdef UTIL_MPI
140  atomStorage().computeNAtomTotal(domain().communicator());
141  #endif
142  if (domain().isMaster()) {
143  T_target_ = simulation().energyEnsemble().temperature();
144  nAtom_ = atomStorage().nAtomTotal();
145  T_kinetic_ = simulation().kineticEnergy()*2.0/double(3*nAtom_);
146  xiDot_ = (T_kinetic_/T_target_ -1.0)*nuT_*nuT_;
147  }
148  #ifdef UTIL_MPI
149  bcast(domain().communicator(), xiDot_, 0);
150  #endif
151 
152  }

154  void NvtIntegrator::integrateStep1()
155  {
156  Vector dv;
157  Vector dr;
158  double prefactor; // = 0.5*dt/mass
159  double dtHalf = 0.5*dt_;
160  double factor;
161  AtomIterator atomIter;
162 
163 
164  T_target_ = simulation().energyEnsemble().temperature();
165  factor = exp(-dtHalf*(xi_ + xiDot_*dtHalf));
166 
167  // 1st half of velocity Verlet.
168  atomStorage().begin(atomIter);
169  for ( ; atomIter.notEnd(); ++atomIter) {
170  atomIter->velocity() *= factor;
171  prefactor = prefactors_[atomIter->typeId()];
172  dv.multiply(atomIter->force(), prefactor);
173  atomIter->velocity() += dv;
174  dr.multiply(atomIter->velocity(), dt_);
175  atomIter->position() += dr;
176  }
177  }

184  double factor;

190  // 2nd half of velocity Verlet

193  prefactor = prefactors_[atomIter->typeId()];
194  dv.multiply(atomIter->force(), prefactor);
195  atomIter->velocity() += dv;
196  atomIter->velocity() *=factor;
197  }
198 
199  // Notify observers of change in velocity
200  simulation().velocitySignal().notify();
201 
202  // Update xiDot_ and xi_
203  simulation().computeKineticEnergy();
204  if (domain().isMaster()) {
205  xi_ += xiDot_*dtHalf;
206  T_kinetic_ = simulation().kineticEnergy()*2.0/double(3*nAtom_);
207  xiDot_ = (T_kinetic_/T_target_ - 1.0)*nuT_*nuT_;
208  xi_ += xiDot_*dtHalf;
