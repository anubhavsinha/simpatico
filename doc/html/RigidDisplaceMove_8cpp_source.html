77  * Generate, attempt and accept or reject a rigid molecule translation.
78  */
79  bool RigidDisplaceMove::move()
80  {
81  Vector oldPos, newPos, dr;
82  double newEnergy, oldEnergy;
83  Molecule* molPtr;
84  Atom* atomPtr;
85  int iAtom, j;
86 
87  incrementNAttempt();
88 
89  // Choose a molecule at random
90  molPtr = &(system().randomMolecule(speciesId_));
91 
92  // Calculate current energy and store old positions.
93  // Ignore covalent bonding energy, because it won't change.
94  oldEnergy = 0.0;
95  for (iAtom = 0; iAtom < nAtom_; ++iAtom) {
96  atomPtr = &molPtr->atom(iAtom);
97  oldPositions_[iAtom] = atomPtr->position();
98  #ifndef INTER_NOPAIR
99  oldEnergy += system().pairPotential().atomEnergy(*atomPtr);
100  #endif
101  #ifdef INTER_EXTERNAL
102  oldEnergy += system().externalPotential().atomEnergy(*atomPtr);
103  #endif
104  #ifdef INTER_TETHER
105  oldEnergy += system().atomTetherEnergy(*atomPtr);
106  #endif
107  }
108 
109  // Generate trial displacement Vector dr
110  for (j = 0; j < Dimension; ++j) {
111  dr[j] = random().uniform(-delta_, delta_);
112  }
113 
114  // Move every atom by dr and calculate new trial energy.
115  newEnergy = 0.0;
116  for (iAtom = 0; iAtom < nAtom_; ++iAtom) {
117  atomPtr = &molPtr->atom(iAtom);
118  atomPtr->position() += dr;
119  boundary().shift(atomPtr->position());
120  #ifndef INTER_NOPAIR
121  newEnergy += system().pairPotential().atomEnergy(*atomPtr);
122  #endif
123  #ifdef INTER_EXTERNAL
124  newEnergy += system().externalPotential().atomEnergy(*atomPtr);
125  #endif
126  #ifdef INTER_TETHER
127  newEnergy += system().atomTetherEnergy(*atomPtr);
128  #endif
129  }
130 
131  // Decide whether to accept the move
132  bool accept = random().metropolis(boltzmann(newEnergy - oldEnergy));
133 
134  if (accept) {
135 
136  #ifndef INTER_NOPAIR
137  // Update cell
138  for (iAtom = 0; iAtom < nAtom_; ++iAtom) {
139  system().pairPotential().updateAtomCell(molPtr->atom(iAtom));
140  }
141  #endif
142 
143  incrementNAccept();
144 
145  } else {
146 
147  // Return atoms to original positions
148  for (iAtom = 0; iAtom < nAtom_; ++iAtom) {
149  molPtr->atom(iAtom).position() = oldPositions_[iAtom];
150  }
151 
152  }
153  return accept;
154  }
