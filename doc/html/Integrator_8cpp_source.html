00089       if (nDihedralType()) {

00091          timer_.stamp(DIHEDRAL_FORCE);
00092       }
00093       #endif
00094       #ifdef INTER_EXTERNAL
00095       if (hasExternal()) {
00096          externalPotential().computeForces();
00097       }
00098       #endif
00099 
00100       // Reverse communication (if any)
00101       if (reverseUpdateFlag()) {
00102          exchanger().reverseUpdate();
00103       }
00104 
00105       // Send signal indicating change in atomic forces
00106       simulation().forceSignal().notify();
