103  /*
104  * Evaluate change in energy, add Boltzmann factor to accumulator.
105  */
106  void McMuExchange::sample(long iStep)
107  {
108  if (isAtInterval(iStep)) {
109 
110  // Preconditions
111  if (!system().energyEnsemble().isIsothermal()) {
112  UTIL_THROW("EnergyEnsemble is not isothermal");
113  }
114  if (nMolecule_ != system().nMolecule(speciesId_)) {
115  UTIL_THROW("nMolecule has changed since setup");
116  }
117 
118  McPairPotential& potential = system().pairPotential();
119  const CellList& cellList = potential.cellList();
120  System::MoleculeIterator molIter;
121  Atom* ptr0 = 0; // Pointer to first atom in molecule
122  Atom* ptr1 = 0; // Pointer to flipped atom
123  Atom* ptr2 = 0; // Pointer to neighboring atom
124  Mask* maskPtr = 0; // Mask of flipped atom
125  double rsq, dE, beta, boltzmann;
126  int j, k, nNeighbor, nFlip;
127  int i1, i2, id1, id2, t1, t2, t1New, iMol;
128  beta = system().energyEnsemble().beta();
129  nFlip = flipAtomIds_.size();
130 
131  // Loop over molecules in species
132  iMol = 0;
133  system().begin(speciesId_, molIter);
134  for ( ; molIter.notEnd(); ++molIter) {
135  dE = 0.0;
136  ptr0 = &molIter->atom(0);
137 
138  // Loop over flipped atoms
139  for (j = 0; j < nFlip; ++j) {
140  i1 = flipAtomIds_[j];
141  t1New = newTypeIds_[i1];
142  ptr1 = &molIter->atom(i1);
143  id1 = ptr1->id();
144  t1 = ptr1->typeId();
145  maskPtr = &(ptr1->mask());
146 
147  // Loop over neighboring atoms
148  cellList.getNeighbors(ptr1->position(), neighbors_);
149  nNeighbor = neighbors_.size();
150  for (k = 0; k < nNeighbor; ++k) {
151  ptr2 = neighbors_[k];
152  id2 = ptr2->id();
153 
154  // Check if atoms are identical
155  if (id2 != id1) {
156 
157  // Check if pair is masked
158  if (!maskPtr->isMasked(*ptr2)) {
159 
160  rsq = boundary().distanceSq(ptr1->position(),
161  ptr2->position());
162  t2 = ptr2->typeId();
163  if (&(ptr1->molecule()) != &(ptr2->molecule())) {
164 
165  // Intermolecular atom pair
166  dE -= potential.energy(rsq, t1, t2);
167  dE += potential.energy(rsq, t1New, t2);
168 
169  } else {
170 
171  // Intramolecular atom pair
172  if (id2 > id1) {
173  dE -= potential.energy(rsq, t1, t2);
174  i2 = (int)(ptr2 - ptr0);
175  t2 = newTypeIds_[i2];
176  dE += potential.energy(rsq, t1New, t2);
177  } else {
178  i2 = (int)(ptr2 - ptr0);
179  if (!isAtomFlipped_[i2]) {
180  dE -= potential.energy(rsq, t1, t2);
181  t2 = newTypeIds_[i2];
182  dE += potential.energy(rsq, t1New, t2);
183  }
184  }
185 
186  }
187  }
188  }
189  } // end loop over neighbors
190  } // end loop over flipped atoms
191 
192  boltzmann = exp(-beta*dE);
193  accumulators_[iMol].sample(boltzmann);
194  ++iMol;
195  } // end loop over molecules
196  }
197  }

199  /*
200  * Output results to file after simulation is completed.
201  */
202  void McMuExchange::output()
203  {
204 
205  fileMaster().openOutputFile(outputFileName(".prm"), outputFile_);
206  writeParam(outputFile_);
207  outputFile_.close();
208 
209  double ave, err;
210  double sumAve = 0.0;
211  double sumAveSq = 0.0;
212  double sumErrSq = 0.0;
213  for (int iMol=0; iMol < nMolecule_; ++iMol) {
214  ave = accumulators_[iMol].average();
215  err = accumulators_[iMol].blockingError();
216  sumAve += ave;
217  sumAveSq += ave*ave;
218  sumErrSq += err*err;
219  }
220  double rMol = double(nMolecule_);
221  ave = sumAve/rMol;
222  err = sqrt(sumErrSq/rMol);
223  double dev = sqrt((sumAveSq/rMol) - ave*ave);
224 
225  fileMaster().openOutputFile(outputFileName(".ave"), outputFile_);
226  outputFile_ << "Average = " << ave << " +- "
227  << dev/sqrt(rMol) << std::endl;
228  outputFile_ << "Error via molecule variance = "
229  << dev/sqrt(rMol) << std::endl;
230  outputFile_ << "Error via time series analysis = "
231  << err/sqrt(rMol) << std::endl;
232  outputFile_ << std::endl;
233  //for (int iMol=0; iMol < nMolecule_; ++iMol) {
234  // accumulators_[iMol].output(outputFile_);
235  //}
236  outputFile_.close();
237  }
