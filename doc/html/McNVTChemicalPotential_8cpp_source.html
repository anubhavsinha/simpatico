379  McNVTChemicalPotential::addEndAtom(Molecule* molPtr, int atomId, double &rosenbluth, double &energy)

387  Vector pvt2Pos = molPtr->atom(atomId-2).position();
388  #endif
389 
390  #ifdef INTER_DIHEDRAL
391  Vector pvt3Pos = molPtr->atom(atomId-3).position();
392  #endif
393 
394  double trialProb[MaxTrial_], trialEnergy[MaxTrial_];
395  double beta, length;
396  int bondTypeId = molPtr->bond(atomId - 1).typeId();
397 
398  #ifdef INTER_ANGLE
399  double angle;
400  int angleTypeId = molPtr->angle(atomId - 2).typeId();
401  #endif
402 
403  int iTrial;
404  double eMin = 0;
405 
406  // Generate a random bond-length and bond-angle.
407  beta = energyEnsemble().beta();
408  length = system().bondPotential().randomBondLength(&random(), beta, molPtr->bond(atomId-1).typeId());
409  #ifdef INTER_ANGLE
410  Vector n = pvt1Pos;
411  n -= pvt2Pos;
412  angle = system().anglePotential().randomAngle(&random(), beta, molPtr->angle(atomId-2).typeId());
413  #endif
414  #ifdef INTER_DIHEDRAL
415  Vector dR1 = pvt3Pos;
416  Vector dR2 = pvt2Pos;
417  Vector dR3 = pvt1Pos;
418  dR1 -= pvt2Pos;
419  dR2 -= pvt1Pos;
420  dR3 -= endPtr->position();
421  #endif
422 
423  // Loop over nTrial trial positions:
424  rosenbluth = 0.0;
425  for (iTrial=0; iTrial < nTrial_; ++iTrial) {
426 
427  #ifdef INTER_ANGLE
428  if (system().hasAnglePotential()) {
429  uniformCone(length, angle, n, bondVec);
430  } else {
431  random().unitVector(bondVec);
432  bondVec *= length;
433  }
434  #else
435  random().unitVector(bondVec);
436  bondVec *= length;
437  #endif
438 
439  trialPos[iTrial].add(pvt1Pos, bondVec);
440  boundary().shift(trialPos[iTrial]);
441  endPtr->position() = trialPos[iTrial];
442 
443  #ifndef INTER_NOPAIR
444  trialEnergy[iTrial] = system().pairPotential().atomEnergy(*endPtr);
445  #else
446  trialEnergy[iTrial] = 0.0;
447  #endif
448 
449  #ifdef INTER_DIHEDRAL
450  if (system().hasDihedralPotential()) {
451  trialEnergy[iTrial] += system().dihedralPotential().energy(dR1, dR2, dR3, molPtr->dihedral(atomId-3).typeId());
452  }
453  #endif
454 
455  #ifdef INTER_EXTERNAL
456  trialEnergy[iTrial] += system().externalPotential().atomEnergy(*endPtr);
457  #endif
458 
459  // Finding the Minimum of the trialEnergy vector
460  if (eMin > trialEnergy[iTrial])
461  eMin = trialEnergy[iTrial];
462  Eaccumulator_.sample(trialEnergy[iTrial]);
463 
464  trialProb[iTrial] = boltzmann(trialEnergy[iTrial]);
465  rosenbluth += trialProb[iTrial];
466  }
467 
468  Emaccumulator_.sample(eMin);
469  BRaccumulator_.sample(rosenbluth);
470  // Normalize trial probabilities
471  for (iTrial = 0; iTrial < nTrial_; ++iTrial) {
472  trialProb[iTrial] = trialProb[iTrial]/rosenbluth;
473  }
474 
475  // Choose trial position
476  iTrial = random().drawFrom(trialProb, nTrial_);
477 
478  // Calculate total energy for chosen trial.
479  energy = system().bondPotential().energy(length*length, bondTypeId);
480  #ifdef INTER_ANGLE
481  energy += system().anglePotential().energy(cos(angle), angleTypeId);
482  #endif
483 
484  energy += trialEnergy[iTrial];
485  Ecaccumulator_.sample(energy);
486 
487  // Set position of new end atom to chosen value
488  endPtr->position() = trialPos[iTrial];
489  }

491  /*
492  * Finding a vector which make an angle \theta with vector n and has length b.
493  */
494  void McNVTChemicalPotential::uniformCone(const double length, const double angle, const Vector n, Vector &p)
495  {
496  Vector v1;
497 
498  do {
499  random().unitVector(p);
500  v1.divide(n,n.abs());
501  v1.multiply(v1,p.projection(v1));
502  p -= v1;
503  } while (p.abs() > 1.0E-8);
504 
505  v1.divide(n,n.abs());
506  v1 *= length*cos(angle);
507  p /= p.abs();
508  p *= length*sin(angle);
509  p += (v1);
510  }

512 }
