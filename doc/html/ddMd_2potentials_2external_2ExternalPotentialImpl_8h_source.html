320  /*
321  * Increment atomic forces and/or external energy (private).
322  */
323  template <class Interaction>
324  double
325  ExternalPotentialImpl<Interaction>::computeForces(bool needForce, bool needEnergy)
326  {
327  // Preconditions
328  //if (!storage().isInitialized()) {
329  // UTIL_THROW("AtomStorage must be initialized");
330  //}
331 
332  Vector f;
333  double energy = 0.0;
334  AtomIterator iter;
335  int type;
336 
337  storage().begin(iter);
338  for ( ; iter.notEnd(); ++iter) {
339  type = iter->typeId();
340  if (needEnergy) {
341  energy +=
342  interactionPtr_->energy(iter->position(), type);
343  }
344  if (needForce) {
345  interactionPtr_->getForce(iter->position(), type, f);
346  iter->force() += f;
347  }
348  }
349  return energy;
350  }
