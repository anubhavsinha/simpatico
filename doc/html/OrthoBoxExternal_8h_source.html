208  /*
209  * Calculate repulsive potential energy for particle of type i.
210  *
211  * 7.905694 = 25/sqrt(10), which is the negative of the minimum of the
212  * unshifted 9-3 potential.
213  */
214  inline double OrthoBoxExternal::energy(double d, int type) const
215  {
216  if (d < cutoff_) {
217  double r3i;
218  r3i = d * d * d;
219  if (r3i < 0.129615*sigmaCb_) r3i = 0.129615 * sigmaCb_;
220  r3i = sigmaCb_ / r3i;
221  return coeff_ * (r3i*(r3i*r3i - 7.5) + 7.905694);
222  } else {
223  return 0.0;
224  }
225  }

230  inline double OrthoBoxExternal::energy(const Vector& position, int type) const
231  {
232  double d, totalEnergy;
233  Vector halfL = boundaryPtr_->lengths();
234  halfL *= 0.5;
235  totalEnergy = 0.0;
236  for (int i = indConfine_; i < Dimension; ++i) {
237  d = position[i];
238  if (d > halfL[i]) d = halfL[i] + halfL[i] - d;
239  totalEnergy += energy(d, type);
240  }
241  return totalEnergy;

247  inline double OrthoBoxExternal::forceScalar(double d, int i) const

253  r3i = sigmaCb_ / r3i;
