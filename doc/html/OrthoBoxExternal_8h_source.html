00209    * Calculate repulsive potential energy for particle of type i.
00210    *
00211    * 7.905694 = 25/sqrt(10), which is the negative of the minimum of the
00212    * unshifted 9-3 potential.
00213    */
00214    inline double OrthoBoxExternal::energy(double d, int type) const
00215    {
00216       if (d < cutoff_) {
00217          double r3i;
00218          r3i = d * d * d;
00219          if (r3i < 0.129615*sigmaCb_) r3i = 0.129615 * sigmaCb_;
00220          r3i = sigmaCb_ / r3i;
00221          return coeff_ * (r3i*(r3i*r3i - 7.5) + 7.905694);
00222       } else {
00223          return 0.0;
00224       }
00225    }
