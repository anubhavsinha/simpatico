44  * Calculate the energy change of rebridging a tetrahedron formed from
45  * 4 atoms.
46  *
47  * a - b a b
48  * -> X
49  * d - c d c
50  *
51  * Erase ab & cd bonds, and add ac & bd bonds. Energy change includes the
52  * bonding potential changes and the pairing potential changes.
53  */
54  void GroupRebridgeBase::
55  tetraEnergy(Atom* aPtr, Atom* bPtr, Atom* cPtr, Atom* dPtr,
56  int bondType, double &energy)
57  {
58  Vector aPos = aPtr->position();
59  Vector bPos = bPtr->position();
60  Vector cPos = cPtr->position();
61  Vector dPos = dPtr->position();
62  double lab, lcd, lac, lbd;
63 
64  energy = 0.0;
65 
66  // Calculate bond lengthSq.
67  lab = boundary().distanceSq(aPos, bPos);
68  lcd = boundary().distanceSq(cPos, dPos);
69  lac = boundary().distanceSq(aPos, cPos);
70  lbd = boundary().distanceSq(bPos, dPos);
71 
72  // Subtract old bonding energies.
73  energy -= system().bondPotential().energy(lab, bondType);
74  energy -= system().bondPotential().energy(lcd, bondType);
75 
76  // Add new bonding energies.
77  energy += system().bondPotential().energy(lac, bondType);
78  energy += system().bondPotential().energy(lbd, bondType);

83  int cId = cPtr->typeId();
84  int dId = dPtr->typeId();

87  energy -= system().pairPotential().energy(lac, aId, cId);
88  energy -= system().pairPotential().energy(lbd, bId, dId);
89 
90  // Add new pairing potentials.
91  energy += system().pairPotential().energy(lab, aId, bId);
92  energy += system().pairPotential().energy(lcd, cId, dId);
93  #endif

95  #ifdef INTER_ANGLE

188  * the pair of atoms involved in the newly added bonds lost their pairing
189  * interaction.
