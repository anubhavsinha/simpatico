00071    * Reset all quantities that depend on unit cell lengths.
00072    */
00073    void MonoclinicBoundary::reset()
00074    {
00075       double twoPi = 2.0*Constants::Pi;
00076       for (int i = 0; i < Dimension; ++i) {
00077          minima_[i] = 0.0;
00078          maxima_[i] = l_[i];
00079          halfL_[i] = 0.5*l_[i];
00080          invL_[i] = 1.0/l_[i];
00081          bravaisBasisVectors_[i][i] = l_[i];
00082          reciprocalBasisVectors_[i][i] = twoPi/l_[i];
00083       }
00084       bravaisBasisVectors_[1][2] = d_;
00085       reciprocalBasisVectors_[2][1] = -twoPi*d_/(l_[1]*l_[2]);
00086 
00087       volume_ = l_[0] * l_[1] * l_[2];
00088       e_ = sqrt(l_[1]*l_[1] + d_*d_);          
00089       c3_ = -d_/l_[1];
00090 
00091       lengths_[0] = l_[0];
00092       lengths_[1] = l_[1];
00093       lengths_[2] = l_[2] / sqrt(1.0 + c3_*c3_);
00094 
00095       minLength_ = lengths_[0];
00096       for (int i = 1; i < Dimension; ++i) {
00097          if (lengths_[i] < minLength_) {
00098             minLength_ = lengths_[i];
00099          }
00100       }
00101    }

00121    bool MonoclinicBoundary::isValid() 
00122    {  
00123       double dot;
00124       double twoPi = 2.0*Constants::Pi;
00125       int i, j;
00126       for (i = 0; i < Dimension; ++i) {
00127          if (maxima_[i] <= minima_[i]) {
00128             UTIL_THROW("maxima_[i] <= minima_[i]");
00129          }
00130          if (!feq(l_[i], maxima_[i] - minima_[i])) {
00131             UTIL_THROW("l_[i] != maxima_[i] - minima_[i]");
00132          }
00133          if (!feq(halfL_[i], 0.5*l_[i])) {
00134             UTIL_THROW("halfL_[i] != 0.5*l_[i]");
00135          }
00136          if (!feq(minima_[i], 0.0)) {
00137             UTIL_THROW("minima_[i] != 0");
00138          }
00139          if (!feq(l_[i]*invL_[i], 1.0)) {
00140             UTIL_THROW("invL_[i]*l_[i] != 1.0");
00141          }
00142          for (j = 0; j < Dimension; ++j) {
00143             dot = bravaisBasisVectors_[i].dot(reciprocalBasisVectors_[j]);
00144             if (i == j) {
00145                if (!feq(dot, twoPi)) {
00146                   UTIL_THROW("a[i].b[i] != twoPi");
00147                } 
00148             } else {
00149                if (!feq(dot, 0.0)) {
00150                   UTIL_THROW("a[i].b[j] != 0 for i != j");
00151                }
00152             }
00153          }
00154       }
00155       if (!feq(volume_, l_[0]*l_[1]*l_[2])) {
00156          UTIL_THROW("volume_ != product of l_");
00157       }
00158       return true;
00159    }
