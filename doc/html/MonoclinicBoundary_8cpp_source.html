00068    * Reset all quantities that depend on unit cell lengths.
00069    */
00070    void MonoclinicBoundary::reset()
00071    {
00072       double twoPi = 2.0*Constants::Pi;
00073       for (int i = 0; i < Dimension; ++i) {
00074          minima_[i] = 0.0;
00075          maxima_[i] = l_[i];
00076          halfL_[i] = 0.5*l_[i];
00077          invL_[i] = 1.0/l_[i];
00078          bravaisBasisVectors_[i][i] = l_[i];
00079          reciprocalBasisVectors_[i][i] = twoPi/l_[i];
00080       }
00081       bravaisBasisVectors_[1][2] = d_;
00082       reciprocalBasisVectors_[2][1] = -twoPi*d_/(l_[1]*l_[2]);
00083 
00084       volume_ = l_[0] * l_[1] * l_[2];
00085       e_ = sqrt(l_[1]*l_[1] + d_*d_);          
00086       c3_ = -d_/l_[1];
00087 
00088       lengths_[0] = l_[0];
00089       lengths_[1] = l_[1];
00090       lengths_[2] = l_[2] / sqrt(1.0 + c3_*c3_);
00091 
00092       minLength_ = lengths_[0];
00093       for (int i = 1; i < Dimension; ++i) {
00094          if (lengths_[i] < minLength_) {
00095             minLength_ = lengths_[i];
00096          }
00097       }
00098    }

00118    bool MonoclinicBoundary::isValid() 
00119    {  
00120       double dot;
00121       double twoPi = 2.0*Constants::Pi;
00122       int i, j;
00123       for (i = 0; i < Dimension; ++i) {
00124          if (maxima_[i] <= minima_[i]) {
00125             UTIL_THROW("maxima_[i] <= minima_[i]");
00126          }
00127          if (!feq(l_[i], maxima_[i] - minima_[i])) {
00128             UTIL_THROW("l_[i] != maxima_[i] - minima_[i]");
00129          }
00130          if (!feq(halfL_[i], 0.5*l_[i])) {
00131             UTIL_THROW("halfL_[i] != 0.5*l_[i]");
00132          }
00133          if (!feq(minima_[i], 0.0)) {
00134             UTIL_THROW("minima_[i] != 0");
00135          }
00136          if (!feq(l_[i]*invL_[i], 1.0)) {
00137             UTIL_THROW("invL_[i]*l_[i] != 1.0");
00138          }
00139          for (j = 0; j < Dimension; ++j) {
00140             dot = bravaisBasisVectors_[i].dot(reciprocalBasisVectors_[j]);
00141             if (i == j) {
00142                if (!feq(dot, twoPi)) {
00143                   UTIL_THROW("a[i].b[i] != twoPi");
00144                } 
00145             } else {
00146                if (!feq(dot, 0.0)) {
00147                   UTIL_THROW("a[i].b[j] != 0 for i != j");
00148                }
00149             }
00150          }
00151       }
00152       if (!feq(volume_, l_[0]*l_[1]*l_[2])) {
00153          UTIL_THROW("volume_ != product of l_");
00154       }
00155       return true;
00156    }
