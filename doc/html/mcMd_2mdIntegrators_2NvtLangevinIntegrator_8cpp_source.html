80  /*
81  * Initialize constants.
82  */
83  void NvtLangevinIntegrator::setup()
84  {
85  const EnergyEnsemble& energyEnsemble = system().energyEnsemble();
86  if (!energyEnsemble.isIsothermal()) {
87  UTIL_THROW("Energy ensemble is not isothermal");
88  }
89  double cv = (exp(-dt_*gamma_) - 1.0)/dt_;
90  double temp = energyEnsemble.temperature();
91  double d = 2.0/(1.0 + exp(-dt_*gamma_));
92  double cr = 12.0*temp*d*(1.0 - exp(-2.0*dt_*gamma_))/(dt_*dt_);
93 
94  // Loop over atom types
95  double mass;
96  int nAtomType = prefactors_.capacity();
97  for (int i = 0; i < nAtomType; ++i) {
98  mass = simulation().atomType(i).mass();
99  prefactors_[i] = 0.5*dt_/mass;
100  cv_[i] = mass*cv;
101  cr_[i] = sqrt(mass*cr);
102  }
103 
104  }

106  /*
107  * Verlet MD NVE integrator step
108  *
109  * This method implements the algorithm:
110  *
111  * vm(n) = v(n) + 0.5*a(n)*dt
112  * x(n+1) = x(n) + vm(n)*dt
113  *
114  * calculate determinstic force f(n+1)
115  * add Langevin force, calculated using vm(n)
116  *
117  * v(n+1) = vm(n) + 0.5*f(n+1)*dt/m
118  *
119  * where x is position, v is velocity, and f is force.
120  */
121  void NvtLangevinIntegrator::step()
122  {
123  Vector dr;
124  Vector dv;
125  Vector df;
126  double cr;
127  System::MoleculeIterator molIter;
128  int iSpecies, nSpecies, typeId;
129 
130  nSpecies = simulation().nSpecies();
131 
132  // 1st half velocity Verlet, loop over atoms
133  #if USE_ITERATOR
134  Molecule::AtomIterator atomIter;
135  for (iSpecies=0; iSpecies < nSpecies; ++iSpecies) {
136  system().begin(iSpecies, molIter);
137  for ( ; molIter.notEnd(); ++molIter) {
138  molIter->begin(atomIter);
139  for ( ; atomIter.notEnd(); ++atomIter) {
140  typeId = atomIter->typeId();
141 
142  // Update velocity one half step, prefactor = dt/(2m)
143  dv.multiply(atomIter->force(), prefactors_[typeId]);
144  atomIter->velocity() += dv;
145 
146  // Update position (full step)
147  dr.multiply(atomIter->velocity(), dt_);
148  atomIter->position() += dr;
149  }
150  }
151  }
152  #else
153  Atom* atomPtr;
154  int ia;
155  for (iSpecies=0; iSpecies < nSpecies; ++iSpecies) {
156  system().begin(iSpecies, molIter);
157  for ( ; molIter.notEnd(); ++molIter) {
158  for (ia=0; ia < molIter->nAtom(); ++ia) {
159  atomPtr = &molIter->atom(ia);
160  typeId = atomPtr->typeId();
161 
162  // Update velocity one half step, prefactor = dt/(2m)
163  dv.multiply(atomPtr->force(), prefactors_[typeId]);
164  atomPtr->velocity() += dv;
165 
166  // Position update (full step)
167  dr.multiply(atomPtr->velocity(), dt_);
168  atomPtr->position() += dr;
169  }
170  }
171  }
172  #endif
173 
174  // Calculate conservative force
175  system().calculateForces();
176 
177  // 2nd half velocity Verlet, loop over atoms
178  Random& random = simulation().random();
179  int j;
180  #if USE_ITERATOR
181  for (iSpecies=0; iSpecies < nSpecies; ++iSpecies) {
182  system().begin(iSpecies, molIter);
183  for ( ; molIter.notEnd(); ++molIter) {
184  molIter->begin(atomIter);
185  for ( ; atomIter.notEnd(); ++atomIter) {
186  typeId = atomIter->typeId();
187 
188  // Add Langevin drag and random force to atomic force
189  df.multiply(atomIter->velocity(), cv_[typeId]);
190  cr = cr_[typeId];
191  for (j=0; j < Dimension; ++j) {
192  df[j] += (random.uniform() - 0.5)*cr;
193  }
194  atomIter->force() += df;
195 
196  // Velocity update by half step
197  dv.multiply(atomIter->force(), prefactors_[typeId]);
198  atomIter->velocity() += dv;
199  }
200  }
201  }
202  #else
203  for (iSpecies=0; iSpecies < nSpecies; ++iSpecies) {
204  system().begin(iSpecies, molIter);
205  for ( ; molIter.notEnd(); ++molIter) {
206  for (ia=0; ia < molIter->nAtom(); ++ia) {
207  atomPtr = &molIter->atom(ia);
208  typeId = atomPtr->typeId();
209 
210  // Add Langevin drag and random force to atomic force
211  df.multiply(atomPtr->velocity(), cv_[typeId]);
212  cr = cr_[typeId];
213  for (j=0; j < Dimension; ++j) {
214  df[j] += (random.uniform() - 0.5)*cr;
215  }
216  atomPtr->force() += df;
217 
218  // Velocity update by half step
219  dv.multiply(atomPtr->force(), prefactors_[typeId]);
220  atomPtr->velocity() += dv;
221  }
222  }
223  }
224  #endif
225 
226  #ifndef INTER_NOPAIR
227  if (!system().pairPotential().isPairListCurrent()) {
228  system().pairPotential().buildPairList();
229  }
230  #endif
231 
232  }
