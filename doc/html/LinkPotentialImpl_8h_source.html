327  template <class Interaction>
328  void LinkPotentialImpl<Interaction>::addForces()
329  {
330  Vector force;
331  double rsq;
332  Link* linkPtr;
333  Atom* atom0Ptr;
334  Atom* atom1Ptr;
335  int iLink, nLink;
336  nLink = linkMasterPtr_->nLink();
337  for (iLink = 0; iLink < nLink; ++iLink) {
338  linkPtr = &linkMasterPtr_->link(iLink);
339  atom0Ptr = &(linkPtr->atom0());
340  atom1Ptr = &(linkPtr->atom1());
341  rsq = boundary().
342  distanceSq(atom0Ptr->position(), atom1Ptr->position(), force);
343  force *= interaction().forceOverR(rsq, linkPtr->typeId());
344  atom0Ptr->force() += force;
345  atom1Ptr->force() -= force;

350  * Compute the link pressure or stress

352  template <class Interaction>
353  template <typename T>
354  void LinkPotentialImpl<Interaction>::computeStressImpl(T& stress) const
355  {
356  Vector dr;
357  Vector force;
358  double rsq;
359  Link* linkPtr;
360  int iLink, nLink;
361 
362  setToZero(stress);
363 
364  // Loop over links
365  nLink = linkMasterPtr_->nLink();
366  for (iLink = 0; iLink < nLink; ++iLink) {
367  linkPtr = &linkMasterPtr_->link(iLink);
368  rsq = boundary().distanceSq(linkPtr->atom0().position(),
369  linkPtr->atom1().position(), dr);
370  force = dr;
371  force *= interaction().forceOverR(rsq, linkPtr->typeId());
372  incrementPairStress(force, dr, stress);
373  }
374 
375  // Normalize by volume
376  stress /= boundary().volume();
377  normalizeStress(stress);
378  }
