00101       double prefactor;

00104       nSpecies = simulation().nSpecies();
00105 
00106       /* perform the first half step of the explicitly reversible NPH integration scheme.
00107   
00108          This follows from operator factorization
00109   
00110          - orthorhombic case:
00111   
00112            1) eta_alpha(t+dt/2) = eta_alpha(t) + dt/2 * V/L_alpha * ( P_{alpha,alpha}(t) - P_ext)
00113            2) v' = v(t) + (1/2)a(t)*dt
00114            3) L(t+dt/2) = L(t) + dt*eta(t+dt/2)/(2*W)
00115            4) r'_alpha = r(t) + v'*dt* (L_{alpha}^2(t)/L_{alpha}^2(t+dt/2))
00116            5a) L(t+dt) = L(t+dt/2) + dt*eta(t+dt/2)/2/W
00117            5b) r_alpha(t+dt) = L_alpha(t+dt)/L_alpha(t)*r'_alpha
00118            5c) v''_alpha = L_alpha(t)/L_alpha(t+dt) * v'_alpha
00119   
00120          alpha denotes a cartesian index.
00121   
00122          - isotropic case:
00123   
00124            only eta_x := eta is used and instead of step 1) we have
00125   
00126            1') eta(t+dt/2) = eta(t) + dt/2 * (1/3*Tr P(t) - P_ext)
00127   
00128            furthermore, in step 3) and 5a) L is replaced with V=L^3
00129   
00130          - tetragonal case:
00131   
00132            Lx := L_perp, Ly = Lz := L_par
00133            eta_x := eta_perp, etay := eta_par, etaz unused
00134   
00135            instead of step 1) we have
00136   
00137            1'a) eta_perp(t+dt/2) = eta_perp + dt/2 * V/L_perp * ( P_xx(t) - P_ext)
00138            1'b) eta_par(t+dt/2) = eta_par + dt/2 * V/L_par * ( P_yy(t) + P_zz(t) - 2*P_ext)
00139   
00140            steps 3) and 5a) are split into two sub-steps
00141   
00142            L_perp(i+1) = L_perp(i) + dt/(2*W)*eta_perp
00143            L_par(i+1) = L_par(i) + dt/(4*W)*eta_par
00144       */
00145 
00146       // obtain current stress tensor (diagonal components)
00147       system().computeStress(currP_);
00148 
00149       // obtain current box lengths
00150       Vector lengths  = system().boundary().lengths();
00151       double volume = lengths[0]*lengths[1]*lengths[2];
00152 
00153       double extP = system().boundaryEnsemble().pressure();
00154 
00155       // advance eta(t)->eta(t+dt/2) (step one)
00156       if (mode_ == Orthorhombic) {
00157          double Vdthalf = 1.0/2.0 * volume * dt_;
00158          eta_[0] += Vdthalf/lengths[0] * (currP_[0] - extP);
00159          eta_[1] += Vdthalf/lengths[1] * (currP_[1] - extP);
00160          eta_[2] += Vdthalf/lengths[2] * (currP_[2] - extP);
00161       } else if (mode_ == Tetragonal) { 
00162          double Vdthalf = 1.0/2.0* volume * dt_; 
00163          eta_[0] += Vdthalf/lengths[0]*(currP_[0] - extP);
00164          eta_[1] += Vdthalf/lengths[1]*(currP_[1] + currP_[2] - 2.0*extP);
00165       } else if (mode_ == Cubic) {
00166          eta_[0] += 1.0/2.0*dt_*(1.0/3.0*(currP_[0]+currP_[1]+currP_[2]) - extP);
00167       }
00168 
00169       // update the box length L(t) -> L(t+dt/2) (step three)
00170       // (since we still keep the accelerations a(t) computed for box length L_alpha(t) in memory,
00171       // needed in step two, we can exchange the order of the two steps)
00172       // also pre-calculate L(t+dt) (step 5a, only depends on eta(t) of step one)
00173 
00174       Vector lengthsOld = lengths;
00175       Vector lengthsFinal;
00176       double volumeFinal = 0.0;
00177 
00178       double dthalfoverW = 1.0/2.0*dt_/W_;
00179 
00180       if (mode_ == Orthorhombic) {
00181          lengths[0] += dthalfoverW*eta_[0];
00182          lengths[1] += dthalfoverW*eta_[1];
00183          lengths[2] += dthalfoverW*eta_[2];
00184          lengthsFinal[0] = lengths[0] + dthalfoverW*eta_[0];
00185          lengthsFinal[1] = lengths[1] + dthalfoverW*eta_[1];
00186          lengthsFinal[2] = lengths[2] + dthalfoverW*eta_[2];
00187          volumeFinal = lengthsFinal[0]*lengthsFinal[1]*lengthsFinal[2];
00188       } else if (mode_ == Tetragonal) {
00189          lengths[0] += dthalfoverW*eta_[0];
00190          lengths[1] += (1.0/2.0)*dthalfoverW*eta_[1];
00191          lengths[2] = lengths[1];
00192          lengthsFinal[0] = lengths[0] + dthalfoverW*eta_[0];
00193          lengthsFinal[1] = lengths[1] + (1.0/2.0)*dthalfoverW*eta_[1];
00194          lengthsFinal[2] = lengthsFinal[1];
00195          volumeFinal = lengthsFinal[0]*lengthsFinal[1]*lengthsFinal[2];
00196       } else if (mode_ == Cubic) {
00197          volume += dthalfoverW*eta_[0];
00198          lengths[0] = pow(volume,1.0/3.0); // Lx = Ly = Lz = V^(1/3)
00199          lengths[1] = lengths[0];
00200          lengths[2] = lengths[0];
00201          volumeFinal = volume + dthalfoverW*eta_[0];
00202          lengthsFinal[0] = pow(volumeFinal,1./3.);
00203          lengthsFinal[1] = lengthsFinal[0];
00204          lengthsFinal[2] = lengthsFinal[0];
00205       }
00206          
00207       // update simulation box 
00208       system().boundary().setOrthorhombic(lengthsFinal); 
00209 
00210       
00211       // update particles
00212       Atom* atomPtr;
00213       int   ia;

00215       Vector lengthsFac, dtLengthsFac2;
00216 
00217       for (int i=0; i<3; i++) {

00219          dtLengthsFac2[i] = dt_ * lengthsOld[i]*lengthsOld[i]/lengths[i]/lengths[i];
00220       }

00223          system().begin(iSpecies, molIter);

00225             for (ia=0; ia < molIter->nAtom(); ++ia) {
00226                atomPtr = &molIter->atom(ia);
00227                prefactor = prefactors_[atomPtr->typeId()];
00228 
00229                dv.multiply(atomPtr->force(), prefactor);
00230                vtmp.add(atomPtr->velocity(),dv);
00231 
00232                dr[0] = vtmp[0] * dtLengthsFac2[0];
00233                dr[1] = vtmp[1] * dtLengthsFac2[1];
00234                dr[2] = vtmp[2] * dtLengthsFac2[2];
00235 
00236                rtmp.add(atomPtr->position(), dr);
00237 
00238                rtmp[0] *= lengthsFac[0];
00239                rtmp[1] *= lengthsFac[1];
00240                rtmp[2] *= lengthsFac[2];
00241 
00242                atomPtr->position() = rtmp;
00243 
00244                vtmp[0] /= lengthsFac[0];
00245                vtmp[1] /= lengthsFac[1];
00246                vtmp[2] /= lengthsFac[2];
00247 
00248                atomPtr->velocity() = vtmp;
00249             }
00250          }
00251       }
00252 
00253       #ifndef INTER_NOPAIR
00254       if (!system().pairPotential().isPairListCurrent()) {
00255          system().pairPotential().buildPairList();
00256       } 
00257       #endif
00258 
00259       system().calculateForces();
00260 
00261       /* the second step of the explicitly reversible integrator consists of the following to sub-steps
00262 
00263          6) v(t+dt) = v'' + 1/2 * a(t+dt)*dt
00264          7) eta(t+dt/2) -> eta(t+dt)
00265        */
00266 
00267       // v(t+dt) = v'' + 1/2 * a(t+dt)*dt
00268       for (iSpecies=0; iSpecies < nSpecies; ++iSpecies) {
00269          system().begin(iSpecies, molIter); 
00270          for ( ; molIter.notEnd(); ++molIter)
00271          {
00272             for (ia=0; ia < molIter->nAtom(); ++ia) {
00273                atomPtr = &molIter->atom(ia);
00274                prefactor = prefactors_[atomPtr->typeId()];
00275                dv.multiply(atomPtr->force(), prefactor);
00276                atomPtr->velocity() += dv;
00277             }
00278          }
00279       }
00280 
00281       // now compute pressure tensor with updated virial and velocities
00282       system().computeStress(currP_);
00283 
00284       //  advance eta(t+dt/2) -> eta(t+dt)
00285       if (mode_ == Orthorhombic) {
00286          double Vdthalf = 1.0/2.0 * volumeFinal * dt_;
00287          eta_[0] += Vdthalf/lengthsFinal[0] * (currP_[0] - extP);
00288          eta_[1] += Vdthalf/lengthsFinal[1] * (currP_[1] - extP);
00289          eta_[2] += Vdthalf/lengthsFinal[2] * (currP_[2] - extP);

00291          double Vdthalf = 1.0/2.0 * volumeFinal * dt_;

00293          eta_[1] += Vdthalf/lengthsFinal[1]*(currP_[1] + currP_[2] - 2.0*extP);
