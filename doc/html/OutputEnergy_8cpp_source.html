80  * Dump configuration to file

82  void OutputEnergy::sample(long iStep)
83  {
84  if (isAtInterval(iStep)) {
85  Simulation& sys = simulation();
86  sys.computeKineticEnergy();
87  sys.computePotentialEnergies();
88  if (sys.domain().isMaster()) {
89  double kinetic = sys.kineticEnergy();
90  double pair = sys.pairPotential().energy();
91  double potential = pair;
92  double bond = sys.bondPotential().energy();
93  potential += bond;
94  Log::file() << Int(iStep, 10)
95  << Dbl(kinetic, 15)
96  << Dbl(pair, 15)
97  << Dbl(bond, 15);
98  #ifdef INTER_ANGLE
99  if (sys.nAngleType()) {
100  double angle = sys.anglePotential().energy();
101  potential += angle;
102  Log::file() << Dbl(angle, 15);
103  }
104  #endif
105  #ifdef INTER_DIHEDRAL
106  if (sys.nDihedralType()) {
107  double dihedral = sys.dihedralPotential().energy();
108  potential += dihedral;
109  Log::file() << Dbl(dihedral, 15);
110  }
111  #endif
112  Log::file() << Dbl(kinetic + potential, 20)
113  << std::endl;
114  }
115  ++nSample_;
116  }
117  }
