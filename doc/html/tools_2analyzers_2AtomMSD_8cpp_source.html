00136    * Evaluate end-to-end vectors of all chains, add to ensemble.
00137    */
00138    void AtomMSD::sample(long iStep) 
00139    { 
00140       if (isAtInterval(iStep)) {
00141          Species&  species  = configuration().species(speciesId_);
00142          Boundary& boundary = configuration().boundary();
00143    
00144          // Confirm that nMolecule is positive, and unchanged.
00145          if (nMolecule_ <= 0) {
00146             UTIL_THROW("nMolecule <= 0");
00147          }
00148          if (nMolecule_ != species.size()) {
00149             UTIL_THROW("Number of molecules has changed.");
00150          }
00151    
00152          Vector r;
00153          IntVector shift;
00154          Vector lengths = boundary.lengths();
00155          int i, j;
00156          Species::Iterator iter;
00157    
00158          i = 0;
00159          for (species.begin(iter); iter.notEnd(); ++iter) {
00160    
00161             r = iter->atom(atomId_).position;
00162             boundary.shift(r);
00163    
00164             // Compare current r to previous position, oldPositions_[i]
00165             boundary.distanceSq(r, oldPositions_[i], shift);
00166    
00167             // If a boundary was crossed, increment shift vector
00168             shifts_[i] += shift;
00169    
00170             // Reconstruct true position
00171             for (j = 0; j < Dimension; ++j) {
00172                truePositions_[i][j] = r[j] + shifts_[i][j]*lengths[j];
00173             }
00174    
00175             // Store current position in box for comparison to next one
00176             oldPositions_[i] = r;
00177    
00178             ++i;  
00179          }
00180          accumulator_.sample(truePositions_);
00181       }

00183    }
