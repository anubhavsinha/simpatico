00162    void NvtDpdVvIntegrator::computeDpdForces(bool computeRandom)
00163    {

00165       Vector f;   // force vector 
00166       Vector dv;  // difference in velocities v0 - v1;
00167       double rsq; // square of distance between atoms.
00168       double r;   // distance between atoms.
00169       double fr;  // magnitude of random pair force.
00170       double fd;  // magnitude of dissipative pair force.
00171       double wr;  // weighting function for random forces.
00172       Atom* atom0Ptr;
00173       Atom* atom1Ptr;
00174       PairIterator iter;
00175       int i;
00176 
00177       // Set all dissipative forces to zero.
00178       for (i=0; i < atomCapacity_; ++i) {
00179          dissipativeForces_[i].zero();
00180       }
00181       if (computeRandom) {
00182          for (i=0; i < atomCapacity_; ++i) {
00183             randomForces_[i].zero();
00184          }
00185       }
00186 
00187       // Iterator over atom pairs
00188       for (pairListPtr_->begin(iter); iter.notEnd(); ++iter) {
00189          iter.getPair(atom0Ptr, atom1Ptr);
00190          rsq = boundaryPtr_->distanceSq(atom0Ptr->position(), 
00191                                         atom1Ptr->position(), e);
00192          if (rsq < cutoffSq_) {
00193             r = sqrt(rsq);
00194             e /= r;
00195 
00196             #if MCMD_DPD_TYPE == 0
00197             wr  = 1.0;
00198             #endif
00199             #if MCMD_DPD_TYPE == 1
00200             wr  = 1.0 - (r/cutoff_);
00201             #endif
00202             #if MCMD_DPD_TYPE == 2
00203             wr  = 1.0 - rsq/cutoffSq_;
00204             #endif
00205            
00206             // Add random forces to atomic vectors.
00207             if (computeRandom) {
00208                fr  = sigma_*wr*randomPtr_->gaussian();
00209                f.multiply(e, fr);
00210                randomForces_[atom0Ptr->id()] += f;
00211                randomForces_[atom1Ptr->id()] -= f;
00212             }
00213 
00214             // Add dissipative forces to array
00215             dv.subtract(atom0Ptr->velocity(), atom1Ptr->velocity());
00216             fd = -gamma_*wr*wr*dv.dot(e);
00217             f.multiply(e, fd);
00218             dissipativeForces_[atom0Ptr->id()] += f;
00219             dissipativeForces_[atom1Ptr->id()] -= f;
00220 
00221          }
00222 
00223       }
