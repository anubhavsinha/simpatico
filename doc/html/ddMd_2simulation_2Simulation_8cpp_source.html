00309       int  nSystem = 1;
00310 
00311       // Read command-line arguments
00312       int c;
00313       opterr = 0;
00314       while ((c = getopt(argc, argv, "er:s:")) != -1) {
00315          switch (c) {
00316          case 'e':
00317            eFlag = true;
00318            break;
00319          case 'r':
00320            rFlag = true;
00321            rArg  = optarg;
00322            break;

00324            sFlag = true;

00326            nSystem = atoi(sArg);
00327            break;
00328          case '?':
00329            Log::file() << "Unknown option -" << optopt << std::endl;
00330          }
00331       }
00332 
00333       // If option -s, split the communicator
00334       if (sFlag) {
00335          if (nSystem <= 1) {
00336             UTIL_THROW("nSystem must be greater than 1");
00337          }
00338          int worldRank = communicator_.Get_rank();
00339          int worldSize = communicator_.Get_size();
00340          if (worldSize % nSystem != 0) {
00341             UTIL_THROW("World communicator size not a multiple of nSystem");
00342          }
00343 
00344          // Split the communicator
00345          int systemSize = worldSize/nSystem;
00346          int systemId  = worldRank/systemSize;
00347          communicator_ = communicator_.Split(systemId, worldRank);
00348 
00349          // Set param and grid communicators
00350          setIoCommunicator(communicator_);
00351          domain_.setGridCommunicator(communicator_);
00352          fileMaster().setDirectoryId(systemId);
00353 
00354          // Set log file for processor n to a new file named "n/log".
00355          // Relies on initialization of outputPrefix to empty string "".
00356          fileMaster().openOutputFile("log", logFile_);
00357          Log::setFile(logFile_);
00358       }
00359 
00360       // If option -e, enable echoing of parameters as they are read
00361       if (eFlag) {
00362          ParamComponent::setEcho(true);
00363       }
00364 
00365       // If option -r, load state from a restart file.
00366       if (rFlag) {
00367          if (isIoProcessor()) {
00368             Log::file() << "Begin reading restart, base file name "

00772       #endif

00792       exchanger_.exchange();

01396    {

01398       AtomIterator atomIter;
01399       int i;
01400 
01401       atomStorage_.begin(atomIter);

01539    /*
01540    * Compute total kinetic stress, store on master proc.
01541    *
01542    * Call on all processors.
01543    */
01544    void Simulation::computeKineticStress()
01545    {
01546 
01547       // Do nothing and return if kinetic stress is already set
01548       if (kineticStress_.isSet()) return;
01549 
01550       Tensor localStress;
01551       double  mass;
01552       Vector  velocity;
01553       int typeId, i, j;
01554 
01555       localStress.zero();
01556 
01557       // For each local atoms on this processor, stress(i, j) += m*v[i]*v[j]
01558       AtomIterator atomIter;
01559       atomStorage_.begin(atomIter);
01560       for( ; atomIter.notEnd(); ++atomIter){
01561          typeId = atomIter->typeId();
01562          velocity = atomIter->velocity();
01563          mass = atomTypes_[typeId].mass();
01564          for (i = 0; i < Dimension; ++i) {
01565             for (j = 0; j < Dimension; ++j) {
01566                localStress(i, j) += mass*velocity[i]*velocity[j];
01567             }
01568          }
01569       }
01570 
01571       // Divide dyad sum by system volume
01572       localStress /= boundary().volume();
01573 
01574       #ifdef UTIL_MPI
01575       // Sum values from all processors
01576       Tensor totalStress;
01577       domain_.communicator().Reduce(&localStress(0, 0), &totalStress(0, 0),
01578                                     Dimension*Dimension, MPI::DOUBLE, MPI::SUM, 0);
01579       if (domain_.communicator().Get_rank() != 0) {
01580          totalStress.zero();
01581       }
01582       kineticStress_.set(totalStress);
01583       #else
01584       kineticStress_.set(localStress);
01585       #endif
01586    }
