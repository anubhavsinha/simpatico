112  /*
113  * Output statistical properties to file
114  */
115  void Average::output(std::ostream& out)
116  {
117  // Find first stage (descending) with nSample >= 16
118  AverageStage* ptr = 0;
119  int n = descendants_.size();
120  int i = n;
121  int nSample = 1;
122  while (nSample < 16 && i > 0) {
123  --i;
124  ptr = descendants_[i];
125  nSample = ptr->nSample();
126  }
127 
128  double error = ptr->error();
129  double sigma = error/sqrt(2.0*double(nSample-1));
130  double weight = 1.0/(sigma*sigma);
131  double sum = error*weight;
132  double norm = weight;
133  double aveErr = error;
134  double oldSig;
135 
136  // Find weighted average within plateau
137  bool next = true;
138  while (next && i > 0) {
139  oldSig = sigma;
140  --i;
141  ptr = descendants_[i];
142  error = ptr->error();
143  if (fabs(error - aveErr) < 2.0*oldSig) {
144  nSample = ptr->nSample();
145  sigma = error/sqrt(2.0*double(nSample-1));
146  weight = 1.0/(sigma*sigma);
147  sum += error*weight;
148  norm += weight;
149  aveErr = sum/norm;
150  } else {
151  next = false;
152  }
153  }
154 
155  out << "Average " << Dbl(average())
156  << " +- " << Dbl(aveErr, 9, 2) << std::endl;
157  out << "Variance " << Dbl(variance()) << std::endl;
158  out << "Std Dev " << Dbl(stdDeviation()) << std::endl;
159  out << std::endl;
160 
161  out << "Hierarchichal Error Analysis:" << std::endl;
162  int interval;
163  for (int i = 0; i < n; ++i) {
164  ptr = descendants_[i];
165  error = ptr->error();
166  nSample = ptr->nSample();
167  interval = ptr->stageInterval();
168  if (nSample >= 16) {
169  out << Int(i)
170  << Int(interval)
171  << Dbl(error)
172  << Dbl(error/sqrt(double(nSample)))
173  << Int(nSample) << std::endl;
174  }
175  }
176  out << std::endl;
177  }
