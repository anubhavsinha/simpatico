00123          --i;

00129       double sigma  = error/sqrt(2.0*double(nSample-1));
00130       double weight = 1.0/(sigma*sigma);
00131       double sum    = error*weight;
00132       double norm   = weight;

00134       double oldSig;
00135 
00136       // Find weighted average within plateau
00137       bool next = true;
00138       while (next && i > 0) {
00139          oldSig = sigma;
00140          --i;
00141          ptr = descendants_[i];
00142          error = ptr->error();
00143          if (fabs(error - aveErr) < 2.0*oldSig) {
00144             nSample = ptr->nSample();
00145             sigma  = error/sqrt(2.0*double(nSample-1));
00146             weight = 1.0/(sigma*sigma);
00147             sum   += error*weight;
00148             norm  += weight;
00149             aveErr = sum/norm;
00150          } else {
00151             next = false;
00152          }
00153       }

00156           <<  "  +- " << Dbl(aveErr, 9, 2) << std::endl;
