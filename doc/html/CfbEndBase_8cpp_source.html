192  double &rosenbluth, double &energy)
193  {

195  Vector bondVec;
196  Vector pvtPos = pvtPtr->position();

198  double beta, length;
199  int iTrial;
200 
201  #ifdef INTER_ANGLE
202  Species::AtomAngleArray angles;
203  const Angle *anglePtr;
204  const Atom *pvtPtr2(NULL);
205  Vector dr1, dr2;
206  int iAngle, angleTypeId(0);
207  double rsq1, rsq2, cosTheta;
208  #endif
209 
210  // Generate a random bond length
211  beta = energyEnsemble().beta();
212  length =
213  system().bondPotential().randomBondLength(&random(), beta, bondType);
214 
215  // Loop over nTrial trial positions:
216  rosenbluth = 0.0;
217  for (iTrial=0; iTrial < nTrial_; ++iTrial) {
218  random().unitVector(bondVec);
219  bondVec *= length;
220  // trialPos = pvtPos + bondVec
221  trialPos[iTrial].add(pvtPos, bondVec);
222  boundary().shift(trialPos[iTrial]);
223  endPtr->position() = trialPos[iTrial];
224  #ifndef INTER_NOPAIR
225  trialEnergy[iTrial] = system().pairPotential().atomEnergy(*endPtr);
226  #else
227  trialEnergy[iTrial] = 0.0;
228  #endif
229 
230  #ifdef INTER_ANGLE
231  if (system().hasAnglePotential()) {
232 
233  endPtr->molecule().species().getAtomAngles(*endPtr, angles);
234  for (iAngle = 0; iAngle < angles.size(); ++iAngle) {
235  anglePtr = angles[iAngle];
236  if (&anglePtr->atom(1) == pvtPtr) {
237  if (&anglePtr->atom(0) == endPtr) {
238  pvtPtr2 = &anglePtr->atom(2);
239  } else {
240  pvtPtr2 = &anglePtr->atom(0);
241  }
242  angleTypeId = anglePtr->typeId();
243  }
244  }
245 
246  // Get the angle spanned.
247  rsq1 = boundary().distanceSq(pvtPtr->position(),
248  pvtPtr2->position(), dr1);
249  rsq2 = boundary().distanceSq(endPtr->position(),
250  pvtPtr->position(), dr2);
251  cosTheta = dr1.dot(dr2) / sqrt(rsq1 * rsq2);
252 
253  trialEnergy[iTrial] += system().anglePotential().
254  energy(cosTheta, angleTypeId);
255  }
256  #endif
257 
258  #ifdef INTER_EXTERNAL
259  trialEnergy[iTrial] += system().externalPotential().atomEnergy(*endPtr);
260  #endif
261 
262  trialProb[iTrial] = boltzmann(trialEnergy[iTrial]);
263  rosenbluth += trialProb[iTrial];
264  }
265 
266  // Normalize trial probabilities
267  for (iTrial = 0; iTrial < nTrial_; ++iTrial) {
268  trialProb[iTrial] = trialProb[iTrial]/rosenbluth;
269  }
270 
271  // Choose trial position
272  iTrial = random().drawFrom(trialProb, nTrial_);
273 
274  // Calculate total energy for chosen trial.
275  energy = system().bondPotential().energy(length*length, bondType);
276  energy += trialEnergy[iTrial];
277 
278  // Set position of new end atom to chosen value
279  endPtr->position() = trialPos[iTrial];
280 
281  }
