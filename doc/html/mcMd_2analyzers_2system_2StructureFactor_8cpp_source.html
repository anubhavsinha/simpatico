139  Vector position;

144  int nSpecies, iSpecies, typeId, i, j;

149  for (i = 0; i < nWave_; ++i) {

156  nSpecies = system().simulation().nSpecies();
157  for (iSpecies = 0; iSpecies < nSpecies; ++iSpecies) {
158  system().begin(iSpecies, molIter);
159  for ( ; molIter.notEnd(); ++molIter) {
160  molIter->begin(atomIter);
161  for ( ; atomIter.notEnd(); ++atomIter) {
162  position = atomIter->position();
163  typeId = atomIter->typeId();
164 
165  // Loop over wavevectors
166  for (i = 0; i < nWave_; ++i) {
167 
168  product = position.dot(waveVectors_[i]);
169  expFactor = exp( product*Constants::Im );
170  for (j = 0; j < nMode_; ++j) {
171  fourierModes_(i, j) += modes_(j, typeId)*expFactor;
172  }
173 
174  }
175 
176  }
177  }
178 
179  }
180 
181  // Increment structure factors
182  double volume = system().boundary().volume();
183  double norm;
184  for (j = 0; j < nMode_; ++j) {
185  double maxValue = 0.0;
186  IntVector maxIntVector;
187  double maxQ;
188  for (i = 0; i < nWave_; ++i) {
189  norm = std::norm(fourierModes_(i, j));
190  if (double(norm/volume) >= maxValue) {
191  maxValue = double(norm/volume);
192  maxIntVector = waveIntVectors_[i];
193  maxQ = waveVectors_[i].abs();
194  }
195  structureFactors_(i, j) += norm/volume;
196  }
197 
198  // Output current maximum S(q)
199  outputFile_ << maxIntVector;
200  outputFile_ << Dbl(maxQ,20,8);
201  outputFile_ << Dbl(maxValue,20,8);
202  outputFile_ << std::endl;
203  }
204 
205  ++nSample_;
206 
207  outputFile_ << std::endl;
208  outputFile_.close();
