00099    * Generate, attempt and accept or reject a Monte Carlo move.
00100    *
00101    * Convention for index
00102    *                          nAtom
00103    *          _____________________________________
00104    *          |                                   |
00105    *          0   1   2   3 ...                nAtom-1
00106    *          o---o---o---o---o---o---o---o---o---o
00107    *  bonds:    0   1   2   3                nAtom-1
00108    *
00109    * The algorithm is slightly different from that for linear molecules since a
00110    * ring molecule does not have ends. The operation of moving the atom pointers
00111    * have to account for the periodic boundary conditions, which is achived with
00112    * the help of modId(int s, int n) method.
00113    *
00114    */

00118       Atom     *thisPtr;   // pointer to the current end atom
00119       Atom     *tempPtr;   // pointer to the current end atom
00120       int      *bonds, iBond;
00121       int       nAtom, sign, beginId, endId, i;
00122       double    rosen_r,  rosen_f;
00123       double    energy_r, energy_f;
00124       bool      accept;
00125 
00126       incrementNAttempt();
00127 
00128       // Choose a molecule at random
00129       molPtr = &(system().randomMolecule(speciesId_));
00130       nAtom  = molPtr->nAtom();
00131 
00132       // Require that chain length - 2 >= nRegrow_
00133       if (nRegrow_ > nAtom - 2) {
00134          UTIL_THROW("nRegrow_  >= chain length");
00135       }
00136 
00137       // Allocate bonds array
00138       bonds = new int[nRegrow_ + 1];
00139 
00140       // Choose the beginId of the growing interior bridge
00141       beginId = random().uniformInt(0, nAtom);
00142 
00143       // Choose direction and fill bonds array: sign = +1 if beginId < endId
00144       if (random().uniform(0.0, 1.0) > 0.5) {
00145          sign = +1;
00146          endId = (beginId + (nRegrow_ - 1)) % nAtom;
00147 
00148          // Save bond types
00149          iBond = endId;
00150          for (i = 0; i < nRegrow_ + 1; ++i) {
00151             bonds[i] = molPtr->bond(iBond).typeId();
00152             iBond--;
00153             if (iBond < 0) iBond += nAtom;
00154          }
00155       } else {
00156          sign = -1;
00157          endId = beginId;
00158          beginId = (endId + (nRegrow_ - 1)) % nAtom;
00159 
00160          // Save bond types
00161          iBond = endId - 1;
00162          if (iBond < 0) iBond += nAtom;
00163          for (i = 0; i < nRegrow_ + 1; ++i) {
00164             bonds[i] = molPtr->bond(iBond).typeId();
00165             iBond++;
00166             if (iBond >= nAtom) iBond -= nAtom;
00167          }
00168       }
00169 
00170       // Store current atomic positions from segment to be regrown
00171       tempPtr = &(molPtr->atom(beginId));
00172       for (i = 0; i < nRegrow_; ++i) {
00173          thisPtr = tempPtr + modId(beginId + i*sign, nAtom) - beginId;
00174          oldPos_[i] = thisPtr->position();
00175       }
00176 
00177       // Delete atoms: endId -> beginId
00178       deleteSequence(sign, molPtr, endId, bonds, rosen_r, energy_r);
00179 
00180       // Regrow atoms: endId -> beginId
00181       addSequence(sign, molPtr, beginId, bonds, rosen_f, energy_f);
00182 
00183       // Release bonds array
00184       delete [] bonds;
00185 
00186       // Decide whether to accept or reject
00187       accept = random().metropolis(rosen_f/rosen_r);
00188       if (accept) {
00189 
00190          // Increment counter for accepted moves of this class.
00191          incrementNAccept();
00192 
00193          // If the move is accepted, keep current positions.
00194 
00195       } else {
00196 
00197          // If the move is rejected, restore old positions
00198          tempPtr = &(molPtr->atom(beginId));
00199          for (i = 0; i < nRegrow_; ++i) {
00200             thisPtr = tempPtr + modId(beginId + i*sign, nAtom) - beginId;
00201             //system().moveAtom(*thisPtr, oldPos_[i]);
00202             thisPtr->position() =  oldPos_[i];
00203             #ifndef INTER_NOPAIR
00204             system().pairPotential().updateAtomCell(*thisPtr);
00205             #endif
00206          }
00207    
00208       }
