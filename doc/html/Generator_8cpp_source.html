45  /*
46  * Attempt to place an atom.
47  */
48  bool Generator::attemptPlaceAtom(Atom& atom,
49  const DArray<double>& diameters,
50  CellList& cellList)
51  {
52  // Shift atom position to primary image within boundary
53  boundary().shift(atom.position());
54  Vector& pos = atom.position();
55 
56  // Loop over neighbors, check distance to each.
57  // Return false immediately if any neighbor is too close.
58  CellList::NeighborArray neighbors;
59  cellList.getNeighbors(pos, neighbors);
60  double di = diameters[atom.typeId()];
61  double rSq, dj, dSq;
62  Atom* neighborPtr;
63  int n = neighbors.size();
64  for (int j = 0; j < n; ++j) {
65  neighborPtr = neighbors[j];
66  rSq = boundary().distanceSq(neighborPtr->position(), pos);
67  dj = diameters[neighborPtr->typeId()];
68  dSq = 0.5*(di + dj);
69  dSq *= dSq;
70  if (rSq < dSq) {
71  return false;
72  }
73  }
74 
75  // If all neighbors are allowed, add atom to cellList
76  cellList.addAtom(atom);
77  return true;
78  }

80  /*
81  * Generate random molecules
82  */
83  bool Generator::generate(int nMolecule,
84  const DArray<double>& diameters,
85  CellList& cellList)
86  {
87  // Preconditions
88  UTIL_CHECK(nMolecule <= species().capacity());
89  UTIL_CHECK(cellList.atomCapacity() == simulation().atomCapacity());
90 
91  // Attempt to place all molecules in Species
92  int speciesId = species().id();
93  int maxAttempt = 500;
94  int iAttempt;
95  bool success;
96  for (int iMol = 0; iMol < nMolecule; ++iMol) {
97  Molecule &newMolecule= simulation().getMolecule(speciesId);
98  system().addMolecule(newMolecule);
99  success = false;
100  iAttempt = 0;
101  while (!success && iAttempt < maxAttempt) {
102  success = attemptPlaceMolecule(newMolecule,
103  diameters, cellList);
104  ++iAttempt;
105  }
106  if (!success) {
107  return false;
108  }
109  }
110  return true;
111  }

114  * Setup cell list for use.

116  void
117  Generator::setupCellList(int atomCapacity,
118  Boundary& boundary,
119  const DArray<double>& diameters,
120  CellList& cellList)
121  {
122  // Set maximum atom id = atomCapacity - 1, and allocate
123  // arrays in cellList that are indexed by atom id
124  cellList.setAtomCapacity(atomCapacity);
125 
126  // Compute maximum exclusion diameter
127  double maxDiameter = 0.0;
128  for (int iType = 0; iType < diameters.capacity(); iType++) {
129  if (diameters[iType] > maxDiameter) {
130  maxDiameter = diameters[iType];
131  }
132  }
133 
134  // Setup grid of empty cells
135  cellList.setup(boundary, maxDiameter);
136  }
