00208       /* 
00209       * Implementation Notes:
00210       *
00211       * The pair list is stored in Atom* pointer arrays atom1Ptrs_, atom2Ptrs_, 
00212       * and in an integer array first_. Each element of atom1Ptrs_ contains a 
00213       * pointer to the first atom in a pair (the primary Atom). Each element 
00214       * of atom2Ptrs_ contains a pointer to the second atom in a pair (the 
00215       * secondary Atom).  Secondary atoms that are neighbors of the same 
00216       * primary atom are listed consecutively.  Element first_[i] contains the 
00217       * array index of the first element in atom2Ptrs_ that contains a neighbor 
00218       * of primary atom atom1Ptrs_[i]. Pointers to neighbors of primary atom
00219       * *atom1Ptrs_[i] are thus stored in elements first_[i] <= j < first_[i+1]
00220       * of atom2Ptrs_.  Each pair is included only once.
00221       *
00222       * The only way legal way to loop over all atom pairs, using the public 
00223       * interface of a PairList, is to use a PairListIterator. See the 
00224       * documentation of PairListIterator for a discussion of its usage. The 
00225       * resulting loop over pairs is equivalent to the following double loop 
00226       * over private member variables of PairList:
00227       * 
00228       *    int i, j;
00229       *    Atom* atom1Ptr;
00230       *    Atom* atom2Ptr;
00231       *
00232       *    \\ Loop over primary Atoms
00233       *    for (i = 0; i < atom1Ptrs_.size(); ++i) {
00234       *       atom1Ptr = atom1Ptrs_[i];
00235       *
00236       *       // Loop over secondary atoms
00237       *       for (j = first_[i]; j < first_[i+1]; ++j) {
00238       *           atom2Ptr = atom2Ptrs_[j];
00239       *
00240       *           // ... Do something with Atoms *atom1Ptr and *atom2Ptr.
00241       *
00242       *       }
00243       *    }
00244       *
00245       * Note that GArray<int> first_ contains one more elements than 
00246       * atom1Ptrs_. The element first_[0] is equal to 0, and the last 
00247       * element is always equal to the total number of pairs.
00248       */
00249
