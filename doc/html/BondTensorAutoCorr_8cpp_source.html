00033    /*
00034    * Compute the bond tensor (Call on all processors).
00035    */
00036    void BondTensorAutoCorr::computeData()
00037    {
00038       MPI::Intracomm& communicator = simulation().domain().communicator();
00039       BondStorage& storage = simulation().bondStorage();
00040       Boundary& boundary = simulation().boundary();
00041 
00042       Tensor localTensor;
00043       Vector dr;
00044       double rsq;
00045       GroupIterator<2> iter;
00046       Atom* atom0Ptr;
00047       Atom* atom1Ptr;
00048       int isLocal0, isLocal1, i, j;
00049 
00050 
00051       // Iterate over bonds
00052       localTensor.zero();
00053       storage.begin(iter);
00054       for ( ; iter.notEnd(); ++iter) {
00055          atom0Ptr = iter->atomPtr(0);
00056          atom1Ptr = iter->atomPtr(1);
00057          isLocal0 = !(atom0Ptr->isGhost());
00058          isLocal1 = !(atom1Ptr->isGhost());
00059          rsq = boundary.distanceSq(atom0Ptr->position(), 
00060                                    atom1Ptr->position(), dr);
00061          if (rsq > 1.0E-6) {
00062             dr /= sqrt(rsq);
00063             assert(isLocal0 || isLocal1);
00064             if (isLocal0 && isLocal1) {
00065                // If both atoms are local
00066                for (i = 0; i < Dimension; ++i) {
00067                   for (j = 0; j < Dimension; ++j) {
00068                      localTensor(i, j) += dr[i]*dr[j];
00069                   }
00070                }
00071             } else {
00072                // If one atoms is local and one is a ghost
00073                for (i = 0; i < Dimension; ++i) {
00074                   for (j = 0; j < Dimension; ++j) {
00075                      localTensor(i, j) += 0.5*dr[i]*dr[j];
00076                   }
00077                }
00078             }
00079          }
00080       }
00081 
00082       // Reduce partial sums from all processors, store on the master.
00083       bondTensor_.zero();
00084       communicator.Reduce(&localTensor(0,0), &bondTensor_(0,0), 
00085                           Dimension*Dimension, MPI::DOUBLE, MPI::SUM, 0);
00086 
00087       if (communicator.Get_rank() != 0) {
00088          bondTensor_.zero();
00089       }
00090    }

00095    Tensor BondTensorAutoCorr::data()

00098       double trace = 0.0;

00107 
00108       // Scale traceless symmetric tensor

00110       for (i = 0; i < Dimension; ++i) {
00111          for (j = 0; j < Dimension; ++j) {
00112             bondTensor_(i,j) *= factor;
