146  std::istream& FileMaster::paramFile()
147  {
148  if ((!hasDirectoryId_) || isSetParamFileStdIn_) {
149  return std::cin;
150  } else {
151  if (!paramFilePtr_) {
152 
153  // Construct parameter filename
154  std::string filename(rootPrefix_);
155  if (hasDirectoryId_) {
156  filename += directoryIdPrefix_;
157  }
158  filename += "param";
159 
160  // Open parameter input file
161  std::ifstream* filePtr = new std::ifstream();
162  filePtr->open(filename.c_str());
163  if (filePtr->fail()) {
164  std::string message = "Error opening parameter file. Filename: ";
165  message += filename;
166  UTIL_THROW(message.c_str());
167  }
168 
169  paramFilePtr_ = filePtr;
170  }
171  return *paramFilePtr_;
172  }
173  }
