31  /*
32  * Read parameter(s) and set system parameters.
33  */
34  void Perturbation::readParameters(std::istream& in)
35  {
36  read<int>(in, "mode", mode_);
37  read<int>(in, "nParameters", nParameters_);
38  parameters_.allocate(size_, nParameters_);
39  parameter_.allocate(nParameters_);
40  initialParameter_.allocate(nParameters_);
41  finalParameter_.allocate(nParameters_);
42  if (mode_ == 0) {
43  readDMatrix<double>(in, "parameters", parameters_, size_, nParameters_);
44  for (int i = 0; i < nParameters_; ++i) {
45  parameter_[i] = parameters_(rank_,i);
46  }
47  } else if (mode_ == 1) {
48  readDArray<double>(in, "initialParameter", initialParameter_, nParameters_);
49  readDArray<double>(in, "finalParameter", finalParameter_, nParameters_);
50  int i, j;
51  for (i = 0; i < nParameters_; ++i) {
52  parameters_(0,i) = initialParameter_[i];
53  parameters_(size_-1,i) = finalParameter_[i];
54  for (j= 1; j < size_-1; ++j) {
55  parameters_(j, i) = parameters_(0, i)
56  + j*((parameters_(size_-1, i) - parameters_(0, i))/(size_-1));
57  }
58  }
59  for (i = 0; i < nParameters_; ++i) {
60  parameter_[i] = parameters_(rank_, i);
61  }
62  }
63 
64  setParameter(); // Modify parameter of associated System
65  }
