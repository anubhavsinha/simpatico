00052            parameters_(0,i) = initialParameter_[i];

00055               parameters_(j, i) = parameters_(0, i)

00057            }
00058         }
00059         for (i  = 0; i < nParameters_; ++i) {
00060            parameter_[i] = parameters_(rank_, i);
00061         }
00062       }
