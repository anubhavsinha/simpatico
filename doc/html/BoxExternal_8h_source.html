00200    /* 
00201    * Calculate repulsive potential energy for particle of type i.
00202    *
00203    * 7.905694 = 25/sqrt(10), which is the negative of the minimum of the
00204    * unshifted 9-3 potential.
00205    */
00206    inline double BoxExternal::energy(double d, int type) const
00207    {
00208       if (d < cutoff_) {
00209          double r3i;
00210          r3i = d * d * d;
00211          if (r3i < 0.129615*sigmaCb_) r3i = 0.129615 * sigmaCb_;
00212          r3i = sigmaCb_ / r3i;
00213          return coeff_ * (r3i*(r3i*r3i - 7.5) + 7.905694);
00214       } else {
00215          return 0.0;
00216       }
00217    }
