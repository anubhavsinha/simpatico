48  UTIL_THROW("Atom coordinates are not Cartesian");
49  }
50  } else {
51  if (atomStorage().isCartesian()) {
52  UTIL_THROW("Atom coordinates are Cartesian");

63  // Setup required before main loop (ghosts, forces, etc)
64  // Atomic coordinates are Cartesian on exit from setup.
65  setup();
66  simulation().diagnosticManager().setup();

68  // Main MD loop
69  timer().start();
70  exchanger().timer().start();
71  int beginStep = iStep_;
72  int endStep = iStep_ + nStep;
73  bool needExchange;
74  for ( ; iStep_ < endStep; ++iStep_) {
75 
76  // Atomic coordinates must be Cartesian on entry to loop body.
77  if (!atomStorage().isCartesian()) {
78  UTIL_THROW("Atomic coordinates are not Cartesian");
79  }
80 
81  // Sample scheduled diagnostics.
82  // Also write restart file, if scheduled.
83  if (Diagnostic::baseInterval > 0) {
84  if (iStep_ % Diagnostic::baseInterval == 0) {
85  if (saveInterval() > 0) {
86  if (iStep_ % saveInterval() == 0) {
87  if (iStep_ > beginStep) {
88  simulation().save(saveFileName());
89  }
90  }
91  }

93  }
94  }

96

98  integrateStep1();
99  timer().stamp(INTEGRATE1);
100 
101  // Unset precomputed values of all energies, stresses, etc.
102  simulation().modifySignal().notify();
103 
104  #ifdef DDMD_INTEGRATOR_DEBUG
105  // Sanity check
106  simulation().isValid();
107  #endif
108 
109  // Check if exchange and reneighboring is necessary
110  needExchange = isExchangeNeeded(pairPotential().skin());
111 
112  // Exchange atoms if necessary
113  if (needExchange) {

117  timer().stamp(Integrator::TRANSFORM_F);
118  }

120  timer().stamp(Integrator::EXCHANGE);

123  if (!UTIL_ORTHOGONAL) {
124  atomStorage().transformGenToCart(boundary());
125  timer().stamp(Integrator::TRANSFORM_R);

129  timer().stamp(Integrator::PAIRLIST);

132  timer().stamp(UPDATE);
133  }

136  #ifdef DDMD_INTEGRATOR_DEBUG

138  simulation().isValid();
139  #endif
140

142  // for constant pressure ensemble. Both methods send the modifyForce signal.
143  if (simulation().boundaryEnsemble().isRigid()) {
144  computeForces();
145  } else {
146  computeForcesAndVirial();
147  }
148 
149  // 2nd step of integration, which finishes the velocity update.
150  // This method normally calls simulation().velocitySignal().notify()
151  integrateStep2();
152  timer().stamp(INTEGRATE2);
153 
154  #ifdef DDMD_INTEGRATOR_DEBUG
155  // Sanity check
156  simulation().isValid();
157  #endif
158 
159  }
160  exchanger().timer().stop();
161  timer().stop();
162

164  if (Diagnostic::baseInterval > 0) {
165  if (iStep_ % Diagnostic::baseInterval == 0) {
166  if (saveInterval() > 0) {
167  if (iStep_ % saveInterval() == 0) {
168  simulation().save(saveFileName());
169  }
170  }
171  }
172  }
173 
174  // If !UTIL_ORTHOGONAL, transform to scaled coordinates,
175  // in preparation for beginning of the next run.
176  if (!UTIL_ORTHOGONAL && atomStorage().isCartesian()) {
177  atomStorage().transformCartToGen(boundary());
178  }
179 
180  if (domain().isMaster()) {
181  Log::file() << std::endl;
182  }

184  }
