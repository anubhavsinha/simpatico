42  void Species::initialize()
43  {
44  // Preconditions
45  if (molecules_.capacity() > 0) {
46  UTIL_THROW("Species already initialized");
47  }
48  assert(molecules_.capacity() == 0);
49  assert(atomPtrs_.capacity() == 0);
50  if (capacity_ <= 0) {
51  UTIL_THROW("Capacity <= 0: Value not set");
52  }
53  if (nAtom_ <= 0) {
54  UTIL_THROW("nAtom_ <= 0: Value not set");
55  }
56 
57  // Allocate and initialize atomPtrs_ array
58  atomPtrs_.allocate(capacity_*nAtom_);
59  for (int i=0; i < atomPtrs_.capacity(); ++i) {
60  atomPtrs_[i] = 0;
61  }
62 
63  // Allocate and initialize molecules_ array
64  molecules_.allocate(capacity_);
65  molecules_.resize(capacity_);
66  Atom** atomPtr = &atomPtrs_[0];
67  for (int i=0; i < capacity_; ++i) {
68  molecules_[i].atoms_ = atomPtr;
69  molecules_[i].id_ = i;
70  molecules_[i].nAtom_ = 0;
71  molecules_[i].speciesPtr_ = this;
72  atomPtr += nAtom_;
73  }
74  molecules_.resize(0);
75 
76  }

93  * Add an atom to this Species.
94  */
95  void Species::addAtom(Atom& atom)
96  {
97  if (atom.speciesId != id_) {
98  UTIL_THROW("Inconsistent speciesId");
99  }
100  int mId = atom.moleculeId;
101  int aId = atom.atomId;
102  if (mId < 0) {
103  UTIL_THROW("atom.moleculeId < 0");
104  }
105  if (mId >= capacity_) {
106  UTIL_THROW("atom.moleculeId >= capacity_");
107  }
108  if (aId < 0) {
109  UTIL_THROW("atom.atomId < 0");
110  }
111  if (aId >= nAtom_) {
112  UTIL_THROW("atom.atomId >= nAtom_");
113  }
114  atomPtrs_[mId*nAtom_ + aId] = &atom;
115  if (mId >= molecules_.size()) {
116  molecules_.resize(mId+1);
117  }
118  ++molecules_[mId].nAtom_;
119  }

127  /*
128  * Return true if valid, throw Exception otherwise.
129  */
130  bool Species::isValid() const
131  {
132  const Molecule* mPtr;
133  const Atom* aPtr;
134  int ia, im;
135  for (im = 0; im < molecules_.size(); ++im) {
136  mPtr = &(molecules_[im]);
137  if (mPtr->nAtom_ != nAtom_) {
138  UTIL_THROW("molecule nAtom != species nAtom");
139  }
140  if (mPtr->atoms_ != &atomPtrs_[0] + im*nAtom_) {
141  UTIL_THROW("Incorrect assignment of atoms_ in molecule");
142  }
143  if (mPtr->id() != im) {
144  UTIL_THROW("Incorrect molecule id");
145  }
146  if (mPtr->speciesPtr_ != this) {
147  UTIL_THROW("Incorrect species pointer in molecule");
148  }
149  for (ia = 0; ia < nAtom_; ++ia) {
150  aPtr = mPtr->atoms_[ia];
151  if (aPtr == 0) {
152  UTIL_THROW("Null atom ptr in molecule");
153  }
154  if (aPtr->atomId != ia) {
155  UTIL_THROW("Inconsistent atom index");
156  }
157  if (aPtr->moleculeId != im) {
158  UTIL_THROW("Inconsistent molecule index");
159  }
160  if (aPtr->speciesId != id_) {
161  UTIL_THROW("Inconsistent species index");
162  }
163  }
164  }
165  return true;
166  }
