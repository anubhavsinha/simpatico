00119    * Algorithm: Generate random bond vector and take the angle.

00121    double CosineSqAngle::randomAngle(
00122                          Random *random, double beta, int type) const
00123    {
00124       double Theta, Theta_min;
00125       double U, U_min;
00126       bool chosen = false;
00127       
00128       Theta_min = acos(cosTheta0_[type]);
00129       U_min = energy(cos(Theta_min), type);
00130 
00131       while (chosen == false) {
00132          Theta = random->uniform(0, Constants::Pi);
00133          U = energy(cos(Theta), type);
00134          if (random->uniform() < exp(-beta*(U-U_min))*cos(Theta)/cos(Theta_min))
00135             chosen = true;
00136       }

00138       return Theta;
