506  inline
507  double OrthorhombicBoundary::distanceSq(const Vector &r1, const Vector &r2,
508  IntVector& translate) const
509  {
510  double dr;
511  double norm = 0.0;
512  for (int i = 0; i < Dimension; ++i) {
513  dr = r1[i] - r2[i];
514  if ( fabs(dr) > halfLengths_[i] ) {
515  if (dr > 0.0) {
516  dr -= lengths_[i];
517  translate[i] = -1;
518  } else {
519  dr += lengths_[i];
520  translate[i] = +1;
521  }
522  assert(fabs(dr) <= halfLengths_[i]);
523  } else {
524  translate[i] = 0;
525  }

527  }
