151  for ( ; atomIter.notEnd(); ++atomIter) {
152  position = atomIter->position();
153  coeff = atomTypeCoeffs_[atomIter->typeId()];
154 
155  // Loop over wavevectors
156  for (i = 0; i < nWave_; ++i) {
157 
158  product = position.dot(waveVectors_[i]);
159  expFactor = exp( product*Constants::Im );
160  fourierModes_[i] += (coeff*expFactor);

162  }
163

167  }
