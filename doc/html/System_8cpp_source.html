00604    /*
00605    * Load configuration from an archive.
00606    */
00607    void System::loadConfig(Serializable::IArchive &ar)
00608    {
00609       ar >> boundary();
00610 
00611       Species* speciesPtr;
00612       int nSpecies = simulation().nSpecies();
00613       Molecule* molPtr;
00614       Molecule::AtomIterator atomIter;
00615       int iSpeciesIn, nMoleculeIn;
00616 
00617       for (int iSpecies = 0; iSpecies < nSpecies; ++iSpecies) {
00618          speciesPtr = &simulation().species(iSpecies);
00619          ar >> iSpeciesIn;
00620          if (iSpeciesIn != iSpecies) {
00621             UTIL_THROW("Error: iSpeciesIn != iSpecies");
00622          }
00623          ar >> nMoleculeIn;
00624          for (int iMol = 0; iMol < nMoleculeIn; ++iMol) {
00625             molPtr = &(speciesPtr->reservoir().pop());
00626             addMolecule(*molPtr);
00627             if (molPtr != &molecule(iSpecies, iMol)) {
00628                UTIL_THROW("Molecule index error");
00629             }
00630             molPtr->begin(atomIter); 
00631             for ( ; atomIter.notEnd(); ++atomIter) {
00632                ar >> atomIter->position();
00633                ar >> atomIter->velocity();
00634                #ifdef MCMD_SHIFT
00635                ar >> atomIter->shift();
00636                boundary().shift(atomIter->position(), atom.shift());
00637                #else
00638                boundary().shift(atomIter->position());
00639                #endif
00640             }
00641          }
00642       }
00643    }
