00614    /*
00615    * Load configuration from an archive.
00616    */
00617    void System::loadConfig(Serializable::IArchive &ar)
00618    {
00619       ar >> boundary();
00620 
00621       Species* speciesPtr;
00622       int nSpecies = simulation().nSpecies();
00623       Molecule* molPtr;
00624       Molecule::AtomIterator atomIter;
00625       int iSpeciesIn, nMoleculeIn;
00626 
00627       for (int iSpecies = 0; iSpecies < nSpecies; ++iSpecies) {
00628          speciesPtr = &simulation().species(iSpecies);
00629          ar >> iSpeciesIn;
00630          if (iSpeciesIn != iSpecies) {
00631             UTIL_THROW("Error: iSpeciesIn != iSpecies");
00632          }
00633          ar >> nMoleculeIn;
00634          for (int iMol = 0; iMol < nMoleculeIn; ++iMol) {
00635             molPtr = &(speciesPtr->reservoir().pop());
00636             addMolecule(*molPtr);
00637             if (molPtr != &molecule(iSpecies, iMol)) {
00638                UTIL_THROW("Molecule index error");
00639             }
00640             molPtr->begin(atomIter); 
00641             for ( ; atomIter.notEnd(); ++atomIter) {
00642                ar >> atomIter->position();
00643                ar >> atomIter->velocity();
00644                #ifdef MCMD_SHIFT
00645                ar >> atomIter->shift();
00646                boundary().shift(atomIter->position(), atom.shift());
00647                #else
00648                boundary().shift(atomIter->position());
00649                #endif
00650             }
00651          }
00652       }
00653    }

00659    {
00660       ar << boundary();
00661 
00662       int nSpecies = simulation().nSpecies();
00663       System::MoleculeIterator molIter;
00664       Molecule::AtomIterator   atomIter;
00665       int nMoleculeOut;
00666 
00667       for (int iSpecies = 0; iSpecies < nSpecies; ++iSpecies) {
00668          ar << iSpecies;
00669          nMoleculeOut = nMolecule(iSpecies);
00670          ar << nMoleculeOut;
00671          begin(iSpecies, molIter); 
00672          for ( ; molIter.notEnd(); ++molIter) {
00673             molIter->begin(atomIter); 
00674             for ( ; atomIter.notEnd(); ++atomIter) {
00675                #ifdef MCMD_SHIFT
00676                boundary().shift(atomIter->position(), atomIter->shift());
00677                #else
00678                boundary().shift(atomIter->position());
00679                #endif
00680                ar << atomIter->position();
00681                ar << atomIter->velocity();
00682                #ifdef MCMD_SHIFT
00683                ar << atomIter->shift();
00684                #endif
00685             }
00686          }
00687       }
