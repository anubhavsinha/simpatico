00073    void DpdPair::readParameters(std::istream &in) 
00074    {
00075       // Preconditions
00076       if (nAtomType_ == 0) {
00077          UTIL_THROW( "nAtomType must be set before readParam");
00078       }
00079    
00080       // Read parameters
00081       readCArray2D<double> (in, "epsilon", epsilon_[0], 
00082                             nAtomType_, nAtomType_, MaxAtomType);

00084                            nAtomType_, nAtomType_, MaxAtomType);
00085    
00086       // Calculate sigmaSq_, ce_, cf_, and maxPairCutoff_
00087       int i, j;
00088       maxPairCutoff_ = 0.0;
00089       for (i = 0; i < nAtomType_; ++i) {
00090          for (j = 0; j < nAtomType_; ++j) {
00091             sigmaSq_[i][j]  = sigma_[i][j]*sigma_[i][j];
00092             if (sigma_[i][j] > maxPairCutoff_) {
00093                maxPairCutoff_ = sigma_[i][j];
00094             }
00095             cf_[i][j] = epsilon_[i][j]/sigmaSq_[i][j];
00096             ce_[i][j] = 0.5*cf_[i][j];
00097          }
00098       }
00099   
00100       isInitialized_ = true; 
00101    }
