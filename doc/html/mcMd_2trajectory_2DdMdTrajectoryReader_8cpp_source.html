40  * Open trajectory file and setup to read.
41  */
42  void DdMdTrajectoryReader::open(std::string filename)
43  {
44  // Open trajectory file
45  file_.open(filename.c_str());
46  if (file_.fail()) {
47  std::string message;
48  message= "Error opening trajectory file. Filename: " + filename;
49  UTIL_THROW(message.c_str());
50  }
51 
52  BinaryFileIArchive ar(file_);
53  int nAtom;
54  ar >> nAtom;
55  //std::cout << nAtom << std::endl;
56 
57  // Add all molecules to system and check consistency of nAtom.
58  addMolecules();
59  if (nAtom != nAtomTotal_) {
60  UTIL_THROW("Inconsistent values: nAtom != nAtomTotal_");
61  }
62 
63  // Allocate private array of atomic positions_
64  if (!positions_.isAllocated()) {
65  positions_.allocate(nAtomTotal_);
66  } else {
67  if (nAtomTotal_ != positions_.capacity()) {
68  UTIL_THROW("Inconsistent values of atom capacity");
69  }
70  }

73  /*
74  * Read frame, return false if end-of-file
75  */
76  bool DdMdTrajectoryReader::readFrame()
77  {
78  // Preconditions
79  if (!positions_.isAllocated()) {
80  UTIL_THROW("positions_ array is not allocated");
81  }
82 
83  BinaryFileIArchive ar(file_);
84 
85  // Attempt to read iStep
86  long iStep = -1;
87  ar >> iStep;
88 
89  // Return false if read failed, indicating end of file.
90  if (file_.eof()) {
91  return false;
92  }
93 
94  // Read boundary dimensions
95  ar >> boundary();
96 
97  // Loop over atoms, read atomic positions
98  Vector r;
99  double h = 1.0/(double(UINT_MAX) + 1.0);
100  int id, i, j;
101  unsigned int ir;
102  for (i = 0; i < nAtomTotal_; ++i) {
103  ar >> id;
104  for (j = 0; j < Dimension; ++j) {
105  ar >> ir;
106  r[j] = ir*h;
107  }
108  boundary().transformGenToCart(r, positions_[id]);
109  }
110 
111  // Assign atom positions, assuming ordered atom ids
112  int iSpecies, iMol;
113  Species *speciesPtr;
114  Molecule::AtomIterator atomIter;
115  Molecule *molPtr;
116  id = 0;
117  for (iSpecies = 0; iSpecies < simulation().nSpecies(); ++iSpecies) {
118  speciesPtr = &simulation().species(iSpecies);
119  for (iMol = 0; iMol < speciesPtr->capacity(); ++iMol) {
120  molPtr = &system().molecule(iSpecies, iMol);
121  for (molPtr->begin(atomIter); atomIter.notEnd(); ++atomIter) {
122  atomIter->position() = positions_[id];
123  id++;
124  }
125  }
126  }
127 
128  return true;
129  }
