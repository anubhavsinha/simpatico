00075    /*
00076    * Compute and output pair energies at regular intervals.
00077    */
00078    void OutputPairEnergies::sample(long iStep) 
00079    {
00080       if (isAtInterval(iStep))  {
00081          Simulation& sys = simulation();
00082          sys.computePairEnergies();
00083          if (sys.domain().isMaster()) {
00084             DMatrix<double> pair = sys.pairEnergies();
00085             for (int i = 0; i < simulation().nAtomType(); ++i){
00086                for (int j = 0; j < simulation().nAtomType(); ++j){
00087                   pair(i,j) = 0.5*( pair(i,j)+pair(j,i) );
00088                   pair(j,i) = pair(i,j);
00089                }
00090             }
00091             outputFile_ << Int(iStep, 10);
00092             for (int i = 0; i < simulation().nAtomType(); ++i){
00093                for (int j = 0; j < simulation().nAtomType(); ++j){
00094                   outputFile_ << Dbl(pair(i,j), 20);
00095                }
00096             }
00097             outputFile_  << std::endl;
00098          }
00099 
00100          ++nSample_;
00101       }
00102    }
