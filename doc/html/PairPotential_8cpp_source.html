157  /*
158  * Build the cell list.
159  */
160  void PairPotential::buildCellList()
161  {
162  if (storage().isCartesian()) {
163  UTIL_THROW("Coordinates are Cartesian entering buildCellList");
164  }
165 
166  // Set cutoff and domain bounds.
167  Vector cutoffs;
168  Vector lower;
169  Vector upper;
170  for (int i = 0; i < Dimension; ++i) {
171  cutoffs[i] = cutoff_/boundaryPtr_->length(i);
172  lower[i] = domain().domainBound(i, 0);
173  upper[i] = domain().domainBound(i, 1);
174  }
175  cellList_.makeGrid(lower, upper, cutoffs, nCellCut_);
176  cellList_.clear();
177 
178  // Add all atoms to the cell list.
179  AtomIterator atomIter;
180  storage().begin(atomIter);
181  for ( ; atomIter.notEnd(); ++atomIter) {
182  cellList_.placeAtom(*atomIter);
183  }
184 
185  // Add all ghosts to the cell list.
186  GhostIterator ghostIter;
187  storage().begin(ghostIter);
188  for ( ; ghostIter.notEnd(); ++ghostIter) {
189  cellList_.placeAtom(*ghostIter);
190  }
191 
192  // Finalize cell list
193  cellList_.build();
194 
195  // Postconditions
196  assert(cellList_.isValid());
197  assert(cellList_.nAtom() + cellList_.nReject()
198  == storage().nAtom() + storage().nGhost());
199  if (storage().isCartesian()) {
200  UTIL_THROW("Coordinates are Cartesian exiting buildCellList");
201  }
202  }

273  int PairPotential::nPairList(double cutoffSq)
274  {
275  Vector f;
276  double rsq;
277  PairIterator iter;
278  Atom* atom0Ptr;
279  Atom* atom1Ptr;
280  int count = 0; // atom counter
281  if (reverseUpdateFlag()) {
282  for (pairList_.begin(iter); iter.notEnd(); ++iter) {
283  iter.getPair(atom0Ptr, atom1Ptr);
284  assert(!atom0Ptr->isGhost());
285  f.subtract(atom0Ptr->position(), atom1Ptr->position());
286  rsq = f.square();
287  if (rsq < cutoffSq) {
288  count += 2;
289  }
290  }
291  } else {
292  for (pairList_.begin(iter); iter.notEnd(); ++iter) {
293  iter.getPair(atom0Ptr, atom1Ptr);
294  assert(!atom0Ptr->isGhost());
295  f.subtract(atom0Ptr->position(), atom1Ptr->position());
296  rsq = f.square();
297  if (rsq < cutoffSq) {
298  if (!atom1Ptr->isGhost()) {
299  count += 2;
300  } else {
301  count += 1;
302  }
303  }
304  }

310  * Count number pairs using cell list (private).
311  */
312  int PairPotential::nPairCell(double cutoffSq)
313  {
314  // Find all neighbors (cell list)
315  Cell::NeighborArray neighbors;
316  Vector f;
317  double rsq;
318  Atom* atomPtr0;
319  Atom* atomPtr1;
320  const Cell* cellPtr;
321  int na, nn, i, j;
322  int count = 0;
323 
324  // Iterate over linked list of local cells.
325  cellPtr = cellList_.begin();
326  while (cellPtr) {
327  cellPtr->getNeighbors(neighbors, reverseUpdateFlag());
328  na = cellPtr->nAtom();
329  nn = neighbors.size();
330  for (i = 0; i < na; ++i) {
331  atomPtr0 = neighbors[i]->ptr();
332 
333  // Loop over atoms in this cell
334  for (j = 0; j < na; ++j) {
335  atomPtr1 = neighbors[j]->ptr();
336  if (atomPtr1 > atomPtr0) {
337  f.subtract(atomPtr0->position(), atomPtr1->position());
338  rsq = f.square();
339  if (rsq < cutoffSq) {
340  count += 2;
341  }
342  }
343  }
344 
345  // Loop over atoms in neighboring cells.
346  if (reverseUpdateFlag()) {
347  for (j = na; j < nn; ++j) {
348  atomPtr1 = neighbors[j]->ptr();
349  f.subtract(atomPtr0->position(), atomPtr1->position());
350  rsq = f.square();
351  if (rsq < cutoffSq) {
352  count += 2;
353  }
354  }
355  } else {
356  for (j = na; j < nn; ++j) {
357  atomPtr1 = neighbors[j]->ptr();
358  f.subtract(atomPtr0->position(), atomPtr1->position());
359  rsq = f.square();
360  if (rsq < cutoffSq) {
361  if (!atomPtr1->isGhost()) {
362  count += 2;
363  } else {
364  count += 1;
365  }
366  }
367  }
368  }
369 
370  }
371  cellPtr = cellPtr->nextCellPtr();
372  } // while (cellPtr)
373  return count;
374  }

379  int PairPotential::nPairNSq(double cutoffSq)
380  {
381  Vector f;
382  double rsq;
383  AtomIterator atomIter0, atomIter1;
384  GhostIterator ghostIter;
385  int id0, id1;
386  int count = 0;
387 
388  // Iterate over local atom 0
389  storage().begin(atomIter0);
390  for ( ; atomIter0.notEnd(); ++atomIter0) {
391  id0 = atomIter0->id();
392 
393  // Iterate over local atom 1
394  storage().begin(atomIter1);
395  for ( ; atomIter1.notEnd(); ++atomIter1) {
396  id1 = atomIter1->id();
397  if (id0 < id1) {
398  if (!atomIter0->mask().isMasked(id1)) {
399  f.subtract(atomIter0->position(), atomIter1->position());
400  rsq = f.square();
401  if (rsq < cutoffSq) {
402  count += 2;
403  }
404  }
405  }
406  }
407 
408  // Iterate over ghost atoms
409  storage().begin(ghostIter);
410  if (reverseUpdateFlag()) {
411  for ( ; ghostIter.notEnd(); ++ghostIter) {
412  id1 = ghostIter->id();
413  if (id0 < id1) {
414  if (!atomIter0->mask().isMasked(id1)) {
415  f.subtract(atomIter0->position(), ghostIter->position());
416  rsq = f.square();
417  if (rsq < cutoffSq) {
418  count += 2;
419  }
420  }
421  }
422  }
423  } else {
424  for ( ; ghostIter.notEnd(); ++ghostIter) {
425  id1 = ghostIter->id();
426  if (!atomIter0->mask().isMasked(id1)) {
427  f.subtract(atomIter0->position(), ghostIter->position());
428  rsq = f.square();
429  if (rsq < cutoffSq) {
430  count += 1;
431  }
432  }
433  }
434  }
435 
436  }
437  return count;
