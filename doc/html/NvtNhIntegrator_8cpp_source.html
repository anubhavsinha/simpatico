00131    *
00132    * This implements a reversible Velocity-Verlet MD NVT integrator step.
00133    *
00134    * Reference: Winkler, Kraus, and Reineker, J. Chem. Phys. 102, 9018 (1995).
00135    */
00136    void NvtNhIntegrator::step()

00142       double  prefactor;
00143       double  factor;
00144       Molecule::AtomIterator atomIter;
00145       int  iSpecies, nSpecies;
00146       int  nAtom;
00147 
00148       T_target_ = energyEnsemblePtr_->temperature();
00149       nSpecies  = simulation().nSpecies();
00150       nAtom     = system().nAtom();
00151 
00152       factor = exp(-dtHalf*(xi_ + xiDot_*dtHalf));
00153 
00154       // 1st half velocity Verlet, loop over atoms 
00155       for (iSpecies = 0; iSpecies < nSpecies; ++iSpecies) {
00156          system().begin(iSpecies, molIter); 
00157          for ( ; molIter.notEnd(); ++molIter) {
00158 
00159             molIter->begin(atomIter); 
00160             for ( ; atomIter.notEnd(); ++atomIter) {
00161 
00162                atomIter->velocity() *= factor;
00163 
00164                prefactor = prefactors_[atomIter->typeId()];
00165                dv.multiply(atomIter->force(), prefactor);
00166                //dv.multiply(atomIter->force(), dtHalf);
00167 
00168                atomIter->velocity() += dv;
00169                dr.multiply(atomIter->velocity(), dt_);
00170 
00171                atomIter->position() += dr;

00173             }
00174 
00175          }
00176       }
00177 
00178       // First half of update of xi_
00179       xi_ += xiDot_*dtHalf;
00180 
00181       #ifndef INTER_NOPAIR
00182       // Rebuild the pair list if necessary
00183       if (!system().pairPotential().isPairListCurrent()) {
00184          system().pairPotential().buildPairList();
00185       }
00186       #endif
00187 
00188       system().calculateForces();
00189 
00190       // 2nd half velocity Verlet, loop over atoms
00191       for (iSpecies=0; iSpecies < nSpecies; ++iSpecies) {
00192          system().begin(iSpecies, molIter); 
00193          for ( ; molIter.notEnd(); ++molIter) {
00194             for (molIter->begin(atomIter); atomIter.notEnd(); ++atomIter) {
00195                prefactor = prefactors_[atomIter->typeId()];
00196                dv.multiply(atomIter->force(), prefactor);
00197                atomIter->velocity() += dv;
00198                atomIter->velocity() *=factor;
00199             }
00200          }
00201       }
00202 
00203       // Update xiDot and complete update of xi_
00204       T_kinetic_ = system().kineticEnergy()*2.0/double(3*nAtom);
00205       xiDot_     = (T_kinetic_/T_target_ -1.0)*nuT_*nuT_;
00206       xi_       += xiDot_*dtHalf;
