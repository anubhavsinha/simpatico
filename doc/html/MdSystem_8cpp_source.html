00609    /*
00610    * Subtract average velocity from all atomic velocities.
00611    */
00612    Vector MdSystem::removeDriftVelocity()
00613    {
00614       Vector average(0.0);
00615       MoleculeIterator molIter;
00616       Molecule::AtomIterator atomIter;
00617       int    nSpec = simulation().nSpecies();
00618       int    nAtom = 0;
00619       int    iSpec;
00620 
00621       for (iSpec = 0; iSpec < nSpec; ++iSpec) {
00622          for (begin(iSpec, molIter); molIter.notEnd(); ++molIter) {
00623             for (molIter->begin(atomIter); atomIter.notEnd(); ++atomIter) {
00624                average += atomIter->velocity();
00625                ++nAtom;
00626             }
00627          }
00628       }
00629       average /= double(nAtom);
00630       for (iSpec = 0; iSpec < nSpec; ++iSpec) {
00631          for (begin(iSpec, molIter); molIter.notEnd(); ++molIter) {
00632             for (molIter->begin(atomIter); atomIter.notEnd(); ++atomIter) {
00633                atomIter->velocity() -= average;
00634             }
00635          }
00636       }
00637 
00638       return average;
00639    }
