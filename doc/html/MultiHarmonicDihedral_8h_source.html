208  bool status;
209  status = torsion.computeAngle(b1, b2, b3); // computes cosPhi
210  if (!status) {
211  double c = torsion.cosPhi;
212  const Parameter* p = &parameters_[type];
213  return (p->a0 + c*(p->a1 + c*(p->a2 + c*(p->a3 + c*p->a4))));
214  } else {
215  return 0.0;
216  }
217  }
