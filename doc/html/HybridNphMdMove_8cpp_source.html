95  /*
96  * Generate, attempt and accept or reject a Hybrid MD/MC move.
97  */
98  bool HybridNphMdMove::move()
99  {
100  System::MoleculeIterator molIter;
101  Molecule::AtomIterator atomIter;
102  double oldEnergy, newEnergy;
103  int iSpec;
104  int nSpec = simulation().nSpecies();
105 
106  bool accept;
107 
108  if (nphIntegratorPtr_ == NULL) {
109  UTIL_THROW("null integrator pointer");
110  }
111  // Increment counter for attempted moves
112  incrementNAttempt();
113 
114  // Store old boundary lengths.
115  Vector oldLengths = system().boundary().lengths();
116 
117  // Store old atom positions in oldPositions_ array.
118  for (iSpec = 0; iSpec < nSpec; ++iSpec) {
119  mdSystemPtr_->begin(iSpec, molIter);
120  for ( ; molIter.notEnd(); ++molIter) {
121  for (molIter->begin(atomIter); atomIter.notEnd(); ++atomIter) {
122  oldPositions_[atomIter->id()] = atomIter->position();
123  }
124  }
125  }
126 
127  // Initialize MdSystem
128  #ifndef INTER_NOPAIR
129  mdSystemPtr_->pairPotential().buildPairList();
130  #endif
131  mdSystemPtr_->calculateForces();
132  mdSystemPtr_->setBoltzmannVelocities(energyEnsemble().temperature());
133  nphIntegratorPtr_->setup();
134 
135  // generate integrator variables from a Gaussian distribution
136  Random& random = simulation().random();
137 
138  double temp = system().energyEnsemble().temperature();
139 
140  double volume = system().boundary().volume();
141 
142  if (mode_ == Cubic) {
143  // one degree of freedom
144  // barostat_energy = 1/2 (1/W) eta_x^2
145  double sigma = sqrt(temp/barostatMass_);
146  nphIntegratorPtr_->setEta(0, sigma*random.gaussian());
147  } else if (mode_ == Tetragonal) {
148  // two degrees of freedom
149  // barostat_energy = 1/2 (1/W) eta_x^2 + 1/2 (1/(2W)) eta_y^2
150  double sigma1 = sqrt(temp/barostatMass_);
151  nphIntegratorPtr_->setEta(0, sigma1*random.gaussian());
152  double sigma2 = sqrt(temp/barostatMass_/2.0);
153  nphIntegratorPtr_->setEta(1, sigma2*random.gaussian());
154  } else if (mode_ == Orthorhombic) {
155  // three degrees of freedom
156  // barostat_energy = 1/2 (1/W) (eta_x^2 + eta_y^2 + eta_z^2)
157  double sigma = sqrt(temp/barostatMass_);
158  nphIntegratorPtr_->setEta(0, sigma*random.gaussian());
159  nphIntegratorPtr_->setEta(1, sigma*random.gaussian());
160  nphIntegratorPtr_->setEta(2, sigma*random.gaussian());
161  }
162 
163  // Store old energy
164  oldEnergy = mdSystemPtr_->potentialEnergy();
165  oldEnergy += mdSystemPtr_->kineticEnergy();
166  oldEnergy += system().boundaryEnsemble().pressure()*volume;
167  oldEnergy += nphIntegratorPtr_->barostatEnergy();
168 
169  // Run a short MD simulation
170  for (int iStep = 0; iStep < nStep_; ++iStep) {
171  nphIntegratorPtr_->step();
172  }
173 
174  volume = system().boundary().volume();
175 
176  // Calculate new energy
177  newEnergy = mdSystemPtr_->potentialEnergy();
178  newEnergy += mdSystemPtr_->kineticEnergy();
179  newEnergy += system().boundaryEnsemble().pressure()*volume;
180  newEnergy += nphIntegratorPtr_->barostatEnergy();
181 
182  // Decide whether to accept or reject
183  accept = random.metropolis( boltzmann(newEnergy-oldEnergy) );
184 
185  // Accept move
186  if (accept) {
187 
188  #ifndef INTER_NOPAIR
189  // Rebuild the McSystem cellList using the new positions.
190  system().pairPotential().buildCellList();
191  #endif
192 
193  // Increment counter for the number of accepted moves.
194  incrementNAccept();
195 
196  } else {
197 
198  // Restore old boundary lengths
199  system().boundary().setOrthorhombic(oldLengths);
200 
201  // Restore old atom positions
202  for (iSpec = 0; iSpec < nSpec; ++iSpec) {
203  mdSystemPtr_->begin(iSpec, molIter);
204  for ( ; molIter.notEnd(); ++molIter) {
205  molIter->begin(atomIter);
206  for ( ; atomIter.notEnd(); ++atomIter) {
207  atomIter->position() = oldPositions_[atomIter->id()];
208  }
209  }
210  }
211 
212  }
213 
214  return accept;
215 
216  }
