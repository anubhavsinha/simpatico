46  void DdMdConfigIo::read(std::istream &in)
47  {
48  // Calculate atom and bond capacities for entire simulation
49  int atomCapacity = 0;
50  int bondCapacity = 0;
51  int nSpecies = simulation().nSpecies();
52  int speciesCapacity = 0;
53  int iSpec;
54  Species* speciesPtr;
55  for (iSpec = 0; iSpec < nSpecies; ++iSpec) {
56  speciesPtr = &simulation().species(iSpec);
57  speciesCapacity = speciesPtr->capacity();
58  atomCapacity += speciesCapacity*speciesPtr->nAtom();
59  bondCapacity += speciesCapacity*speciesPtr->nBond();
60  }
61 
62  // Read boundary
63  in >> Label("BOUNDARY");
64  in >> system().boundary();
65 
66  // Read atomic positions
67  // Atom tags must appear in order, numbered from 0
68  // Number of atoms must match atomCapacity.
69  in >> Label("ATOMS");
70  int nAtom;
71  in >> Label("nAtom") >> nAtom;
72  if (nAtom != atomCapacity) {
73  UTIL_THROW("nAtom != atomCapacity");
74  }
75  Molecule* molPtr;
76  Molecule::AtomIterator atomIter;
77  int iMol, iAtom, nMolecule, atomId, atomTypeId;
78  int sId, mId, aId;
79  for (iSpec=0; iSpec < nSpecies; ++iSpec) {
80  speciesPtr = &simulation().species(iSpec);
81  nMolecule = speciesPtr->capacity();
82  for (iMol = 0; iMol < nMolecule; ++iMol) {
83  molPtr = &(speciesPtr->reservoir().pop());
84  system().addMolecule(*molPtr);
85 
86  // Read positions
87  iAtom = 0;
88  for (molPtr->begin(atomIter); atomIter.notEnd(); ++atomIter) {
89 
90  in >> atomId >> atomTypeId;
91  if (atomId != atomIter->id()) {
92  UTIL_THROW("Atom tags not ordered");
93  }
94  if (hasMolecules_) {
95  in >> sId >> mId >> aId;
96  if (sId != iSpec) {
97  UTIL_THROW("Invalid species id");
98  }
99  if (mId != iMol) {
100  UTIL_THROW("Invalid molecule id");
101  }
102  if (aId != iAtom) {
103  UTIL_THROW("Invalid local atom id");
104  }
105  }
106 
107  in >> atomIter->position();
108  in >> atomIter->velocity();
109 
110  ++iAtom;
111  }
112  }
113  }
