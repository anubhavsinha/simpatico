80  void LinkMaster::removeLink(int id)
81  {
82  Link* linkPtr = &link(id);
83  int atom0Id;
84  int atom1Id;
85 
86  if (!link(id).isActive()) {
87  UTIL_THROW("Attempt to remove a nonactive link");
88  }
89  atom0Id = linkPtr->atom0().id();
90  atom1Id = linkPtr->atom1().id();
91  // Remove link from atom0 and atom1 link sets
92  if (!atomLinkSets_[atom0Id].isElement(*linkPtr)) {
93  UTIL_THROW("Link is not in atomLinkSets of atom0");
94  }
95  if (!atomLinkSets_[atom1Id].isElement(*linkPtr)) {
96  UTIL_THROW("Link is not in atomLinkSets of atom1");
97  }
98  atomLinkSets_[atom0Id].remove(*linkPtr);
99  atomLinkSets_[atom1Id].remove(*linkPtr);
100 
101  // Notify observers of removal of this Link.
102  // Notify before clearing Link so atomPtrs and typeId are available.
103  LinkRemoveEvent event(linkPtr);
104  Notifier<LinkRemoveEvent>::notifyObservers(event);
105 
106  // Clear the link: nullify atomPtrs, set typeId = -1, isActive = false
107  linkPtr->clear();
108 
109  // Return link to reservoir
110  linkSet_.remove(*linkPtr);
111  reservoir_.push(*linkPtr);
112

229  // Set tags for all Links.

245  /*
246  * Return true if this LinkMaster is valid, or throw an Exception.
247  */
248  bool LinkMaster::isValid() const
249  {
250 
251  if (links_.isAllocated()) {
252 
253  if (links_.capacity() != reservoir_.capacity()) {
254  UTIL_THROW("Unequal capacities for links_ and reservoir_");
255  }
256  if (links_.capacity() != reservoir_.size() + linkSet_.size()) {
257  UTIL_THROW("links_ capacity != reservoir_ + linkSet_ sizes");
258  }
259 
260  // Check consistency of all links in linkSet
261  const Link* linkPtr;
262  const Atom* atom0Ptr;
263  const Atom* atom1Ptr;
264  int i;
265  for (i = 0; i < linkSet_.size(); ++i) {
266 
267  linkPtr = &(linkSet_[i]);
268  atom0Ptr = &(linkPtr->atom0());
269  atom1Ptr = &(linkPtr->atom1());
270 
271  // Check:
272 
273  // that neither atom0Ptr nor atom1Ptr is null
274  if (atom0Ptr == 0 || atom1Ptr == 0) {
275  UTIL_THROW("Link with one or both atoms missing");
276  }
277 
278  // that atom0Ptr and atom1Ptr are distinct
279  if (atom0Ptr == atom1Ptr) {
280  UTIL_THROW("Link connects an atom to itself");
281  }
282 
283  // that the link is active
284  if (!linkPtr->isActive()) {
285  UTIL_THROW("Link in linkSet is not Active");
286  }
287 
288  // that the Link is in the AtomLinkSet of atom0 and of atom1
289  if (!atomLinkSets_[atom0Ptr->id()].isElement(*linkPtr)) {
290  UTIL_THROW("Link is not in atomLinkSets of atom0");
291  }
292  if (!atomLinkSets_[atom1Ptr->id()].isElement(*linkPtr)) {
293  UTIL_THROW("Link is not in atomLinkSets of atom1");
294  }
295 
296  }
297 
298  // Count all active links in links_ array
299  int nCount = 0;
300  for (i = 0; i < linkCapacity_; ++i) {
301  linkPtr = &links_[i];
302  if (linkPtr->isActive()) {
303  ++nCount;
304  }
305  }
306  if (nCount != linkSet_.size()) {
307  UTIL_THROW("Number of Links with atoms != linkSet.size()");
308  }
309 
310  // Count all links int AtomLinkSets_ (each appears twice)
311  nCount = 0;
312  for (i = 0; i < atomCapacity_; ++i) {
313  nCount += atomLinkSets_[i].size();
314  }
315  if (nCount != 2*linkSet_.size()) {
316  UTIL_THROW("Number links in atom sets != 2*linkSet.size()");
317  }
318 
319  }
320 
321  return true;
322  }
