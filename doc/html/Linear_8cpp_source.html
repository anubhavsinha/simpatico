116  bool Linear::tryPlaceAtom(Molecule& molecule, int atomId,
117  DArray<double> exclusionRadius, System& system, CellList &cellList,
118  BondPotential *bondPotentialPtr, const Boundary &boundary)
119  {
120  Atom& lastAtom = molecule.atom(atomId);
121  Atom& thisAtom = molecule.atom(++atomId);
122  Random& random = system.simulation().random();
123 
124  bool hasBeenPlaced = false;
125 
126  for (int iAttempt = 0; iAttempt < maxPlacementAttempts_; iAttempt++)
127  {
128  // draw a random bond vector
129  int beta = 1;
130  Vector v;
131  random.unitVector(v);
132  v *= bondPotentialPtr->randomBondLength(&random, beta,
133  calculateBondTypeId(lastAtom.indexInMolecule()));
134 
135  Vector newPos;
136  newPos = lastAtom.position();
137  newPos += v;
138  // shift into simulation cell
139  boundary.shift(newPos);
140 
141  // check if the atom can be placed at the new position
142  CellList::NeighborArray neighbors;
143  cellList.getNeighbors(newPos, neighbors);
144  int nNeighbor = neighbors.size();
145  bool canBePlaced = true;
146  for (int j = 0; j < nNeighbor; ++j) {
147  Atom *jAtomPtr = neighbors[j];
148 
149  double r = sqrt(boundary.distanceSq(
150  jAtomPtr->position(), newPos));
151  if (r < (exclusionRadius[thisAtom.typeId()] +
152  exclusionRadius[jAtomPtr->typeId()])) {
153  canBePlaced = false;
154  break;
155  }
156  }
157  if (canBePlaced) {
158  // place the particle
159  thisAtom.position() = newPos;
160 
161  // add to cell list
162  cellList.addAtom(thisAtom);
163 
164  // are we add the end of the chain?
165  if (atomId == molecule.nAtom()-1)
166  return true;
167 
168  // recursion step
169  if (! tryPlaceAtom(molecule, atomId, exclusionRadius, system,
170  cellList, bondPotentialPtr, boundary) ) {
171  // If the next monomer cannot be inserted, delete this monomer
172  // again
173  cellList.deleteAtom(thisAtom);
174  } else {
175  hasBeenPlaced = true;
176  break;
177  }
178  }
179  }
180 
181  return hasBeenPlaced;
182  }
