49  // Preconditions

62  #endif

64  if (hasDihedrals_) {
65  if (nAtom() > 3) {

72  }
73  }
74  }
75  #endif

77  int i;

79  allocate();
80 
81  // Set Atom Type Ids
82  for (i = 0; i < nAtom(); ++i) {
83  setAtomType(i, calculateAtomTypeId(i));
84  }
85 
86  // Build Bonds
87  for (i = 0; i < nBond(); ++i) {
88  makeBond(i, i, i+1, calculateBondTypeId(i));
89  }

114  bool Linear::tryPlaceAtom(Molecule& molecule, int atomId,
115  DArray<double> exclusionRadius, System& system, CellList &cellList,
116  BondPotential *bondPotentialPtr, const Boundary &boundary)
117  {
118  Atom& lastAtom = molecule.atom(atomId);
119  Atom& thisAtom = molecule.atom(++atomId);
120  Random& random = system.simulation().random();
121 
122  bool hasBeenPlaced = false;
123 
124  for (int iAttempt = 0; iAttempt < maxPlacementAttempts_; iAttempt++)
125  {
126  // draw a random bond vector
127  int beta = 1;
128  Vector v;
129  random.unitVector(v);
130  v *= bondPotentialPtr->randomBondLength(&random, beta,
131  calculateBondTypeId(lastAtom.indexInMolecule()));
132 
133  Vector newPos;
134  newPos = lastAtom.position();
135  newPos += v;
136  // shift into simulation cell
137  boundary.shift(newPos);
138 
139  // check if the atom can be placed at the new position
140  CellList::NeighborArray neighbors;
141  cellList.getNeighbors(newPos, neighbors);
142  int nNeighbor = neighbors.size();
143  bool canBePlaced = true;
144  for (int j = 0; j < nNeighbor; ++j) {
145  Atom *jAtomPtr = neighbors[j];
146 
147  double r = sqrt(boundary.distanceSq(
148  jAtomPtr->position(), newPos));
149  if (r < (exclusionRadius[thisAtom.typeId()] +
150  exclusionRadius[jAtomPtr->typeId()])) {
151  canBePlaced = false;
152  break;
153  }
154  }
155  if (canBePlaced) {
156  // place the particle
157  thisAtom.position() = newPos;
158 
159  // add to cell list
160  cellList.addAtom(thisAtom);
161 
162  // are we add the end of the chain?
163  if (atomId == molecule.nAtom()-1)
164  return true;
165 
166  // recursion step
167  if (! tryPlaceAtom(molecule, atomId, exclusionRadius, system,
168  cellList, bondPotentialPtr, boundary) ) {
169  // If the next monomer cannot be inserted, delete this monomer
170  // again
171  cellList.deleteAtom(thisAtom);
172  } else {
173  hasBeenPlaced = true;
174  break;
175  }
176  }
177  }
178 
179  return hasBeenPlaced;
180  }

185  void Linear::generateMolecules(int nMolecule,
186  DArray<double> exclusionRadius, System& system,
187  BondPotential *bondPotentialPtr, const Boundary &boundary)
188  {
189  int iMol;
190 
191  // Set up a cell list with twice the maxium exclusion radius as the
192  // cell size
193  double maxExclusionRadius = 0.0;
194  for (int iType = 0; iType < system.simulation().nAtomType(); iType++) {
195  if (exclusionRadius[iType] > maxExclusionRadius)
196  maxExclusionRadius = exclusionRadius[iType];
197  }
198 
199  // the minimum cell size is twice the maxExclusionRadius,
200  // but to save memory, we take 2 times that value
201  CellList cellList;
202  cellList.allocate(system.simulation().atomCapacity(),
203  boundary, 2.0*2.0*maxExclusionRadius);
204 
205  if (nMolecule > capacity())
206  UTIL_THROW("nMolecule > Species.capacity()!");
207 
208  Simulation& sim = system.simulation();
209  for (iMol = 0; iMol < nMolecule; ++iMol) {
210  // Add a new molecule to the system
211  Molecule &newMolecule= sim.getMolecule(id());
212  system.addMolecule(newMolecule);
213 
214  // Try placing atoms
215  bool moleculeHasBeenPlaced = false;
216  for (int iAttempt = 0; iAttempt< maxPlacementAttempts_; iAttempt++) {
217  // Place first atom
218  Vector pos;
219  system.boundary().randomPosition(system.simulation().random(),pos);
220  Atom &thisAtom = newMolecule.atom(0);
221 
222  // check if the first atom can be placed at the new position
223  CellList::NeighborArray neighbors;
224  cellList.getNeighbors(pos, neighbors);
225  int nNeighbor = neighbors.size();
226  bool canBePlaced = true;
227  for (int j = 0; j < nNeighbor; ++j) {
228  Atom *jAtomPtr = neighbors[j];
229 
230  double r = sqrt(system.boundary().distanceSq(
231  jAtomPtr->position(), pos));
232  if (r < (exclusionRadius[thisAtom.typeId()] +
233  exclusionRadius[jAtomPtr->typeId()])) {
234  canBePlaced = false;
235  break;
236  }
237  }
238  if (canBePlaced) {
239  thisAtom.position() = pos;
240  cellList.addAtom(thisAtom);
241 
242  // Try to recursively place other atoms
243  if (tryPlaceAtom(newMolecule, 0, exclusionRadius, system,
244  cellList, bondPotentialPtr, system.boundary())) {
245  moleculeHasBeenPlaced = true;
246  break;
247  } else {
248  cellList.deleteAtom(thisAtom);
249  }
250  }
251  }
252  if (! moleculeHasBeenPlaced) {
253  std::ostringstream oss;
254  oss << "Failed to place molecule " << newMolecule.id();
255  UTIL_THROW(oss.str().c_str());
