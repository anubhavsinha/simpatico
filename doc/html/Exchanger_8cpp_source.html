156  /*
157  * Initialize group ghost communication plan, before exchanging atoms.
158  *
159  * This method is called by exchangeAtoms, after computing plans for
160  * exchanging atoms, based on their position, but before exchanging
161  * atoms, and before clearing ghosts from previous exchange.
162  *
163  * Algorithm: Loop over all Group<N> objects in the storage, identify
164  * groups that span boundaries of the processor domain associated with
165  * each of 6 transfer directions (3 Cartesian directions, and transfer
166  * "up" and "down" in each direction). This requires information about
167  * positions of ghost as well as local atoms. For each boundary of the
168  * domain, identify atoms whose positions are "inside" and "outside".
169  * Count ghost atoms very near the boundary as both inside and outside,
170  * for saftey. If a group has atoms both inside and outside a domain
171  * boundary, it is marked for sending in the associated communication
172  * step.
173  *
174  * After calculating ghost communication plan for each group, clear
175  * the pointers to all ghost atoms in the group. The exchangeAtoms
176  * method will clear the actual ghost atoms from the AtomStorage.
177  */
178  template <int N>
179  void Exchanger::initGroupGhostPlan (GroupStorage<N>& storage)
180  {
181  double coordinate;
182  GroupIterator<N> groupIter;
183  Atom* atomPtr;
184  //int nEx[2];
185  int nIn;
186  int nOut;
187  int i, j, k;
188  bool choose;
189 
190  // Loop over groups
191  storage.begin(groupIter);
192  for ( ; groupIter.notEnd(); ++groupIter) {
193 
194  // Compute ghost communication plan for group
195  groupIter->plan().clearFlags();
196  for (i = 0; i < Dimension; ++i) {
197  if (multiProcessorDirection_[i]) {
198  for (j = 0; j < 2; ++j) {
199  choose = false;
200  nIn = 0;
201  nOut = 0;
202  // nEx[j] = 0;
203  // Loop over atoms in group
204  for (k = 0; k < N; ++k) {
205  atomPtr = groupIter->atomPtr(k);
206  if (atomPtr) {
207  coordinate = atomPtr->position()[i];
208  if (atomPtr->isGhost()) {
209  if (j == 0) {
210  assert(inner_(i, j) > bound_(i, j));
211  if (coordinate < inner_(i, j)) {
212  ++nOut;
213  //if (coordinate < bound_(i, j)) {
214  // nEx[j] += 1;
215  //}
216  }
217  if (coordinate > outer_(i, j)) {
218  ++nIn;
219  }
220  } else { // if j = 1
221  assert(inner_(i, j) < bound_(i, j));
222  if (coordinate > inner_(i, j)) {
223  ++nOut;
224  //if (coordinate > bound_(i, j)) {
225  // nEx[j] += 1;
226  //}
227  }
228  if (coordinate < outer_(i, j)) {
229  ++nIn;
230  }
231  }
232  } else { // if atomPtr points to local atom
233  if (atomPtr->plan().exchange(i, j)) {
234  ++nOut;
235  //nEx[j] += 1;
236  } else {
237  ++nIn;
238  }
239  }
240  } else { // if atomPtr is null
241  choose = true;
242  break;
243  }
244  } // end for k (atoms in group)
245  if (nOut > 0 && nIn > 0) {
246  choose = true;
247  }
248  if (choose) {
249  groupIter->plan().setGhost(i, j);
250  } else {
251  groupIter->plan().clearGhost(i, j);
252  }
253  } // end for j = 0, 1
254 
255  #if 0
256  if (nEx[0] > 0 && nEx[1] > 0) {
257  UTIL_THROW("Group spanning 3 nodes");
258  }
259  #endif
260 
261  } // end if multiProcessorDirection
262  } // end for i (Cartesian axes)
263 
264  // Clear pointers to all ghost atoms in this group
265  for (k = 0; k < N; ++k) {
266  atomPtr = groupIter->atomPtr(k);
267  if (atomPtr) {
268  if (atomPtr->isGhost()) {
269  groupIter->clearAtomPtr(k);
270  }
271  }
272  }
273 
274  }

293  Atom* atomPtr;
294  int k, nAtom;

296 
297  bufferPtr_->beginSendBlock(Buffer::GROUP, N);
298  emptyGroups.clear();
299  storage.begin(groupIter);
300  for ( ; groupIter.notEnd(); ++groupIter) {

302  nAtom = 0;

380  /*
381  * Set ghost communication flags for all atoms in incomplete groups.
382  *
383  * Precondition: This is called by exchangeAtoms after exchanging atoms
384  * and groups between neighboring processors. All ghosts are cleared.
385  *
386  * Algorithm: Loop over all Group<N> objects in the group storage.
387  * For each group, check if the group is incomplete, implying that one or
388  * more atoms in the group are owned by another processor. If the group
389  * is incomplete, loop over 6 transfer directions. For each direction,
390  * if the group is marked for sending in that direction, set the ghost
391  * ghost communication flag for transfer in that direction for every
392  * local atom in the group.
393  *
394  * Note: The algorithm assumes that the ghost communication flag for each
395  * atom within a group whose atoms are divided among two or more processors
396  * will be set on the processor that owns the atom.
397  */
398  template <int N>
399  void Exchanger::finishGroupGhostPlan(GroupStorage<N>& storage)
400  {
401  GroupIterator<N> groupIter;
402  Atom* atomPtr;
403  Plan* planPtr;
404  int i, j, k, nAtom;
405 
406  // Loop over groups
407  storage.begin(groupIter);
408  for ( ; groupIter.notEnd(); ++groupIter) {
409 
410  #ifdef UTIL_DEBUG
411  #ifdef DDMD_EXCHANGER_DEBUG
412  // Validate group
413  int atomId;
414  nAtom = 0;
415  for (k = 0; k < N; ++k) {
416  atomPtr = groupIter->atomPtr(k);
417  atomId = groupIter->atomId(k);
418  if (atomPtr != 0) {
419  if (atomPtr != atomStoragePtr_->find(atomId)) {
420  UTIL_THROW("Error in atom pointer in bond");
421  }
422  if (atomPtr->isGhost()) {
423  UTIL_THROW("Pointer to ghost atom in bond");
424  } else {
425  ++nAtom;
426  }
427  } else { // if atomPtr == 0
428  atomPtr = atomStoragePtr_->find(atomId);
429  if (atomPtr) {
430  if (!atomPtr->isGhost()) {
431  UTIL_THROW("Missing pointer to local atom in bond");
432  }
433  }
434  }
435  }
436  assert(nAtom == groupIter->nPtr());
437  if (nAtom == 0) {
438  UTIL_THROW("Empty group");
439  }
440  #endif // ifdef DDMD_EXCHANGER_DEBUG
441  #endif // ifdef UTIL_DEBUG
442 
443  // If this group is incomplete, set ghost flags for atoms
444  nAtom = groupIter->nPtr();
445  if (nAtom < N) {
446  for (i = 0; i < Dimension; ++i) {
447  if (multiProcessorDirection_[i]) {
448  for (j = 0; j < 2; ++j) {
449  if (groupIter->plan().ghost(i, j)) {
450  for (k = 0; k < N; ++k) {
451  atomPtr = groupIter->atomPtr(k);
452  if (atomPtr) {
453  assert(!atomPtr->isGhost());
454  planPtr = &atomPtr->plan();
455  if (!planPtr->ghost(i, j)) {
456  planPtr->setGhost(i, j);
457  sendArray_(i,j).append(*atomPtr);
458  }
459  }
460  }
461  }
462  }
463  }
464  }
465  }
466 
467  }
468  }

470  /*
471  * Exchange ownership of local atoms and groups, set ghost plan.
472  *
473  * Atomic coordinates must be in generalized coordinates on entry and
474  * exit if not UTIL_ORTHOGONAL, and Cartesian if UTIL_ORTHOGONAL.
475  *
476  * Algorithm:
477  *
478  * - Loop over local atoms, set exchange and ghost communication
479  * flags for those beyond or near processor domain boundaries.
480  *
481  * - Add local atoms that will be retained by this processor but
482  * sent as ghosts to appropriate send arrays.
483  *
484  * - Call initGroupPlan each type of group (bond, angle, dihedral).
485  * initGroupPlan<N>{
486  *
487  * For each group{
488  * - Set ghost communication flags for groups that span
489  * or may span boundaries.
490  * - Clear pointers to ghost atoms in the group.
491  * }
492  * }
493  *
494  * Clear all ghosts from the AtomStorage
495  *
496  * For each transfer directions (i and j) {
497  *
498  * For each local atom {
499  * if marked for exchange(i, j) {
500  * if gridDimension[i] > 1 {
501  * - add to sendAtoms array for removal
502  * - pack into send buffer
503  * } else {
504  * shift position to apply periodic b.c.
505  * }
506  * }
507  * }
508  *
509  * if gridDimension[i] > 1 {
510  * - Call packGroups for each group type.
511  * This packs groups containing atoms that are sent.
512  * - Remove exchanged atoms and empty groups
513  * - Send and receive data buffers
514  * for each atom in the receive buffer {
515  * - Unpack atom into AtomStorage
516  * - shift periodic boundary conditions
517  * - Determine if this is new home (or way station)
518  * - If atom is home, add to appropriate ghost arrays.
519  * }
520  * - Call unpackGroups for each group type.
521  * }
522  *
523  * }
524  *
525  * - Call finishGroupPlan<N> each type of group (bond, angle, dihedral).
526  * finishGroupPlan<N> {
527  * for each group{
528  * if group is incomplete{
529  * for each direction (i and j) {
530  * if group is marked for ghost communication {
531  * mark local atoms in group for ghost communication
532  * }
533  * }
534  * }
535  * }
536  * }
537  *
538  * }
539  *
540  * Postconditions. Upon return:
541  * Each processor owns all atoms in its domain.
542  * Each processor owns all atoms containing one or more local atoms.
543  * Ghost plans are set for all local atoms.
544  * Send arrays contain local atoms marked for sending as ghosts.
545  * The AtomStorage contains no ghost atoms.
546  */
547  void Exchanger::exchangeAtoms()
548  {
549  stamp(START);
550  Vector lengths = boundaryPtr_->lengths();
551  double bound, inner, slabWidth;
552  double coordinate, rshift;
553  AtomIterator atomIter;
554  Atom* atomPtr;
555  Plan* planPtr;
556  int i, j, jc, ip, jp, k, source, dest, nSend;
557  int shift;
558  bool isHome;
559  bool isGhost;
560 
561  // Set domain and slab boundaries
562  for (i = 0; i < Dimension; ++i) {
563  if (UTIL_ORTHOGONAL) {
564  slabWidth = pairCutoff_;
565  } else {
566  slabWidth = pairCutoff_/lengths[i];
567  }
568  for (j = 0; j < 2; ++j) {
569  // j = 0 sends to lower coordinate i, bound is minimum
570  // j = 1 sends to higher coordinate i, bound is maximum
571  bound = domainPtr_->domainBound(i, j);
572  bound_(i, j) = bound;
573  if (j == 0) { // Communicate with lower index
574  inner_(i,j) = bound + slabWidth;
575  outer_(i, j)= bound - slabWidth;
576  } else { // j == 1, communicate with upper index
577  inner_(i, j) = bound - slabWidth;
578  outer_(i, j) = bound + slabWidth;
579  }
580  sendArray_(i, j).clear();
581  }
582  if (domainPtr_->grid().dimension(i) > 1) {
583  multiProcessorDirection_[i] = 1;
584  } else {
585  multiProcessorDirection_[i] = 0;
586  }
587  }
588 
589  // Compute communication plan for every local atom
590  atomStoragePtr_->begin(atomIter);
591  for ( ; atomIter.notEnd(); ++atomIter) {
592 
593  planPtr = &atomIter->plan();
594  planPtr->clearFlags();
595  isHome = true;
596  isGhost = false;
597 
598  // Cartesian directions
599  for (i = 0; i < Dimension; ++i) {
600 
601  coordinate = atomIter->position()[i];
602 
603  // Transmission direction
604  for (j = 0; j < 2; ++j) {
605 
606  // j = 0 sends to lower coordinate i
607  // j = 1 sends to higher coordinate i
608 
609  // Index for conjugate (reverse) direction
610  if (j == 0) jc = 1;
611  if (j == 1) jc = 0;
612 
613  if (j == 0) { // Communicate with lower index
614  if (coordinate < bound_(i, j)) {
615  planPtr->setExchange(i, j);
616  if (multiProcessorDirection_[i]) {
617  isHome = false;
618  }
619  if (coordinate > outer_(i, j)) {
620  planPtr->setGhost(i, jc);
621  isGhost = true;
622  }
623  } else {
624  if (coordinate < inner_(i, j)) {
625  planPtr->setGhost(i, j);
626  isGhost = true;
627  }
628  }
629  } else { // j == 1, communicate with upper index
630  if (coordinate > bound_(i, j)) {
631  planPtr->setExchange(i, j);
632  if (multiProcessorDirection_[i]) {
633  isHome = false;
634  }
635  if (coordinate < outer_(i, j)) {
636  planPtr->setGhost(i, jc);
637  isGhost = true;
638  }
639  } else {
640  if (coordinate > inner_(i, j)) {
641  planPtr->setGhost(i, j);
642  isGhost = true;
643  }
644  }
645  }
646 
647  } // end for j
648  } // end for i
649 
650  // Add atoms that will be retained by this processor,
651  // but will be communicated as ghosts to sendArray_
652  if (isGhost && isHome) {
653  for (i = 0; i < Dimension; ++i) {
654  for (j = 0; j < 2; ++j) {
655  if (planPtr->ghost(i, j)) {
656  sendArray_(i, j).append(*atomIter);
657  }
658  }
659  }
660  }
661 
662  } // end atom loop, end compute plan
663  stamp(ATOM_PLAN);
664 
665  /*
666  * Find groups that span boundaries (uses information about ghosts).
667  * Clear pointers to ghosts in each Group after inspecting the Group.
668  *
669  * Atoms in Group<N> objects that span boundaries, and are incomplete
670  * after atom migration, are marked for sending as ghosts in the
671  * finishGroupGhostPlan<N> function, further below.
672  */
673  initGroupGhostPlan<2>(*bondStoragePtr_); // bonds
674  #ifdef INTER_ANGLE
675  initGroupGhostPlan<3>(*angleStoragePtr_); // angles
676  #endif
677  #ifdef INTER_DIHEDRAL
678  initGroupGhostPlan<4>(*dihedralStoragePtr_); // dihedrals
679  #endif
680  stamp(INIT_GROUP_PLAN);
681 
682  // Clear all ghost atoms from AtomStorage
683  atomStoragePtr_->clearGhosts();
684  stamp(CLEAR_GHOSTS);
685 
686  #ifdef UTIL_DEBUG
687  #ifdef DDMD_EXCHANGER_DEBUG
688  int nAtomTotal;
689  atomStoragePtr_->computeNAtomTotal(domainPtr_->communicator());
690  int myRank = domainPtr_->gridRank();
691  if (myRank == 0) {
692  nAtomTotal = atomStoragePtr_->nAtomTotal();
693  }
694  bondStoragePtr_->isValid(*atomStoragePtr_,
695  domainPtr_->communicator(), false);
696  #endif
697  #endif
698 
699  // Cartesian directions for exchange (0=x, 1=y, 2=z)
700  for (i = 0; i < Dimension; ++i) {
701 
702  // Transmission direction
703  // j = 0 sends to processor with lower grid coordinate i
704  // j = 1 sends to processor with higher grid coordinate i
705  for (j = 0; j < 2; ++j) {
706 
707  // Index for conjugate (reverse) direction
708  if (j == 0) jc = 1;
709  if (j == 1) jc = 0;
710 
711  source = domainPtr_->sourceRank(i, j); // rank to receive from
712  dest = domainPtr_->destRank(i, j); // rank to send to
713  bound = domainPtr_->domainBound(i, j); // bound for send
714  inner = inner_(i, jc); // inner bound upon receipt
715  shift = domainPtr_->shift(i, j); // shift for periodic b.c.
716  if (UTIL_ORTHOGONAL) {
717  rshift = lengths[i]*double(shift);
718  } else {
719  rshift = double(shift);
720  }
721 
722  #ifdef UTIL_MPI
723  if (multiProcessorDirection_[i]) {
724  bufferPtr_->clearSendBuffer();
725  bufferPtr_->beginSendBlock(Buffer::ATOM);
726  }
727  #endif
728 
729  // Choose atoms for sending, pack and mark for removal.
730  sentAtoms_.clear();
731  atomStoragePtr_->begin(atomIter);
732  for ( ; atomIter.notEnd(); ++atomIter) {
733 
734  #ifdef UTIL_DEBUG
735  coordinate = atomIter->position()[i];
736  #ifdef DDMD_EXCHANGER_DEBUG
737  {
738  bool choose;
739  if (j == 0) {
740  choose = (coordinate < bound);
741  } else {
742  choose = (coordinate > bound);
743  }
744  assert(choose == atomIter->plan().exchange(i, j));
745  }
746  #endif
747  #endif
748

753

757  } else

760 
761  #ifdef UTIL_DEBUG

764  assert(coordinate > -1.0*fabs(rshift));
765  assert(coordinate < 2.0*fabs(rshift));
766  #endif
767  #endif
768

770  if (shift) {
771  atomIter->position()[i] += rshift;
772  }
773 
774  #ifdef UTIL_DEBUG
775  coordinate = atomIter->position()[i];
776  assert(coordinate >= domainPtr_->domainBound(i, 0));
777  assert(coordinate < domainPtr_->domainBound(i, 1));
778  #endif
779

781  // The following assertion applies to these.
782  assert(!atomIter->plan().ghost(i, j));
783 
784  #if UTIL_DEBUG
785  // Check ghost communication plan
786  if (j == 0 && atomIter->position()[i] > inner) {
787  assert(atomIter->plan().ghost(i, 1));
788  } else
789  if (j == 1 && atomIter->position()[i] < inner) {
790  assert(atomIter->plan().ghost(i, 0));
791  }
792  #endif
793 
794  }
795  }
796 
797  } // end atom loop
798  stamp(PACK_ATOMS);
799 
800  #ifdef UTIL_MPI
801  // Send and receive only if processor grid dimension(i) > 1
802  if (multiProcessorDirection_[i]) {
803 
804  // End atom send block
805  bufferPtr_->endSendBlock();
806 
807  // Pack groups that contain postmarked atoms.
808  packGroups<2>(i, j, *bondStoragePtr_, emptyBonds_);
809  #ifdef INTER_ANGLE
810  packGroups<3>(i, j, *angleStoragePtr_, emptyAngles_);
811  #endif
812  #ifdef INTER_DIHEDRAL
813  packGroups<4>(i, j, *dihedralStoragePtr_, emptyDihedrals_);
814  #endif
815  stamp(PACK_GROUPS);
816

820  */

823  nSend = sentAtoms_.size();

825  atomStoragePtr_->removeAtom(&sentAtoms_[k]);

830  removeEmptyGroups<2>(*bondStoragePtr_, emptyBonds_);
831  #ifdef INTER_ANGLE
832  removeEmptyGroups<3>(*angleStoragePtr_, emptyAngles_);
833  #endif
834  #ifdef INTER_DIHEDRAL
835  removeEmptyGroups<4>(*dihedralStoragePtr_, emptyDihedrals_);
836  #endif
837  stamp(REMOVE_GROUPS);

840  bufferPtr_->sendRecv(domainPtr_->communicator(),
841  source, dest);

843

845  bufferPtr_->beginRecvBlock();

851  atomStoragePtr_->addNewAtom();
852

854  atomPtr->position()[i] += rshift;
855  }
856 
857  #ifdef UTIL_DEBUG
858  // Check bounds on atom coordinate
859  coordinate = atomPtr->position()[i];
860  assert(coordinate > domainPtr_->domainBound(i, 0));
861  assert(coordinate < domainPtr_->domainBound(i, 1));
862 
863  // Check ghost plan
864  assert(!planPtr->ghost(i, j));
865  if (j == 0) {
866  assert(planPtr->ghost(i, 1) == (coordinate > inner) );
867  } else {
868  assert(planPtr->ghost(i, 0) == (coordinate < inner) );
869  }
870  #endif
871 
872  // Determine if new atom will stay on this processor.
873  isHome = true;
874  if (i < Dimension - 1) {
875  for (ip = i + 1; ip < Dimension; ++ip) {
876  if (multiProcessorDirection_[ip]) {
877  for (jp = 0; jp < 2; ++jp) {
878  if (planPtr->exchange(ip, jp)) {
879  isHome = false;
880  }
881  }
882  }
883  }
884  }
885 
886  // If atom will stay, add to sendArrays for ghosts
887  if (isHome) {
888  for (ip = 0; ip < Dimension; ++ip) {
889  for (jp = 0; jp < 2; ++jp) {
890  if (planPtr->ghost(ip, jp)) {
891  sendArray_(ip, jp).append(*atomPtr);
892  }
893  }
894  }
895  }
896 
897  }
898  assert(bufferPtr_->recvSize() == 0);
899  stamp(UNPACK_ATOMS);
900 
901  // Unpack groups
902  unpackGroups<2>(*bondStoragePtr_);
903  #ifdef INTER_ANGLE
904  unpackGroups<3>(*angleStoragePtr_);
905  #endif
906  #ifdef INTER_DIHEDRAL
907  unpackGroups<4>(*dihedralStoragePtr_);
908  #endif
909  stamp(UNPACK_GROUPS);
910 
911  } // end if gridDimension > 1
912  #endif // ifdef UTIL_MPI
913 
914  } // end for j (direction 0, 1)
915  } // end for i (Cartesian index)
916

918  * At this point:
919  * No ghost atoms exist.
920  * All atoms are on correct processor.
921  * No Groups are empty.
922  * All pointer to local atoms in Groups are set.
923  * All pointers to ghost atoms in Groups are null.
924  */
925 
926  #ifdef UTIL_DEBUG
927  #ifdef DDMD_EXCHANGER_DEBUG
928  // Validity checks
929  atomStoragePtr_->computeNAtomTotal(domainPtr_->communicator());
930  if (myRank == 0) {
931  assert(nAtomTotal = atomStoragePtr_->nAtomTotal());
932  }
933  atomStoragePtr_->isValid();
934  bondStoragePtr_->isValid(*atomStoragePtr_,
935  domainPtr_->communicator(), false);
936  #ifdef INTER_ANGLE
937  angleStoragePtr_->isValid(*atomStoragePtr_,
938  domainPtr_->communicator(), false);
939  #endif
940  #ifdef INTER_DIHEDRAL
941  dihedralStoragePtr_->isValid(*atomStoragePtr_,
942  domainPtr_->communicator(), false);
943  #endif
944  #endif // ifdef DDMD_EXCHANGER_DEBUG
945  #endif // ifdef UTIL_DEBUG
946 
947  // Set ghost communication flags for atoms in incomplete groups
948  finishGroupGhostPlan<2>(*bondStoragePtr_);
949  #ifdef INTER_ANGLE
950  finishGroupGhostPlan<3>(*angleStoragePtr_);
951  #endif
952  #ifdef INTER_DIHEDRAL
953  finishGroupGhostPlan<4>(*dihedralStoragePtr_);
954  #endif
955 
956  stamp(FINISH_GROUP_PLAN);
957  }

981  *

989

991  assert(bufferPtr_->isInitialized());

993  assert(domainPtr_->hasBoundary());
994  if (atomStoragePtr_->nGhost() != 0) {
995  UTIL_THROW("atomStoragePtr_->nGhost() != 0");
996  }

1002  int i, j, jc, ip, jp, k, source, dest, shift, size;
1003 
1004  // Clear all receive arrays
1005  for (i = 0; i < Dimension; ++i) {
1006  for (j = 0; j < 2; ++j) {
1007  recvArray_(i, j).clear();
1008  }
1009  }
1010 
1011  #ifdef UTIL_DEBUG
1012  double coordinate;
1013  #ifdef DDMD_EXCHANGER_DEBUG
1014  // Check send arrays
1015  {

1017  FMatrix<int, Dimension, 2> sendCounter;

1037  for (i = 0; i < Dimension; ++i) {
1038  for (j = 0; j < 2; ++j) {

1040  UTIL_THROW("Incorrect sendArray size");

1042  }
1043  }
1044  }
1045  #endif // ifdef DDMD_EXCHANGER_DEBUG
1046  #endif // ifdef UTIL_DEBUG

1050  for (i = 0; i < Dimension; ++i) {
1051 
1052  // Transmit directions
1053  for (j = 0; j < 2; ++j) {
1054 
1055  // j = 0: Send ghosts near minimum bound to lower coordinate
1056  // j = 1: Send ghosts near maximum bound to higher coordinate
1057 
1058  // Set index for reverse direction
1059  if (j == 0) jc = 1;
1060  if (j == 1) jc = 0;

1062  // Shift on receiving node for periodic b.c.s
1063  shift = domainPtr_->shift(i, j);
1064  if (UTIL_ORTHOGONAL) {
1065  rshift = lengths[i]*shift;
1066  } else {
1067  rshift = 1.0*shift;
1068  }

1071  if (multiProcessorDirection_[i]) {
1072  bufferPtr_->clearSendBuffer();
1073  bufferPtr_->beginSendBlock(Buffer::GHOST);
1074  }
1075  #endif
1076

1078  size = sendArray_(i, j).size();
1079  for (k = 0; k < size; ++k) {
1080 
1081  sendPtr = &sendArray_(i, j)[k];
1082 
1083  #ifdef UTIL_MPI
1084  if (multiProcessorDirection_[i]) {
1085 
1086  // If grid dimension > 1, pack atom for sending
1087  bufferPtr_->packGhost(*sendPtr);
1088 
1089  } else
1090  #endif
1091  { // if grid dimension == 1
1092

1094  atomPtr = atomStoragePtr_->newGhostPtr();
1095  recvArray_(i, j).append(*atomPtr);
1096  atomPtr->setId(sendPtr->id());
1097  atomPtr->setTypeId(sendPtr->typeId());
1098  atomPtr->plan().setFlags(sendPtr->plan().flags());
1099  atomPtr->position() = sendPtr->position();
1100  if (shift) {
1101  atomPtr->position()[i] += rshift;
1102  }
1103  atomStoragePtr_->addNewGhost();
1104 
1105  #ifdef UTIL_DEBUG
1106  // Validate shifted ghost coordinate
1107  coordinate = atomPtr->position()[i];
1108  if (j == 0) {
1109  assert(coordinate > bound_(i, 1));
1110  } else {
1111  assert(coordinate < bound_(i, 0));
1112  }
1113  #endif
1114 
1115  // Add to send arrays for any remaining directions
1116  if (i < Dimension - 1) {
1117  for (ip = i + 1; ip < Dimension; ++ip) {
1118  for (jp = 0; jp < 2; ++jp) {
1119  if (atomPtr->plan().ghost(ip, jp)) {
1120  sendArray_(ip, jp).append(*atomPtr);
1121  }
1122  }
1123  }
1124  }
1125 
1126  }
1127 
1128  }
1129  stamp(PACK_GHOSTS);
1130 
1131  #ifdef UTIL_MPI
1132  // Send and receive buffers
1133  if (multiProcessorDirection_[i]) {
1134 
1135  bufferPtr_->endSendBlock();
1136 
1137  source = domainPtr_->sourceRank(i, j);
1138  dest = domainPtr_->destRank(i, j);
1139  bufferPtr_->sendRecv(domainPtr_->communicator(), source, dest);
1140  stamp(SEND_RECV_GHOSTS);
1141

1143  bufferPtr_->beginRecvBlock();
1144  while (bufferPtr_->recvSize() > 0) {
1145 
1146  atomPtr = atomStoragePtr_->newGhostPtr();
1147  bufferPtr_->unpackGhost(*atomPtr);
1148  if (shift) {
1149  atomPtr->position()[i] += rshift;
1150  }
1151  recvArray_(i, j).append(*atomPtr);
1152  atomStoragePtr_->addNewGhost();
1153 
1154  // Prohibit sending back ghost in reverse direction
1155  if (j == 0) {
1156  atomPtr->plan().clearGhost(i, 1);
1157  }
1158 
1159  // Add to send arrays for remaining directions
1160  if (i < Dimension - 1) {
1161  for (ip = i + 1; ip < Dimension; ++ip) {
1162  for (jp = 0; jp < 2; ++jp) {
1163  if (atomPtr->plan().ghost(ip, jp)) {
1164  sendArray_(ip, jp).append(*atomPtr);
1165  }
1166  }
1167  }
1168  }
1169 
1170  #ifdef UTIL_DEBUG
1171  // Validate ghost coordinate on the receiving processor.
1172  coordinate = atomPtr->position()[i];
1173  if (j == 0) {
1174  assert(coordinate > bound_(i, 1));
1175  } else {
1176  assert(coordinate < bound_(i, 0));
1177  }
1178  #endif
1179 
1180  }
1181  stamp(UNPACK_GHOSTS);
1182 
1183  }
1184  #endif // ifdef UTIL_MPI
1185 
1186  } // end for transmit direction j = 0, 1
1187 
1188  } // end for Cartesian index i
1189 
1190 
1191  // Find ghost atoms for all incomplete bonds
1192  findGroupGhosts<2>(*bondStoragePtr_);
1193  #ifdef INTER_ANGLE
1194  findGroupGhosts<3>(*angleStoragePtr_);
1195  #endif
1196  #ifdef INTER_DIHEDRAL
1197  findGroupGhosts<4>(*dihedralStoragePtr_);
1198  #endif
1199 
1200  #ifdef UTIL_DEBUG
1201  #ifdef DDMD_EXCHANGER_DEBUG
1202  atomStoragePtr_->isValid();
1203  bondStoragePtr_->isValid(*atomStoragePtr_,
1204  domainPtr_->communicator(), true);
1205  #endif
1206  #endif
1207

1212  * Update ghost atom coordinates.
1213  *
1214  * Call on time steps for which no reneighboring is required.
1215  */
1216  void Exchanger::update()
1217  {
1218  stamp(START);
1219  if (!atomStoragePtr_->isCartesian()) {
1220  UTIL_THROW("Error: Coordinates not Cartesian on entry to update");
1221  }
1222 
1223  Vector lengths = boundaryPtr_->lengths();
1224  Atom* atomPtr;
1225  int i, j, k, source, dest, size, shift;
1226 
1227  for (i = 0; i < Dimension; ++i) {
1228  for (j = 0; j < 2; ++j) {
1229 
1230  // Shift on receiving processor for periodic boundary conditions
1231  shift = domainPtr_->shift(i, j);
1232 
1233  if (multiProcessorDirection_[i]) {
1234 
1235  // Pack ghost positions for sending
1236  bufferPtr_->clearSendBuffer();
1237  bufferPtr_->beginSendBlock(Buffer::UPDATE);
1238  size = sendArray_(i, j).size();
1239  for (k = 0; k < size; ++k) {
1240  bufferPtr_->packUpdate(sendArray_(i, j)[k]);
1241  }
1242  bufferPtr_->endSendBlock();
1243  stamp(PACK_UPDATE);
1244 
1245  // Send and receive buffers
1246  source = domainPtr_->sourceRank(i, j);
1247  dest = domainPtr_->destRank(i, j);
1248  bufferPtr_->sendRecv(domainPtr_->communicator(), source, dest);
1249  stamp(SEND_RECV_UPDATE);
1250 
1251  // Unpack ghost positions
1252  bufferPtr_->beginRecvBlock();
1253  size = recvArray_(i, j).size();
1254  for (k = 0; k < size; ++k) {
1255  atomPtr = &recvArray_(i, j)[k];
1256  bufferPtr_->unpackUpdate(*atomPtr);
1257  if (shift) {
1258  boundaryPtr_->applyShift(atomPtr->position(), i, shift);
1259  }
1260  }
1261  stamp(UNPACK_UPDATE);
1262 
1263  } else {
1264 
1265  // If grid().dimension(i) == 1, then copy positions of atoms
1266  // listed in sendArray to those listed in the recvArray.
1267 
1268  size = sendArray_(i, j).size();
1269  assert(size == recvArray_(i, j).size());
1270  for (k = 0; k < size; ++k) {
1271  atomPtr = &recvArray_(i, j)[k];
1272  atomPtr->position() = sendArray_(i, j)[k].position();
1273  if (shift) {
1274  boundaryPtr_->applyShift(atomPtr->position(), i, shift);
1275  }
1276  }
1277  stamp(LOCAL_UPDATE);
1278 
1279  }
1280 
1281  } // transmit direction j = 0, 1
1282 
1283  } // Cartesian direction i = 0, ..., Dimension - 1
1284 
1285  }

1346  } // transmit direction j = 1 or 0
1347 
1348  } // Cartesian direction i
