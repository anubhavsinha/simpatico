368  /*
369  * Return index of the cell that contains the position array pos = {x, y, z}.
370  */
371  inline int CellList::cellIndexFromPosition(const Vector &pos) const
372  {
373  int cx, cy, cz;
374 
375  /*
376  * Elements of invCellWidths_ are inverses of the widths of a single
377  * cell. They are defined in makeGrid. If (UTIL_ORTHOGONAL), these
378  * are given in Cartesian coordinates, or otherwise in dimensionless
379  * generalized coordinates.
380  */
381 
382  if (UTIL_ORTHOGONAL) {
383  cx = int(pos[0]*invCellWidths_[0]);
384  cy = int(pos[1]*invCellWidths_[1]);
385  cz = int(pos[2]*invCellWidths_[2]);
386  } else {
387  Vector posG;
388  boundaryPtr_->transformCartToGen(pos, posG);
389  cx = int(posG[0]*invCellWidths_[0]);
390  cy = int(posG[1]*invCellWidths_[1]);
391  cz = int(posG[2]*invCellWidths_[2]);
392  }
393 
394  #if 0
395  // need to handle case where wrong particle bin is assigned
396  // to particles close to the boundary due to roundoff error
397  if (cx == numCells_[0]) {
398  if (pos[0] >= lengths_[0])
399  UTIL_THROW("Particle left boundary.");
400  cx = numCells_[0] - 1;
401  }
402  if (cy == numCells_[1]) {
403  if (pos[1] >= lengths_[1])
404  UTIL_THROW("Particle left boundary.");
405  cy = numCells_[1] - 1;
406  }
407  if (cz == numCells_[2]) {
408  if (pos[2] >= lengths_[2])
409  UTIL_THROW("Particle left boundary.");
410  cz = numCells_[2] - 1;
411  }
412  #endif
413 
414  assert(cx >= 0);
415  assert(cx < numCells_[0]);
416  assert(cy >= 0);
417  assert(cy < numCells_[1]);
418  assert(cz >= 0);
419  assert(cz < numCells_[2]);
420 
421  return cz + cy*numCells_[2] + cx*YZCells_;
422  }
