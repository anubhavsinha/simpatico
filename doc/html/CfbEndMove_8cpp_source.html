102  bool CfbEndMove::move()

108  Molecule *molPtr; // pointer to randomly chosen molecule
109  int length, sign, beginId, endId, bondType, i;
110  bool accept;
111 
112  incrementNAttempt();
113 
114  // Choose a molecule at random
115  molPtr = &(system().randomMolecule(speciesId_));
116  length = molPtr->nAtom();
117 
118  // Require that chain length > nRegrow
119  if (nRegrow_ >= length) {
120  UTIL_THROW("nRegrow_ >= chain length");
121  }
122 
123  // Choose which chain end to regrow
124  if (random().uniform(0.0, 1.0) > 0.5) {
125  sign = +1;
126  beginId = length - nRegrow_;
127  endId = length - 1;
128  } else {
129  sign = -1;
130  beginId = nRegrow_ - 1;
131  endId = 0;
132  }
133 
134  // Store current atomic positions from segment to be regrown
135  endPtr = &(molPtr->atom(beginId));
136  for (i = 0; i < nRegrow_; ++i) {
137  oldPos_[i] = endPtr->position();
138  endPtr += sign;
139  }
140 
141  // Delete monomers, starting from chain end
142  rosen_r = 1.0;
143  energy_r = 0.0;
144  endPtr = &(molPtr->atom(endId));
145  for (i = 0; i < nRegrow_; ++i) {
146  pvtPtr = endPtr - sign;
147  if (sign == 1) {
148  bondType = molPtr->bond(endId - 1 - i).typeId();
149  } else {
150  bondType = molPtr->bond(i).typeId();
151  }
152  deleteEndAtom(endPtr, pvtPtr, bondType, rosenbluth, energy);
153  #ifndef INTER_NOPAIR
154  system().pairPotential().deleteAtom(*endPtr);
155  #endif
156  rosen_r *= rosenbluth;
157  energy_r += energy;
158  endPtr -= sign;
159  }
160 
161  // Regrow monomers
162  rosen_f = 1.0;
163  energy_f = 0.0;
164  endPtr = &(molPtr->atom(beginId));
165  for (i = 0; i < nRegrow_; ++i) {
166  pvtPtr = endPtr - sign;
167  if (sign == 1) {
168  bondType = molPtr->bond(endId - 1 - i).typeId();
169  } else {
170  bondType = molPtr->bond(i).typeId();
171  }
172  addEndAtom(endPtr, pvtPtr, bondType, rosenbluth, energy);
173  rosen_f *= rosenbluth;
174  energy_f += energy;
175 
176  #ifndef INTER_NOPAIR
177  // Add end atom to McSystem cell list
178  system().pairPotential().addAtom(*endPtr);
179  #endif
180 
181  endPtr += sign;
182  }
183

186  if (accept) {

192 
193  } else {
