618  template <int N>
619  int AtomStorage::countGroupAtoms(Group<N>& group) const
620  {
621  Atom* ptr;
622  int nAtom = 0;
623  for (int i = 0; i < N; ++i) {
624  ptr = atomPtrs_[group.atomId(i)];
625  if (ptr) {
626  if (ptr->isGhost()) {
627  UTIL_THROW("Found ghost atom");
628  }
629  ++nAtom;
630  }
631  }
632  return nAtom;
633  }

640  {

643  for (int i = 0; i < N; ++i) {
644  ptr = atomPtrs_[group.atomId(i)];
645  if (ptr) {
646  group.setAtomPtr(i, ptr);
647  ++nAtom;

649  group.clearAtomPtr(i);
