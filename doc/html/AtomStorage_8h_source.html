00618    template <int N>
00619    int AtomStorage::countGroupAtoms(Group<N>& group) const
00620    {
00621       Atom* ptr;
00622       int nAtom = 0;
00623       for (int i = 0; i < N; ++i) {
00624          ptr = atomPtrs_[group.atomId(i)];
00625          if (ptr) {
00626             if (ptr->isGhost()) {
00627                UTIL_THROW("Found ghost atom");
00628             }
00629             ++nAtom;
00630          }
00631       }
00632       return nAtom;
