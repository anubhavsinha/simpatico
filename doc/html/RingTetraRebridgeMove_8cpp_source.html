103  /*
104  * Generate, attempt and accept or reject a Monte Carlo move.
105  *
106  * A complete move involve the following stepts:
107  *
108  * 1) choose a molecule-i randomly;
109  * 2) find molecule-j of the same type as i, and monomer pairs satisfying
110  * the distance criterion exist;
111  *
112  */
113  bool RingTetraRebridgeMove::move()
114  {
115  Molecule *molPtr; // pointer to the rebridging molecule
116  Atom *aPtr, *bPtr, *cPtr, *dPtr; // atom pointers
117  Atom *lPtr, *hPtr, *lTmp, *hTmp;
118  Vector swapV;
119  int ia, ic, ib, id;
120  int ibc, ida, iLow, iHigh, nSwap;
121  int bondType;
122  double energy;
123  bool found, accept;
124 
125  incrementNAttempt();
126 
127  // Randomly pick up a molecule.
128  molPtr = &(system().randomMolecule(speciesId_));
129 
130  // Randomly choose the a atom in the tetra-group.
131  ia = random().uniformInt(0, nAtom_);
132 
133  // Search the rebriding sites.
134  found = scanBridge(molPtr, ia, ic);
135  if (!found) return false;
136 
137  ib = modId(ia+1, nAtom_);
138  id = modId(ic+1, nAtom_);
139 
140  // Calcuate the energy change.
141  aPtr = &(molPtr->atom(ia));
142  bPtr = &(molPtr->atom(ib));
143  cPtr = &(molPtr->atom(ic));
144  dPtr = &(molPtr->atom(id));
145 
146  bondType = molPtr->bond(ia).typeId();
147  tetraEnergy(aPtr, bPtr, cPtr, dPtr, bondType, energy);
148 
149  // Decide whether to accept or reject.
150  accept = random().metropolis(boltzmann(energy));
151 
152  if (accept) {
153 
154  // Find the shorter half loop.
155  ibc = ic - ib;
156  if (ibc < 0) ibc += nAtom_;
157 
158  ida = ia - id;
159  if (ida < 0) ida += nAtom_;
160 
161  if (ibc <= ida) {
162  lPtr = bPtr;
163  hPtr = cPtr;
164  iLow = ib;
165  iHigh = ic;
166  if (ibc%2 == 0)
167  nSwap = ibc / 2;
168  else
169  nSwap = (ibc + 1) / 2;
170  } else {
171  lPtr = dPtr;
172  hPtr = aPtr;
173  iLow = id;
174  iHigh = ia;
175  if (ida%2 == 0)
176  nSwap = ida / 2;
177  else
178  nSwap = (ida + 1) / 2;
179  }
180 
181  // Swap positions.
182  for (int j = 0; j < nSwap; ++j) {
183  // moving pointer close to the atom with higher index
184  hTmp = hPtr + modId(iHigh - j, nAtom_) - iHigh;
185 
186  // moving pointer close to the atom with lower index
187  lTmp = lPtr + modId(iLow + j, nAtom_) - iLow;
188 
189  swapV = lTmp->position();
190 
191  //system().moveAtom(*lTmp, hTmp->position());
192  lTmp->position() = hTmp->position();
193  #ifndef INTER_NOPAIR
194  system().pairPotential().updateAtomCell(*lTmp);
195  #endif
196 
197  //system().moveAtom(*hTmp, swapV);
198  hTmp->position() = swapV;
199  #ifndef INTER_NOPAIR
200  system().pairPotential().updateAtomCell(*hTmp);
201  #endif
202 
203  }
204 
205  // Increment counter for accepted moves of this class.
206  incrementNAccept();
207 
208  }
209 
210  return accept;
211  }

213  /*
214  * Scan the valid rebridging sits.
215  */
216  bool RingTetraRebridgeMove::
217  scanBridge(Molecule* molPtr, int ia, int &ic)
218  {
219  Vector r1, r2, dR;
220  int ib, id, i;
221  int *idList, nGroup;
222  double lenSq, minSq, maxSq;
223  bool found;
224 
225  // Preparation.
226  found = false;
227  maxSq = upperBridge_ * upperBridge_;
228  minSq = lowerBridge_ * lowerBridge_;
229 
230  // b Atom id.
231  ib = modId(ia+1, nAtom_);
232  r1 = molPtr->atom(ia).position();
233  r2 = molPtr->atom(ib).position();
234 
235  // Check the ab bond.
236  lenSq = boundary().distanceSq(r1, r2);
237  if (lenSq >= maxSq || lenSq <= minSq) {
238  ic = -1; // invalid value
239  return found;
240  }
241 
242  // Retrace the non-periodic molecule shape.
243  R_[0] = molPtr->atom(0).position();
244  for (i = 1; i < nAtom_; ++i) {
245  r1 = molPtr->atom(i-1).position();
246  r2 = molPtr->atom(i).position();
247  system().boundary().distanceSq(r2, r1, dR);
248  R_[i].add(R_[i-1], dR);
249  }
250 
251  // Loop over all monomers to find the proper rebridging sites.
252  nGroup = 0;
253  idList = new int[nAtom_ - 3];
254  for (i = 1; i < nAtom_ - 2; ++i) {
255  ic = modId(ib+i,nAtom_);
256  id = modId(ic+1,nAtom_);
257 
258  // Distance test.
259  dR.subtract(R_[ia],R_[ic]);
260  lenSq = dR.square();
261  if (lenSq > minSq && lenSq < maxSq) {
262 
263  dR.subtract(R_[id],R_[ic]);
264  lenSq = dR.square();
265  if (lenSq > minSq && lenSq < maxSq) {
266 
267  dR.subtract(R_[id],R_[ib]);
268  lenSq = dR.square();
269  if (lenSq > minSq && lenSq < maxSq) {
270 
271  idList[nGroup] = ic;
272  ++nGroup;
273 
274  }
275  }
276  }
277 
278  } // Loop over all possible c atoms.
279 
280  if (nGroup > 0) {
281  found = true;
282  ic = idList[random().uniformInt(0,nGroup)];
283  } else {
284  found = false;
285  ic = -1; // Invalid value.
286  }
287 
288  delete [] idList;
289  return found;
290 
291  }
