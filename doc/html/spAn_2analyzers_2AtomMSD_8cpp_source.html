00117    * Evaluate end-to-end vectors of all chains, add to ensemble.
00118    */
00119    void AtomMSD::sample(long iStep) 
00120    { 
00121       if (!isAtInterval(iStep)) return;
00122 
00123       // Confirm that nMolecule has remained constant, and nMolecule > 0.
00124       if (nMolecule_ <= 0) {
00125          UTIL_THROW("nMolecule <= 0");
00126       }
00127       if (nMolecule_ != processor().species(speciesId_).size()) {
00128          UTIL_THROW("Number of molecules has changed.");
00129       }
00130 
00131       Vector r;
00132       IntVector shift;
00133       Vector lengths = processor().boundary().lengths();
00134       int i, j;
00135       Species::Iterator iter;
00136 
00137       i = 0;
00138       processor().species(speciesId_).begin(iter);
00139       for ( ; iter.notEnd(); ++iter) {
00140 
00141          r = iter->atom(atomId_).position;
00142          processor().boundary().shift(r);
00143 
00144          // Compare current r to previous position, oldPositions_[i]
00145          processor().boundary().distanceSq(r, oldPositions_[i], shift);
00146 
00147          // If this atom crossed a boundary, increment its shift vector
00148          shifts_[i] += shift;
00149 
00150          // Reconstruct true position
00151          for (j = 0; j < Dimension; ++j) {
00152             truePositions_[i][j] = r[j] + shifts_[i][j]*lengths[j];
00153          }
00154 
00155          // Store current position in box for comparison to next one
00156          oldPositions_[i] = r;
00157 
00158          ++i;  
00159       }
00160       accumulator_.sample(truePositions_);
00161 
00162    }
