45  void LammpsConfigIo::read(std::istream &in)
46  {
47  // Calculate atomCapacity for entire simulation
48  int atomCapacity = 0;
49  int bondCapacity = 0;
50  int nSpecies = simulation().nSpecies();
51  int speciesCapacity = 0;
52  int iSpec, i;
53  Species* speciesPtr;
54  for (iSpec = 0; iSpec < nSpecies; ++iSpec) {
55  speciesPtr = &simulation().species(iSpec);
56  speciesCapacity = speciesPtr->capacity();
57  atomCapacity += speciesCapacity*speciesPtr->nAtom();
58  bondCapacity += speciesCapacity*speciesPtr->nBond();
59  }
60 
61 
62  // Read and discard title line
63  std::string line;
64  std::getline(in, line);
65 
66  // Read numbers of atoms, bonds, etc.
67  int nAtom;
68  int nBond;
69  int nAngle;
70  int nDihedral;
71  int nImproper;
72  in >> nAtom >> Label("atoms");
73  in >> nBond >> Label("bonds");
74  in >> nAngle >> Label("angles");
75  in >> nDihedral >> Label("dihedrals");
76  in >> nImproper >> Label("impropers");
77 
78  /*
79  * Validate nAtom and nBond
80  * Lammps files can be read only if the number of atoms and bonds
81  * in the lammps file exactly matches the corresponding capacities.
82  */
83  if (nAtom != atomCapacity) {
84  UTIL_THROW("nAtom != atomCapacity");
85  }
86  if (nBond != bondCapacity) {
87  UTIL_THROW("nAtom != atomCapacity");
88  }
89 
90  // Read numbers of atom types, bond types, etc.
91  int nAtomType;
92  int nBondType;
93  int nAngleType;
94  int nDihedralType;
95  int nImproperType;
96  in >> nAtomType >> Label("atom") >> Label("types");
97  in >> nBondType >> Label("bond") >> Label("types");
98  in >> nAngleType >> Label("angle") >> Label("types");
99  in >> nDihedralType >> Label("dihedral") >> Label("types");
100  in >> nImproperType >> Label("improper") >> Label("types");
101 
102  if (nAtomType > simulation().nAtomType()) {
103  UTIL_THROW("nAtomType > simulation().nAtomType()");
104  }
105  // Read boundary dimensions
106  Vector lengths;
107  Vector min;
108  Vector max;
109  in >> min[0] >> max[0] >> Label("xlo") >> Label("xhi");
110  in >> min[1] >> max[1] >> Label("ylo") >> Label("yhi");
111  in >> min[2] >> max[2] >> Label("zlo") >> Label("zhi");
112  lengths.subtract(max, min);
113  boundary().setOrthorhombic(lengths);
114 
115  // Read particle masses (discard values)
116  double mass;
117  int atomTypeId;
118  in >> Label("Masses");
119  for (i = 0; i < nAtomType; ++i) {
120  in >> atomTypeId >> mass;
121  }
122 
123  /*
124  * Read atomic positions
125  *
126  * Atom tags must appear in order, numbered from 1
127  */
128  in >> Label("Atoms");
129  Molecule* molPtr;
130  Molecule::AtomIterator atomIter;
131  IntVector shift;
132  int atomId, molId, iMol, nMolecule;
133  for (iSpec=0; iSpec < nSpecies; ++iSpec) {
134  speciesPtr = &simulation().species(iSpec);
135  nMolecule = speciesPtr->capacity();
136  for (iMol = 0; iMol < nMolecule; ++iMol) {
137  molPtr = &(speciesPtr->reservoir().pop());
138  system().addMolecule(*molPtr);
139 
140  if (molPtr != &system().molecule(iSpec, iMol)) {
141  UTIL_THROW("Molecule index error");
142  }
143 
144  // Read positions, shift into primary cell
145  for (molPtr->begin(atomIter); atomIter.notEnd(); ++atomIter) {
146 
147  in >> atomId >> molId >> atomTypeId;
148  if (atomId != atomIter->id() + 1)
149  UTIL_THROW("Atom tags not ordered");
150 
151  in >> atomIter->position();
152 
153  // Shift coordinates to box (0, length(i));
154  atomIter->position() += min;
155 
156  in >> shift;
157 
158  }
159  }

199  out << Int(simulation().nBondType()) << " bond types" << endl;

219  out << Int(iType+1, 5) << Dbl(1.0) << endl;
