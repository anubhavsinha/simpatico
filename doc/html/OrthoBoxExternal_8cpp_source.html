00095    void OrthoBoxExternal::readParameters(std::istream &in)

00103    
00104       // Read parameters
00105       read<int>(in, "indexConfinement", indConfine_);
00106       if (indConfine_ < 0 || indConfine_ > Dimension) {
00107          UTIL_THROW("Invalid index for confinement directions.");

00109       read<double>(in, "epsilon", epsilon_);
00110       read<double>(in, "sigma", sigma_);
00111       read<double>(in, "cutoff", cutoff_);
00112 
00113       // Initialize dependent variables (particle number density = 0.7)
00114       sigmaCb_ = sigma_ * sigma_ * sigma_;
