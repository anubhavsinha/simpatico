00098    void OrthoBoxExternal::readParameters(std::istream &in)

00100       if (nAtomType_ == 0) {

00106    
00107       // Read parameters
00108       read<int>(in, "indexConfinement", indConfine_);
00109       if (indConfine_ < 0 || indConfine_ > Dimension) {
00110          UTIL_THROW("Invalid index for confinement directions.");

00112       read<double>(in, "epsilon", epsilon_);
00113       read<double>(in, "sigma", sigma_);
00114       read<double>(in, "cutoff", cutoff_);
00115 
00116       // Initialize dependent variables (particle number density = 0.7)
00117       sigmaCb_ = sigma_ * sigma_ * sigma_;
00118       coeff_   = 4.0 * epsilon_ * acos(-1.0) / 45.0 * 0.7 * sigmaCb_;

00120       isInitialized_ = true;
