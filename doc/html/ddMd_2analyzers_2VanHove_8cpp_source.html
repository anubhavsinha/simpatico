00156          // Set all Fourier modes to zero
00157          for (i = 0; i < nWave_; ++i) {
00158             fourierModes_[i] = std::complex<double>(0.0, 0.0);
00159          }
00160  
00161          simulation().atomStorage().begin(atomIter);
00162          for ( ; atomIter.notEnd(); ++atomIter) {
00163             position = atomIter->position();
00164             typeId   = atomIter->typeId();
00165             coeff    = atomTypeCoeffs_[typeId];
00166             
00167             // Loop over wavevectors
00168             for (i = 0; i < nWave_; ++i) {
00169 
00170                product = position.dot(waveVectors_[i]);
00171                expFactor = exp( product*Constants::Im );
00172                fourierModes_[i] += coeff*expFactor;
00173             }
00174          }
00175   
00176          for (i = 0; i < nWave_; ++i) {
00177             totalFourierModes_[i] = std::complex<double>(0.0, 0.0);
00178          }
00179 
00180          #ifdef UTIL_MPI
00181          // Loop over wavevectors
00182          for (int i = 0; i < nWave_; ++i) {
00183             //Sum values from all processors.
00184             simulation().domain().communicator().
00185             Reduce(&fourierModes_[i], &totalFourierModes_[i],
00186                     1, MPI::DOUBLE_COMPLEX, MPI::SUM, 0);
00187          }
00188          #else
00189          for (int i = 0; i < nWave_; ++i) {
00190             totalFourierModes_[i] = fourierModes_[i];
00191          }
00192          #endif
00193          
00194          if (simulation().domain().isMaster()) {
00195             // Add Fourier modes to autocorrelation accumulators
00196             double sqrtV = sqrt(simulation().boundary().volume());
00197             for (int i = 0; i < nWave_; ++i) {
