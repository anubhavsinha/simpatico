62  * Pop and process the next molecule on the workStack (private).
63  */
64  void ClusterIdentifier::processNextMolecule(Cluster& cluster)
65  {
66  CellList::NeighborArray neighborArray;
67  Molecule::AtomIterator atomIter;
68  ClusterLink* thisLinkPtr;
69  ClusterLink* otherLinkPtr;
70  Atom* otherAtomPtr;
71  Boundary& boundary = system().boundary();
72  double cutoffSq = cutoff_*cutoff_;
73  double rsq;
74  int thisMolId, otherMolId, otherClusterId;
75 
76  // Pop this molecule off the stack
77  thisLinkPtr = &workStack_.pop();
78  thisMolId = thisLinkPtr->molecule().id();
79  if (thisLinkPtr->clusterId() != cluster.id()) {
80  UTIL_THROW("Top ClusterLink not marked with this cluster id");
81  }
82 
83  /*
84  * Loop over atoms of this molecule.
85  * For each atom of type atomTypeId_, find neighboring molecules.
86  * Add each new neighbor to the cluster, and to the workStack.
87  */
88  thisLinkPtr->molecule().begin(atomIter);
89  for ( ; atomIter.notEnd(); ++atomIter) {
90  if (atomIter->typeId() == atomTypeId_) {
91  cellList_.getNeighbors(atomIter->position(), neighborArray);
92  for (int i = 0; i < neighborArray.size(); i++) {
93  otherAtomPtr = neighborArray[i];
94  otherMolId = otherAtomPtr->molecule().id();
95  if (otherMolId != thisMolId) {
96  rsq = boundary.distanceSq(atomIter->position(),
97  otherAtomPtr->position());
98  if (rsq < cutoffSq) {
99  otherLinkPtr = &(links_[otherMolId]);
100  assert(&otherLinkPtr->molecule() ==
101  &otherAtomPtr->molecule());
102  otherClusterId = otherLinkPtr->clusterId();
103  if (otherClusterId == -1) {
104  cluster.addLink(*otherLinkPtr);
105  workStack_.push(*otherLinkPtr);
106  } else
107  if (otherClusterId != cluster.id()) {
108  UTIL_THROW("Cluster Clash!");
109  }
110  }
111  }
112  } // neighbor atoms
113  }
114  } // atoms

119  * Identify all clusters in the system.
120  */
121  void ClusterIdentifier::identifyClusters()
122  {
123 
124  // Clear cell list, clusters array and all links
125  cellList_.clear();
126  clusters_.clear();
127  for (int i = 0; i < links_.capacity(); ++i) {
128  links_[i].clear();
129  }
130 
131  // Build the cellList, associate Molecule with ClusterLink.
132  // Iterate over molecules of species speciesId_
133  System::MoleculeIterator molIter;
134  Molecule::AtomIterator atomIter;
135  system().begin(speciesId_, molIter);
136  for ( ; molIter.notEnd(); ++molIter) {
137 
138  // Associate this Molecule with a ClusterLink
139  links_[molIter->id()].setMolecule(*molIter.get());
140 
141  // Add atoms of type = atomTypeId_ to the CellList
142  for (molIter->begin(atomIter); atomIter.notEnd(); ++atomIter) {
143  if (atomIter->typeId() == atomTypeId_) {
144  system().boundary().shift(atomIter->position());
145  cellList_.addAtom(*atomIter);
146  }
147 
148  }
149  }
150 
151  // Identify all clusters
152  Cluster* clusterPtr;
153  ClusterLink* linkPtr;
154  int clusterId = 0;
155  system().begin(speciesId_, molIter);
156  for ( ; molIter.notEnd(); ++molIter) {
157 
158  // Find the link with same index as this molecule
159  linkPtr = &(links_[molIter->id()]);
160  assert (&(linkPtr->molecule()) == molIter.get());
161 
162  // If this link is not in a cluster, begin a new cluster
163  if (linkPtr->clusterId() == -1) {
164 
165  // Add a new empty cluster to clusters_ array
166  clusters_.resize(clusterId+1);
167  clusterPtr = &clusters_[clusterId];
168  clusterPtr->clear();
169  clusterPtr->setId(clusterId);
170 
171  // Identify molecules in this cluster
172  clusterPtr->addLink(*linkPtr);
173  workStack_.push(*linkPtr);
174  while (workStack_.size() > 0) {
175  processNextMolecule(*clusterPtr);
176  }
177 
178  clusterId++;
179  }
180 
181  }
