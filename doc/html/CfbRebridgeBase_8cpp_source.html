00484    /* 
00485    * First determine the preferential orientation angle and angle biasing
00486    * spring constant; then compute the statistical weight for the biasing
00487    * function using Simpson's rule:
00488    *
00489    * Note: the variou branches represent certain geometrical extreme
00490    * situations.
00491    */
00492    void CfbRebridgeBase::orientationBiasTable
00493          (double l_20, double &prefAng, double &kappaAng, double &normConst)
00494    {
00495       double cos_ang;
00496    
00497       // Preferential orientation & effective spring constant
00498       if ( l_20 <= fabs(length21_ - length10_) ) {
00499    
00500          if (length21_ > length10_) {
00501             prefAng   = 0.0;
00502             kappaAng  = kappa10_ * (length21_ - l_20 - length10_);
00503             kappaAng *= length21_ * l_20 / (length21_ - l_20);
00504          } else if (length21_ == length10_) {
00505             prefAng   = acos(-1.0)/2.0;
00506             kappaAng  = 0.0;
00507          } else {
00508             prefAng   = acos(-1.0);
00509             kappaAng  = kappa10_ * (length10_ - l_20 - length21_);
00510             kappaAng *= length21_ * l_20 / (length21_ + l_20);
00511          }
00512    
00513       } else if ( l_20 >= (length21_ + length10_) ) { 
00514    
00515          prefAng  = 0.0;           
00516     kappaAng = kappa10_ * (l_20 - length21_ - length10_) * length21_ * l_20
00517                      / (l_20 - length21_);      
00518    
00519       } else {
00520    
00521          cos_ang  = (l_20*l_20 + length21_*length21_ - length10_*length10_)
00522                     * 0.5 / length21_ / l_20;
00523          prefAng  = acos(cos_ang);
00524          kappaAng = length21_ * l_20 / length10_;
00525          kappaAng = kappa10_ * kappaAng * kappaAng * (1.0 - cos_ang*cos_ang);
00526    
00527       }
00528    
00529       // Calculate normalization factor.
00530       if (kappaAng < 1.0E-7) {
00531          normConst = 2.0;
00532       } else {
00533          computeNormalization(prefAng, kappaAng, normConst);
00534       }
00535    }

00537    /* 
00538    * Use Simpson's rule to compute the normalization factor
00539    *   int { exp(-0.5*beta*kappaAng*(x-prefAng)**2)*sin(x) , {x, 0, pi} }
00540    */
00541    void CfbRebridgeBase::computeNormalization
00542       (double prefAng, double kappaAng, double &normConst)
00543    {
00544       int    i, N;                     // N must be even
00545       double delta, x, sum, tmp;
00546    
00547       // indices for Simpson's rule: 0,1,...,N
00548       N     = 400;                     // default value
00549       tmp   = 0.5*kappaAng;
00550       delta = acos(-1.0)/double(N);
00551    
00552       // the end points contribute nothing since sin(0)=sin(pi)=0.
00553       sum   = 0.0;
00554       x     = -delta;
00555       for (i=0; i < N-1; i=i+2) {
00556          x += delta;
00557          //sum += 2.0 * exp( -tmp * (x-prefAng)*(x-prefAng) ) * sin(x);
00558          sum += 2.0 * boltzmann( tmp * (x-prefAng)*(x-prefAng) ) * sin(x);
00559          x += delta;
00560          //sum += 4.0 * exp( -tmp * (x-prefAng)*(x-prefAng) ) * sin(x);
00561          sum += 4.0 * boltzmann( tmp * (x-prefAng)*(x-prefAng) ) * sin(x);
00562       }
00563       normConst = delta*sum/3.0;
00564    }
