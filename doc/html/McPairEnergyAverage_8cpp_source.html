96  * Evaluate energy per particle, and add to ensemble.
97  */
98  void McPairEnergyAverage::sample(long iStep)
99  {
100 
101  if (!isAtInterval(iStep)) return;
102 
103  Atom *iAtomPtr, *jAtomPtr;
104  int nNeighbor, nInCell;
105  int ic, nc, i, j;
106  double energy;
107  double rsq;
108 
109  // Loop over cells
110  energy = 0.0;
111  nc = system().pairPotential().cellList().totCells();
112  for (ic = 0; ic < nc; ++ic) {
113 
114  // Get array of neighbors_
115  system().pairPotential().cellList().getCellNeighbors(ic, neighbors_, nInCell);
116  nNeighbor = neighbors_.size();
117 
118  // Loop over primary atoms in this cell
119  for (i = 0; i < nInCell; ++i) {
120  iAtomPtr = neighbors_[i];
121 
122  // Loop over secondary atoms in this and neighboring cells
123  for (j = 0; j < nNeighbor; ++j) {
124  jAtomPtr = neighbors_[j];
125 
126  if (selector_.match(*iAtomPtr, *jAtomPtr)) {
127 
128  // Exclude masked pairs
129  if (!iAtomPtr->mask().isMasked(*jAtomPtr)) {
130 
131  rsq = system().boundary().
132  distanceSq(iAtomPtr->position(), jAtomPtr->position());
133  energy += system().pairPotential().
134  energy(rsq, iAtomPtr->typeId(), jAtomPtr->typeId());
135 
136  }
137 
138  }
139 
140  } // secondary atoms
141 
142  } // primary atoms
143 
144  } // cells
145 
146  accumulator_.sample(energy, outputFile_);
147  }
