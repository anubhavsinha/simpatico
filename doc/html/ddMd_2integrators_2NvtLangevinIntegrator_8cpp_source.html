86  /*
87  * Setup at beginning of run, before entering main loop.
88  */
89  void NvtLangevinIntegrator::setup()
90  {
91  // Precondition
92  const EnergyEnsemble& energyEnsemble = simulation().energyEnsemble();
93  if (!energyEnsemble.isIsothermal()) {
94  UTIL_THROW("Energy ensemble is not isothermal");
95  }
96 
97  // Initialize state and clear statistics on first usage.
98  if (!isSetup()) {
99  clear();
100  setIsSetup();
101  }
102 
103  // Exchange atoms, build pair list, compute forces.
104  setupAtoms();
105 
106  // Set constants that are independent of atom type
107  double cv = (exp(-dt_*gamma_) - 1.0)/dt_;
108  double temp = energyEnsemble.temperature();
109  double d = 2.0/(1.0 + exp(-dt_*gamma_));
110  double cr = 12.0*temp*d*(1.0 - exp(-2.0*dt_*gamma_))/(dt_*dt_);
111 
112  // Loop over atom types
113  double dtHalf = 0.5*dt_;
114  double mass;
115  int nAtomType = prefactors_.capacity();
116  for (int i = 0; i < nAtomType; ++i) {
117  mass = simulation().atomType(i).mass();
118  prefactors_[i] = dtHalf/mass;
119  cv_[i] = mass*cv;
120  cr_[i] = sqrt(mass*cr);
121  }
122 
123  }

125  /*
126  * First half of velocity-Verlet update.
127  */
128  void NvtLangevinIntegrator::integrateStep1()
129  {
130  Vector dv;
131  Vector dr;
132  double prefactor; // = 0.5*dt/mass
133  AtomIterator atomIter;
134 
135  // 1st half of velocity Verlet.
136  atomStorage().begin(atomIter);
137  for ( ; atomIter.notEnd(); ++atomIter) {
138  prefactor = prefactors_[atomIter->typeId()];
139 
140  dv.multiply(atomIter->force(), prefactor);
141  atomIter->velocity() += dv;
142 
143  dr.multiply(atomIter->velocity(), dt_);
144  atomIter->position() += dr;
145  }
146  }

148  /*
149  * Second half of velocity-Verlet update.
150  */
151  void NvtLangevinIntegrator::integrateStep2()
152  {
153  Util::Random& random = simulation().random();
154  Vector dv;
155  Vector df;
156  double cr;
157  AtomIterator atomIter;
158  int typeId, j;
159 
160  // 2nd half of velocity Verlet
161  atomStorage().begin(atomIter);
162  for ( ; atomIter.notEnd(); ++atomIter) {
163  typeId = atomIter->typeId();
164 
165  // Add Langevin drag and random force to atomic force
166  df.multiply(atomIter->velocity(), cv_[typeId]);
167  cr = cr_[typeId];
168  for (j=0; j < Dimension; ++j) {
169  df[j] += (random.uniform() - 0.5)*cr;
170  }
171  atomIter->force() += df;
172 
173  // Update velocity (half step)
174  dv.multiply(atomIter->force(), prefactors_[typeId]);
175  atomIter->velocity() += dv;
176  }
177 
178  // Notify observers of change in velocity
179  simulation().velocitySignal().notify();
180  }
