00137             //temprature = sys.kineticEnergy()*2.0/ndof;
00138             temprature = 1;

00140              
00141             elements[0] = (total(0,0) - pressure / 3.0) / (10.0 * temprature);
00142             elements[4] = (total(1,1) - pressure / 3.0) / (10.0 * temprature);
00143             elements[8] = (total(2,2) - pressure / 3.0) / (10.0 * temprature);
00144             elements[1] = (total(0,1) + total(1,0)) / 2.0 / (10.0 * temprature);
00145             elements[2] = (total(0,2) + total(2,0)) / 2.0 / (10.0 * temprature);
00146             elements[5] = (total(1,2) + total(2,1)) / 2.0 / (10.0 * temprature);
00147             elements[3] = elements[1];
00148             elements[6] = elements[2];
