00141    void LinearRouseAutoCorr::setup() 
00142    { 
00143 
00144       if (!isInitialized_) {
00145          UTIL_THROW("Object is not intitialized");
00146       }
00147 
00148       // Get number of molecules and number of atoms per molecule.
00149       nMolecule_ = system().nMolecule(speciesId_);
00150       if (nAtom_ <= 1) UTIL_THROW("Number of atoms per molecule < 2.");
00151 
00152       // Initialize mode projection coefficients.
00153       double qMode = acos(-1.0) * double(p_) / double(nAtom_ - 1);
00154       if (p_ == 0) {
00155          for (int j = 0; j < nAtom_; ++j)
00156             projector_[j] = 1.0 / double(nAtom_);
00157       } else {
00158          for (int j = 0; j < nAtom_; ++j)
00159             projector_[j] = 1.0 / double(nAtom_) * cos(qMode*double(j));
00160       }
00161       projector_[0] -= 0.5/double(nAtom_);
00162       projector_[nAtom_ - 1] -= 0.5 / double(nAtom_);
00163 
00164       // Initialize the AutoCorrArray object
00165       accumulator_.setNEnsemble(nMolecule_);
00166 
00167       accumulator_.clear();
00168    }

00191             atomPos = moleculePtr->atom(0).position();

00193             data_[i] = dR;
00194             for (j = 1; j < nAtom_; j++) {
00195                r1 = moleculePtr->atom(j-1).position();
00196                r2 = moleculePtr->atom(j).position();
00197                system().boundary().distanceSq(r2, r1, dR);
00198                atomPos += dR;
00199                dR.multiply(atomPos, projector_[j]);
00200                data_[i] += dR;
00201             }
00202          }
