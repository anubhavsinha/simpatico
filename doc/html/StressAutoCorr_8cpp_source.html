00112    void StressAutoCorr::sample(long iStep) 
00113    {  
00114       if (isAtInterval(iStep))  {
00115          Simulation& sim = simulation();
00116          sim.computeVirialStress();
00117          sim.computeKineticStress();
00118 
00119          if (sim.domain().isMaster()) {
00120             if (!accumulatorPtr_) {
00121                UTIL_THROW("Null accumulatorPtr_ on master");
00122             }
00123 
00124             Tensor virial  = sim.virialStress();
00125             Tensor kinetic = sim.kineticStress();
00126             Tensor total = total.add(virial, kinetic);
00127 
00128             // Remove trace
00129             double pressure = 0.0;
00130             int i, j;
00131             for (i = 0; i < Dimension; ++i) {
00132                pressure += total(i,i);
00133             }
00134             pressure = pressure/double(Dimension);
00135             for (i = 0; i < Dimension; ++i) {
00136                total(i,i) -= pressure;
00137             }
00138    
00139             double factor = sqrt(sim.boundary().volume()/10.0);
00140             for (i = 0; i < Dimension; ++i) {
00141                for (j = 0; j < Dimension; ++j) {
00142                   total(i,j) *= factor;
00143                }
00144             }
00145    
00146             accumulatorPtr_->sample(total);
00147          }
00148       }
00149    }
