327  anglePtr->atom(1).position(), dr2);

406  void AnglePotentialImpl<Interaction>::computeStressImpl(T& stress) const
407  {
408  Vector dr1, dr2, force1, force2;
409  const Atom *atom0Ptr, *atom1Ptr, *atom2Ptr;
410 
411  setToZero(stress);
412 
413  // Iterate over all angles in System.
414  System::ConstMoleculeIterator molIter;
415  Molecule::ConstAngleIterator angleIter;
416  for (int iSpec = 0; iSpec < simulation().nSpecies(); ++iSpec) {
417  if (simulation().species(iSpec).nAngle() > 0) {
418  for (begin(iSpec, molIter); molIter.notEnd(); ++molIter) {
419  for (molIter->begin(angleIter); angleIter.notEnd(); ++angleIter){
420  atom0Ptr = &(angleIter->atom(0));
421  atom1Ptr = &(angleIter->atom(1));
422  atom2Ptr = &(angleIter->atom(2));
423 
424  boundary().distanceSq(atom1Ptr->position(),
425  atom0Ptr->position(), dr1);
426  boundary().distanceSq(atom2Ptr->position(),
427  atom1Ptr->position(), dr2);
428 
429  // force1 -- along dr1; force2 -- along dr2.
430  interaction().force(dr1, dr2,
431  force1, force2, angleIter->typeId());
432 
433  dr1 *= -1.0;
434  dr2 *= -1.0;
435  incrementPairStress(force1, dr1, stress);
436  incrementPairStress(force2, dr2, stress);
437  }
438  }
439  }
440  }
441 
442  stress /= boundary().volume();
443  normalizeStress(stress);
444  }
