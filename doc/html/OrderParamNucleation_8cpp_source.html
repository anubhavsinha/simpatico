121  /*
122  * Increment cos factor.
123  */
124  void OrderParamNucleation::sample(long iStep)
125  {
126  if (isAtInterval(iStep)) {
127 
128  Vector blengths;
129  Boundary* boundaryPtr = &simulation().boundary();
130  blengths = boundaryPtr->lengths();
131 
132  Vector position;
133  double slabLength, product, cosFactor;
134  AtomIterator atomIter;
135  int typeId, bin;
136 
137  simulation().atomStorage().begin(atomIter);
138  for ( ; atomIter.notEnd(); ++atomIter) {
139  position = atomIter->position();
140  typeId = atomIter->typeId();
141 
142  slabLength = blengths[parallelIndex_]/nBin_;
143  bin = position[parallelIndex_]/slabLength;
144 
145  product = position[perpIndex_]*2.0*Constants::Pi*periodicity_;
146  product /= blengths[perpIndex_];
147  cosFactor = cos(product)*cos(product);
148  cosFactors_(typeId, bin) += cosFactor;
149  }
150 
151  #ifdef UTIL_MPI
152  // Loop over wavevectors
153  for (int i = 0; i < nAtomType_; ++i) {
154  for (int j = 0; j < nBin_; ++j) {
155  //Sum values from all processors.
156  simulation().domain().communicator().
157  Reduce(&cosFactors_(i, j), &totalCosFactors_(i, j),
158  1, MPI::DOUBLE, MPI::SUM, 0);
159  }
160  }
161  #else
162  for (int i = 0; i < nAtomType_; ++i) {
163  for (int j = 0; j < nBin_; ++j) {
164  totalCosFactors_(i, j) = cosFactors_(i, j);
165  }
166  }
167  #endif
168 
169  ++nSample_;
170  }

172  }
