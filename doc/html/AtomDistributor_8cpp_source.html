00128       sendArrays_.allocate(gridSize, sendCapacity_);

00131          sendSizes_[i] = 0;
00132          for (int j = 0; j < sendCapacity_; ++j) {
00133             sendArrays_(i, j) = 0;
00134          }
00135       }
