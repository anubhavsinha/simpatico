00111       #endif
00112 
00113       // If master processor
00114       if (rank == 0) {
00115 
00116          // Default cacheCapacity_ = (# processors)*(max atoms per send)
00117          if (cacheCapacity_ <= 0) {
00118             cacheCapacity_ = gridSize * sendCapacity_;
00119          }
00120 
00121          // Allocate memory for array of atoms on the master processor. 
00122          cache_.allocate(cacheCapacity_);
00123          reservoir_.allocate(cacheCapacity_);
00124 
00125          // Push all atoms onto the reservoir stack, in reverse order.
00126          for (int i = cacheCapacity_ - 1; i >= 0; --i) {
00127             reservoir_.push(cache_[i]);
00128          }
00129 
00130          #ifdef UTIL_MPI
00131          // Allocate memory for sendArrays_ matrix, and nullify all elements.
00132          sendArrays_.allocate(gridSize, sendCapacity_);
00133          sendSizes_.allocate(gridSize);
00134          for (int i = 0; i < gridSize; ++i) {
00135             sendSizes_[i] = 0; 
00136             for (int j = 0; j < sendCapacity_; ++j) {
00137                sendArrays_(i, j) = 0;      
00138             }
00139          }
00140          #endif
00141 
00142       }
