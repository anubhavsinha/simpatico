00108    /* 
00109    * Generate a random bond length chosen from an equilibrium distribution for
00110    * randomly oriented bonds. 
00111    *
00112    * The desired probability distribution is proportional to x*x*G(x) for x > 0,
00113    * where G(x) is a Gaussian distribution with a mean x0=length_[type] and a 
00114    * standard deviation sd = sqrt(temperature/kappa_[type]).
00115    *
00116    * Algorithm:
00117    *   - Generate trial bond length x from the Gaussian distribution G(x).
00118    *   - If x < 0, generate another trial bond length.
00119    *   - Let y = x*x/xm*xm, where xm = x0 + 7.0*sd is a maximum length.
00120    *     The algorithm is correct only for y < 1, or x < xm. The distribution
00121    *     G(y) for generated values of y is proportional to G(x)/x.
00122    *   - Accept y if it is greater than a uniform random number in [0,1]. 
00123    *     For y < 1, or x < xm, the probability of accepting y is equal to y. 
00124    *     The distribution A(y) of accepted values of y for y < 1 is thus 
00125    *     proportional to y*G(y), or to x*G(x).
00126    *   - If y is accepted, return x. The distribution A(x) for accepted 
00127    *     values of x is proportional to x*x*G(x) for x < xm.
00128    *
00129    * Limitations:
00130    *     The distribution is correct only for x < xm. The probability of
00131    *     generating a value x >= xm is, however, less than exp(-49/2).
00132    */
00133    double HarmonicBond::randomBondLength(
00134                          Random *random, double beta, int type) const
00135    {
00136    
00137       double sd  = 1.0/sqrt( beta*kappa_[type] );
00138       double x0  = length_[type];
00139       double xm2 = length_[type] + 7.0*sd;
00140              xm2 = xm2*xm2;
00141       double x;
00142    
00143       // Generate trials until one is accepted
00144       for (;;) { 
00145          x = x0 + sd*random->gaussian();
00146          if (x < 0) continue;
00147          if ( x*x/xm2 >= random->uniform(0.0,1.0) ) {
00148             return x;
00149          };
00150       }
00151    }
