00105    /* 
00106    * Generate a random bond length chosen from an equilibrium distribution for
00107    * randomly oriented bonds. 
00108    *
00109    * The desired probability distribution is proportional to x*x*G(x) for x > 0,
00110    * where G(x) is a Gaussian distribution with a mean x0=length_[type] and a 
00111    * standard deviation sd = sqrt(temperature/kappa_[type]).
00112    *
00113    * Algorithm:
00114    *   - Generate trial bond length x from the Gaussian distribution G(x).
00115    *   - If x < 0, generate another trial bond length.
00116    *   - Let y = x*x/xm*xm, where xm = x0 + 7.0*sd is a maximum length.
00117    *     The algorithm is correct only for y < 1, or x < xm. The distribution
00118    *     G(y) for generated values of y is proportional to G(x)/x.
00119    *   - Accept y if it is greater than a uniform random number in [0,1]. 
00120    *     For y < 1, or x < xm, the probability of accepting y is equal to y. 
00121    *     The distribution A(y) of accepted values of y for y < 1 is thus 
00122    *     proportional to y*G(y), or to x*G(x).
00123    *   - If y is accepted, return x. The distribution A(x) for accepted 
00124    *     values of x is proportional to x*x*G(x) for x < xm.
00125    *
00126    * Limitations:
00127    *     The distribution is correct only for x < xm. The probability of
00128    *     generating a value x >= xm is, however, less than exp(-49/2).
00129    */
00130    double HarmonicBond::randomBondLength(
00131                          Random *random, double beta, int type) const
00132    {
00133    
00134       double sd  = 1.0/sqrt( beta*kappa_[type] );
00135       double x0  = length_[type];
00136       double xm2 = length_[type] + 7.0*sd;
00137              xm2 = xm2*xm2;
00138       double x;
00139    
00140       // Generate trials until one is accepted
00141       for (;;) { 
00142          x = x0 + sd*random->gaussian();
00143          if (x < 0) continue;
00144          if ( x*x/xm2 >= random->uniform(0.0,1.0) ) {
00145             return x;
00146          };
00147       }
00148    }
