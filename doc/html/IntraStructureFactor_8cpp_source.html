00136          Vector  position;
00137          std::complex<double> rho[2];
00138          std::complex<double> expFactor;
00139          double  volume = system().boundary().volume();
00140          double  product, dS;

00142          Molecule::ConstAtomIterator   atomIter;
00143          int  typeId, i, j, k;
00144 
00145          makeWaveVectors();
00146 
00147          // Set all Deltas to zero
00148          for (i = 0; i < nWave_; ++i) {
00149             for (int pairId = 0; pairId < nAtomTypeIdPair_; ++pairId) {
00150                structureFactorDelta_(i, pairId) = 0;
00151             }
00152          }
00153  
00154          // Loop over molecules in species
00155          system().begin(speciesId_, molIter); 
00156          for ( ; molIter.notEnd(); ++molIter) {
00157 
00158             // Set all Fourier modes to zero
00159             for (i = 0; i < nWave_; ++i) {
00160                for (typeId = 0; typeId <= nAtomType_; ++typeId) {
00161                   fourierModes_(i, typeId) = 0;
00162                }
00163             }
00164  
00165             // Loop over all atoms in one molecule
00166             molIter->begin(atomIter); 
00167             for ( ; atomIter.notEnd(); ++atomIter) {
00168                position = atomIter->position();
00169                typeId   = atomIter->typeId();
00170  
00171                // Loop over wavevectors
00172                for (i = 0; i < nWave_; ++i) {
00173                   product = position.dot(waveVectors_[i]);
00174                   expFactor = exp( product*Constants::Im );
00175                   fourierModes_(i, typeId)     += expFactor;
00176                   fourierModes_(i, nAtomType_) += expFactor;
00177                }
00178   
00179             }
00180 
00181             // Increment structure factors
00182             for (i = 0; i < nWave_; ++i) {
00183                for (j = 0; j < nAtomTypeIdPair_; ++j) {
00184                   for (k = 0; k < 2; ++k) {
00185                      typeId = atomTypeIdPairs_[j][k];
00186                      if (typeId >= 0) {
00187                         rho[k] = fourierModes_(i, typeId);
00188                      } else {
00189                         rho[k] = fourierModes_(i, nAtomType_);
00190                      }
00191                   }
00192                   rho[0] = std::conj(rho[0]);
00193                   dS = std::real(rho[0]*rho[1])/volume;
00194                   structureFactors_(i, j) += dS;
00195                   structureFactorDelta_(i,j) += dS;
00196                }
00197             }
00198 
00199          }
00200 
00201          ++nSample_;
00202       }
