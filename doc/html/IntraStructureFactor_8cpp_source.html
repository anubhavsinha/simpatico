129  /*
130  * Increment Structure Factor
131  */
132  void IntraStructureFactor::sample(long iStep)
133  {
134  if (isAtInterval(iStep)) {
135 
136  Vector position;
137  std::complex<double> rho[2];
138  std::complex<double> expFactor;
139  double volume = system().boundary().volume();
140  double product, dS;
141  System::ConstMoleculeIterator molIter;
142  Molecule::ConstAtomIterator atomIter;
143  int typeId, i, j, k;
144 
145  makeWaveVectors();
146 
147  // Set all Deltas to zero
148  for (i = 0; i < nWave_; ++i) {
149  for (int pairId = 0; pairId < nAtomTypeIdPair_; ++pairId) {
150  structureFactorDelta_(i, pairId) = 0;
151  }
152  }
153 
154  // Loop over molecules in species
155  system().begin(speciesId_, molIter);
156  for ( ; molIter.notEnd(); ++molIter) {
157 
158  // Set all Fourier modes to zero
159  for (i = 0; i < nWave_; ++i) {
160  for (typeId = 0; typeId <= nAtomType_; ++typeId) {
161  fourierModes_(i, typeId) = 0;
162  }
163  }
164 
165  // Loop over all atoms in one molecule
166  molIter->begin(atomIter);
167  for ( ; atomIter.notEnd(); ++atomIter) {
168  position = atomIter->position();
169  typeId = atomIter->typeId();
170 
171  // Loop over wavevectors
172  for (i = 0; i < nWave_; ++i) {
173  product = position.dot(waveVectors_[i]);
174  expFactor = exp( product*Constants::Im );
175  fourierModes_(i, typeId) += expFactor;
176  fourierModes_(i, nAtomType_) += expFactor;
177  }
178 
179  }
180 
181  // Increment structure factors
182  for (i = 0; i < nWave_; ++i) {
183  for (j = 0; j < nAtomTypeIdPair_; ++j) {
184  for (k = 0; k < 2; ++k) {
185  typeId = atomTypeIdPairs_[j][k];
186  if (typeId >= 0) {
187  rho[k] = fourierModes_(i, typeId);
188  } else {
189  rho[k] = fourierModes_(i, nAtomType_);
190  }
191  }
192  rho[0] = std::conj(rho[0]);
193  dS = std::real(rho[0]*rho[1])/volume;
194  structureFactors_(i, j) += dS;
195  structureFactorDelta_(i,j) += dS;
196  }
197  }
198 
199  }
200 
201  ++nSample_;
202  }
