00163    /*
00164    * Generate a random bond length chosen from an equilibrium distribution for
00165    * randomly oriented bonds.
00166    *
00167    * Algorithm:
00168    *
00169    *   While trial not accepted:
00170    *
00171    *      -Choose random trial vector of squared length rSq from the equilibrium
00172    *       distribution for a harmonic spring with the same spring constant.
00173    *      -Reject if rSq > r0Sq
00174    *      -Accept with probability exp[-beta[U_Fene(r) - U_Harmonic(r)].
00175    */
00176    double FeneBond::randomBondLength(
00177                          Random *random, double beta, int type) const
00178    {
00179 
00180       double x, y, z, sigSq, rSq, eHarm, eFene, ratio;
00181       sigSq = 1.0/(beta*kappa_[type]);
00182 
00183       // Generate trials until one is accepted
00184       for (;;) {
00185          x = random->gaussian();
00186          y = random->gaussian();
00187          z = random->gaussian();
00188          rSq = sqrt(x*x + y*y + z*z)*sigSq;
00189          if (rSq >= r0Sq_[type]) continue;
00190          eHarm = 0.5*kappa_[type]*rSq;
00191          eFene = energy(rSq, type);
00192          ratio = exp(-beta*(eFene - eHarm));
00193          if (random->uniform(0.0,1.0) < ratio) {
00194             return sqrt(rSq);
00195          };
00196       }
00197    }
