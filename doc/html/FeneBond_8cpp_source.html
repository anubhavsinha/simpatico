163  /*
164  * Generate a random bond length chosen from an equilibrium distribution for
165  * randomly oriented bonds.
166  *
167  * Algorithm:
168  *
169  * While trial not accepted:
170  *
171  * -Choose random trial vector of squared length rSq from the equilibrium
172  * distribution for a harmonic spring with the same spring constant.
173  * -Reject if rSq > r0Sq
174  * -Accept with probability exp[-beta[U_Fene(r) - U_Harmonic(r)].
175  */
176  double FeneBond::randomBondLength(
177  Random *random, double beta, int type) const
178  {
179 
180  double x, y, z, sigSq, rSq, eHarm, eFene, ratio;
181  sigSq = 1.0/(beta*kappa_[type]);
182 
183  // Generate trials until one is accepted
184  for (;;) {
185  x = random->gaussian();
186  y = random->gaussian();
187  z = random->gaussian();
188  rSq = sqrt(x*x + y*y + z*z)*sigSq;
189  if (rSq >= r0Sq_[type]) continue;
190  eHarm = 0.5*kappa_[type]*rSq;
191  eFene = energy(rSq, type);
192  ratio = exp(-beta*(eFene - eHarm));
193  if (random->uniform(0.0,1.0) < ratio) {
194  return sqrt(rSq);
195  };
196  }
197  }
