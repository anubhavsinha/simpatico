00071    * Build the PairList, i.e., populate it with atom pairs.
00072    */
00073    void PairList::build(CellList& cellList, bool reverseUpdateFlag)
00074    {
00075       // Precondition
00076       assert(isAllocated());
00077  
00078       Cell::NeighborArray neighbors;
00079       double cutoffSq;
00080       const Cell* cellPtr;
00081       CellAtom* atom1Ptr;
00082       CellAtom* atom2Ptr;
00083       Mask* maskPtr;
00084       Vector dr;
00085       int na;                 // number of atoms in this cell
00086       int nn;                 // number of neighbors for a cell
00087       int i, j;
00088       bool hasNeighbor;
00089   
00090       // Set maximum squared-separation for pairs in Pairlist
00091       cutoffSq = cutoff_*cutoff_;
00092    
00093       // Initialize counters for primary atoms and neighbors
00094       atom1Ptrs_.clear();
00095       atom2Ptrs_.clear();
00096       first_.clear();
00097       first_.append(0);
00098 
00099       // Copy positions and ids into cell list
00100       cellList.update();
00101    
00102       // Find all neighbors (cell list)
00103       cellPtr = cellList.begin();
00104       while (cellPtr) {
00105          na = cellPtr->nAtom(); // # of atoms in cell
00106 
00107          if (na) {
00108             cellPtr->getNeighbors(neighbors, reverseUpdateFlag);
00109             nn = neighbors.size();
00110    
00111             // Loop over primary atoms (atom1) in primary cell
00112             for (i = 0; i < na; ++i) {
00113                atom1Ptr = neighbors[i];
00114                maskPtr  = atom1Ptr->maskPtr();
00115    
00116                // Loop over secondary atoms
00117                hasNeighbor = false;
00118                for (j = i + 1; j < nn; ++j) {
00119                   atom2Ptr = neighbors[j];
00120                   dr.subtract(atom2Ptr->position(), atom1Ptr->position()); 
00121                   if (dr.square() < cutoffSq && !maskPtr->isMasked(atom2Ptr->id())) {
00122                      atom2Ptrs_.append(atom2Ptr->ptr());
00123                      hasNeighbor = true;
00124                   }
00125                }
00126 
00127                // Complete processing of atom1.
00128                if (hasNeighbor) {
00129                   atom1Ptrs_.append(atom1Ptr->ptr());
00130                   first_.append(atom2Ptrs_.size());
00131                }
00132 
00133             } // for ia 
00134          } // if (na)
00135 
00136          // Advance to next cell in a linked list
00137          cellPtr = cellPtr->nextCellPtr();
00138       }
00139 
00140       // Postconditions
00141       if (atom1Ptrs_.size()) {
00142          if (first_.size() != atom1Ptrs_.size() + 1) {
00143             UTIL_THROW("Array size problem");
00144          }
00145          if (first_[0] != 0) {
00146             UTIL_THROW("Incorrect first element of first_");
00147          }
00148          if (first_[atom1Ptrs_.size()] != atom2Ptrs_.size()) {
00149             UTIL_THROW("Incorrect last element of first_");
00150          }
00151       }
00152 
00153       // Increment buildCounter_= number of times the list has been built.
00154       ++buildCounter_;
00155  
00156       // Increment maxima
00157       if (atom1Ptrs_.size() > maxNAtomLocal_) {
00158          maxNAtomLocal_ = atom1Ptrs_.size();
00159       }
00160       if (atom2Ptrs_.size() > maxNPairLocal_) {
00161          maxNPairLocal_ = atom2Ptrs_.size();
00162       }
00163    }
