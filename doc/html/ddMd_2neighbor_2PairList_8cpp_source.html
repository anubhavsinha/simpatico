00097       // Initialize counters for primary atoms and neighbors

00099       atom2Ptrs_.clear();

00102    
00103       // Find all neighbors (cell list)

00105       while (cellPtr) {
00106          cellPtr->getNeighbors(neighbors, reverseUpdateFlag);
00107          na = cellPtr->nAtom();
00108          nn = neighbors.size();
00109 
00110          // Loop over primary atoms (atom1) in primary cell
00111          for (i = 0; i < na; ++i) {
00112             atom1Ptr = neighbors[i];
00113             maskPtr  = &(atom1Ptr->mask());
00114             foundNeighbor = false;
00115 
00116             // Loop over secondary atoms (atom2) in primary cell
00117             for (j = 0; j < na; ++j) {
00118                atom2Ptr = neighbors[j];
00119                if (atom2Ptr > atom1Ptr) {
00120                   atom2Id  = atom2Ptr->id();
00121                   if (!maskPtr->isMasked(atom2Id)) {
00122                      dr.subtract(atom2Ptr->position(), atom1Ptr->position()); 
00123                      dRSq = dr.square();
00124                      if (dRSq < cutoffSq) {
00125                         // If first neighbor of atom1, add atom1 to atom1Ptrs_
00126                         if (!foundNeighbor) {
00127                            atom1Ptrs_.append(atom1Ptr);
00128                            foundNeighbor = true;
00129                         }
00130                         // Append 2nd atom to atom2Ptrs_[]
00131                         atom2Ptrs_.append(atom2Ptr);
00132                      }
00133                   }
00134                }
00135             }
00136 
00137             // Atoms in neighboring cells
00138             for (j = na; j < nn; ++j) {
00139                atom2Ptr = neighbors[j];
00140                atom2Id  = atom2Ptr->id();
00141                if (!maskPtr->isMasked(atom2Id)) {
00142                   dr.subtract(atom2Ptr->position(), atom1Ptr->position()); 
00143                   dRSq = dr.square();
00144                   if (dRSq < cutoffSq) {
00145                      // If first_ neighbor, record iAtomId in atom1Ptrs_
00146                      if (!foundNeighbor) {
00147                         atom1Ptrs_.append(atom1Ptr);
00148                         foundNeighbor = true;
00149                      }
00150                      // Append Id of 2nd atom in pair to atom2Ptrs_[]
00151                      atom2Ptrs_.append(atom2Ptr);
00152                  }
00153               }
00154    
00155             }
00156   
00157             // When finished with atom1, set next element of first_ array.
00158             if (foundNeighbor) {
00159                first_.append(atom2Ptrs_.size());
00160             }
00161          }
00162 
00163          // Advance to next cell in a linked list
00164          cellPtr = cellPtr->nextCellPtr();
00165       }
00166 
00167       // Postconditions
00168       if (atom1Ptrs_.size()) {
00169          if (first_.size() != atom1Ptrs_.size() + 1) {
00170             UTIL_THROW("Array size problem");
00171          }
00172          if (first_[0] != 0) {
00173             UTIL_THROW("Incorrect first element of first_");
00174          }
00175          if (first_[atom1Ptrs_.size()] != atom2Ptrs_.size()) {
00176             UTIL_THROW("Incorrect last element of first_");
00177          }
00178       }
00179 
00180       // Increment buildCounter_= number of times the list has been built.
00181       ++buildCounter_;
00182  
00183       // Increment maxima
00184       if (atom1Ptrs_.size() > maxNAtomLocal_) {
00185          maxNAtomLocal_ = atom1Ptrs_.size();
00186       }
00187       if (atom2Ptrs_.size() > maxNPairLocal_) {
00188          maxNPairLocal_ = atom2Ptrs_.size();
00189       }
