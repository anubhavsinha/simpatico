73  /*
74  * Build the PairList, i.e., populate it with atom pairs.
75  */
76  void PairList::build(CellList& cellList, bool reverseUpdateFlag)
77  {
78  // Precondition
79  assert(isAllocated());
80 
81  Cell::NeighborArray neighbors;
82  double cutoffSq;
83  const Cell* cellPtr;
84  CellAtom* atom1Ptr;
85  CellAtom* atom2Ptr;
86  Mask* maskPtr;
87  Vector dr;
88  int na; // number of atoms in this cell
89  int nn; // number of neighbors for a cell
90  int i, j;
91  bool hasNeighbor;
92 
93  // Set maximum squared-separation for pairs in Pairlist
94  cutoffSq = cutoff_*cutoff_;
95 
96  // Initialize counters for primary atoms and neighbors
97  atom1Ptrs_.clear();
98  atom2Ptrs_.clear();
99  first_.clear();
100  first_.append(0);
101 
102  // Copy positions and ids into cell list
103  cellList.update();
104 
105  // Find all neighbors (cell list)
106  cellPtr = cellList.begin();
107  while (cellPtr) {
108  na = cellPtr->nAtom(); // # of atoms in cell
109 
110  if (na) {
111  cellPtr->getNeighbors(neighbors, reverseUpdateFlag);
112  nn = neighbors.size();
113 
114  // Loop over primary atoms (atom1) in primary cell
115  for (i = 0; i < na; ++i) {
116  atom1Ptr = neighbors[i];
117  maskPtr = atom1Ptr->maskPtr();
118 
119  // Loop over secondary atoms
120  hasNeighbor = false;
121  for (j = i + 1; j < nn; ++j) {
122  atom2Ptr = neighbors[j];
123  dr.subtract(atom2Ptr->position(), atom1Ptr->position());
124  if (dr.square() < cutoffSq && !maskPtr->isMasked(atom2Ptr->id())) {
125  atom2Ptrs_.append(atom2Ptr->ptr());
126  hasNeighbor = true;
127  }
128  }
129 
130  // Complete processing of atom1.
131  if (hasNeighbor) {
132  atom1Ptrs_.append(atom1Ptr->ptr());
133  first_.append(atom2Ptrs_.size());
134  }
135 
136  } // for ia
137  } // if (na)
138 
139  // Advance to next cell in a linked list
140  cellPtr = cellPtr->nextCellPtr();
141  }
142 
143  // Postconditions
144  if (atom1Ptrs_.size()) {
145  if (first_.size() != atom1Ptrs_.size() + 1) {
146  UTIL_THROW("Array size problem");
147  }
148  if (first_[0] != 0) {
149  UTIL_THROW("Incorrect first element of first_");
150  }
151  if (first_[atom1Ptrs_.size()] != atom2Ptrs_.size()) {
152  UTIL_THROW("Incorrect last element of first_");
153  }
154  }
155 
156  // Increment buildCounter_= number of times the list has been built.
157  ++buildCounter_;
158 
159  // Increment maxima
160  if (atom1Ptrs_.size() > maxNAtomLocal_) {
161  maxNAtomLocal_ = atom1Ptrs_.size();
162  }
163  if (atom2Ptrs_.size() > maxNPairLocal_) {
164  maxNPairLocal_ = atom2Ptrs_.size();
165  }
166  }
