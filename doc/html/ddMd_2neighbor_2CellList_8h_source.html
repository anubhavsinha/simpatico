389  /*
390  * Identify the cell for an Atom, based on its position.
391  */
392  inline int CellList::cellIndexFromPosition(const Vector& position) const
393  {
394  IntVector r;
395  for (int i = 0; i < Dimension; ++i) {
396  if (position[i] < lowerOuter_[i]) {
397  return -1;
398  }
399  if (position[i] > upperOuter_[i]) {
400  return -1;
401  }
402  if (position[i] < lower_[i]) {
403  r[i] = 0;
404  } else {
405  r[i] = int( (position[i] - lower_[i])/ cellLengths_[i] ) + 1;
406  }
407  assert(r[i] < grid_.dimension(i));
408  }
409  return grid_.rank(r);
410  }

413  * Add an Atom to the appropriate cell, based on its position.
414  */
415  inline void CellList::placeAtom(Atom &atom)
416  {
417  // Preconditon
418  assert(nAtom_ < tags_.capacity());
419 
420  int rank = cellIndexFromPosition(atom.position());
421  if (rank >= 0) {
422  tags_[nAtom_].cellRank = rank;
423  tags_[nAtom_].handle = &atom;
424  cells_[rank].incrementCapacity();
425  ++nAtom_;
426  } else {
427  ++nReject_;
428  }
