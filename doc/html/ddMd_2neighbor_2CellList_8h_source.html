00382    inline int CellList::cellIndexFromPosition(const Vector& position) const
00383    {
00384       IntVector r;
00385       for (int i = 0; i < Dimension; ++i) {
00386          if (position[i] <= lowerOuter_[i]) {
00387             return -1;
00388          }
00389          if (position[i] >= upperOuter_[i]) {
00390             return -1;
00391          }
00392          r[i] = int( (position[i] - lowerOuter_[i])/ cellLengths_[i] );
00393          assert(r[i] < grid_.dimension(i));
00394       }
00395       return grid_.rank(r);
00396    }
