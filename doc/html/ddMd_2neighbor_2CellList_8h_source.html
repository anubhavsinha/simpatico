00391    * Identify the cell for an Atom, based on its position.
00392    */
00393    inline int CellList::cellIndexFromPosition(const Vector& position) const
00394    {
00395       IntVector r;
00396       for (int i = 0; i < Dimension; ++i) {
00397          if (position[i] < lowerOuter_[i]) {
00398             return -1;
00399          }
00400          if (position[i] > upperOuter_[i]) {
00401             return -1;
00402          }
00403          if (position[i] < lower_[i]) {
00404             r[i] = 0;
00405          } else {
00406             r[i] = int( (position[i] - lower_[i])/ cellLengths_[i] ) + 1;
00407          }
00408          assert(r[i] < grid_.dimension(i));
00409       }
00410       return grid_.rank(r);
00411    }
