00165    */
00166    void AtomMSD::sample(long iStep) 
00167    { 
00168       if (!isAtInterval(iStep)) return;
00169 
00170       // Confirm that nMolecule has remained constant, and nMolecule > 0.
00171       if (nMolecule_ <= 0) {
00172          UTIL_THROW("nMolecule <= 0");
00173       }
00174       if (nMolecule_ != system().nMolecule(speciesId_)) {
00175          UTIL_THROW("Number of molecules has changed.");
00176       }
00177 
00178       Vector     r;
00179       IntVector  shift;
00180       Vector     lengths = system().boundary().lengths();
00181       int        i, j;
00182       for (i = 0; i < nMolecule_; ++i) {
00183 
00184          r = system().molecule(speciesId_, i).atom(atomId_).position();
00185          system().boundary().shift(r);
00186 
00187          // Compare current r to previous position, oldPositions_[i]
00188          system().boundary().distanceSq(r, oldPositions_[i], shift);
00189 
00190          // If this atom crossed a boundary, increment its shift vector
00191          shifts_[i] += shift;
00192 
00193          // Reconstruct true position
00194          for (j = 0; j < Dimension; ++j) {
00195             truePositions_[i][j] = r[j] + shifts_[i][j]*lengths[j];
00196          }
00197 
00198          // Store current position in box for comparison to next one
00199          oldPositions_[i] = r;
00200   
00201       }
00202       accumulator_.sample(truePositions_);
00203 
00204    }
