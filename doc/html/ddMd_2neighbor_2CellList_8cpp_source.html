00188       // in a contiguous strip of cells for which at least some of the
00189       // cell lies within a cutoff distance of the primary cell.
00190       offsets_.clear();

00192

00196       strip.first  = 0;
00197       strip.second = 0;
00198       offsets_.append(strip);
