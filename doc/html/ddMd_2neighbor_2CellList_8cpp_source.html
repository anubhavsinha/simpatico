00145             for (p[2] = 1; p[2] < grid_.dimension(2) - 1; ++p[2]) {

00149                if (prevPtr) {

00151                } else {
00152                   begin_ = cellPtr;
00153                }
00154                prevPtr = cellPtr;

00156          }
00157       }
00158       cellPtr->setLastCell();
00159

00161       IntVector dmin, dmax;
00162       for (int i = 0; i < Dimension; ++i) {
00163          if (grid_.dimension(i) > 2) {
00164             dmin[i] = -1;

00166          } else if (grid_.dimension(i) == 2) {
00167             dmin[i] =  0;

00170             dmin[i] =  0;

00176       IntVector span;
