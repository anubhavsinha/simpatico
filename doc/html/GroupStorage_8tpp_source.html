00367    template <int N>
00368    #ifdef UTIL_MPI
00369    bool GroupStorage<N>::isValid(AtomStorage& atomStorage, 
00370                                  MPI::Intracomm& communicator,
00371                                  bool hasGhosts)
00372    #else
00373    bool GroupStorage<N>::isValid(AtomStorage& atomStorage, bool hasGhosts)
00374    #endif
00375    {
00376       int i;
00377       int atomId;
00378       int nAtom;  // Number of local atoms in particular group.
00379       int nGhost; // Number of local atoms in particular group.
00380       int nAtomGroup = 0; // Number of local atoms in all groups on processor
00381       Atom* atomPtr;
00382       ConstGroupIterator<N> groupIter;
00383 
00384       // Call simpler function that only checks storage data structures.
00385       isValid();
00386 
00387       // Loop over groups.
00388       begin(groupIter);
00389       for ( ; groupIter.notEnd(); ++groupIter) {
00390          nAtom = 0;
00391          nGhost = 0;
00392          for (i = 0; i < N; ++i) {
00393             atomId  = groupIter->atomId(i);
00394             if (atomId < 0 || atomId >= atomStorage.totalAtomCapacity()) {
00395                UTIL_THROW("Invalid atom id in Group");
00396             }
00397             atomPtr = groupIter->atomPtr(i);
00398             if (atomPtr) {
00399                if (atomPtr != atomStorage.find(atomId)) {
00400                   UTIL_THROW("Inconsistent non-null atom pointer in Group");
00401                }
00402                if (atomPtr->isGhost()) {
00403                   ++nGhost;
00404                } else {
00405                   ++nAtom;
00406                }
00407             } else {
00408                atomPtr = atomStorage.find(atomId);
00409                if (atomPtr != 0) {
00410                   if (atomPtr->isGhost()) {
00411                      if (hasGhosts) {
00412                           UTIL_THROW("Missing ghost atom");
00413                      }
00414                   } else {
00415                      UTIL_THROW("Missing local atom");
00416                   }
00417                }
00418             }
00419          }
00420          if (nAtom == 0) {
00421             UTIL_THROW("Empty group");
00422          }
00423          if (hasGhosts && (nAtom + nGhost) < N) {
00424             UTIL_THROW("Incomplete group");
00425          }
00426          nAtomGroup += nAtom;
00427       }
00428 
00429       // Count number distinct groups.
00430       #ifdef UTIL_MPI
00431       unsetNTotal();
00432       computeNTotal(communicator);
00433       #endif
00434 
00435       #ifdef UTIL_MPI
00436       // Count & return number of local atoms in groups on all processors.
00437       int nAtomGroupTotal;
00438       const int source = 0;
00439       communicator.Reduce(&nAtomGroup, &nAtomGroupTotal, 1, 
00440                           MPI::INT, MPI::SUM, source);
00441       if (communicator.Get_rank() == source) {
00442          if (!nTotal_.isSet()) {
00443             UTIL_THROW("nTotal not set");
00444          }
00445          if (nAtomGroupTotal != N*nTotal()) {
00446             std::cout << "nAtomGroupTotal = " << nAtomGroupTotal << std::endl;
00447             std::cout << "nTotal*N        = " << N*nTotal() << std::endl;
00448             UTIL_THROW("Discrepancy in number of local atoms in Group objects");
00449          }
00450       }
00451       #endif
00452

00454    }

00464    * Identify groups that span boundaries.
00465    *
00466    * This method is called by exchangeAtoms, after computing plans for
00467    * exchanging atoms, based on their position, but before exchanging
00468    * atoms, and before clearing ghosts from any previous exchange.
00469    *
00470    * Algorithm: Loop over all Group<N> objects in the storage, identify
00471    * groups that span boundaries of the processor domain associated with
00472    * each of 6 transfer directions (3 Cartesian directions, and transfer
00473    * "down" (j=0) and "up" (j=1) in each direction). This requires information
00474    * about positions of ghost as well as local atoms. For each boundary of 
00475    * the domain, identify atoms whose positions are "inside" and "outside".
00476    * Count ghost atoms very near the boundary as both inside and outside,
00477    * for safety. If a group has atoms both inside and outside a domain 
00478    * boundary, it is marked for sending in the associated communication 
00479    * step. 
00480    *
00481    * After calculating a ghost communication plan for each group, clear 
00482    * the pointers to all ghost atoms in the group. The exchangeAtoms 
00483    * method will clear the actual ghost atoms from the AtomStorage.
00484    */
00485    template <int N> void

00488                                        FMatrix<double, Dimension, 2>& outer, 
00489                                        IntVector& gridFlags)
00490    {

00493       Atom* atomPtr;
00494       int nIn, nOut, i, j, k;
00495       bool isComplete;
00496       bool choose;
00497 
00498       // Loop over groups
00499       begin(groupIter);
00500       for ( ; groupIter.notEnd(); ++groupIter) {
00501          groupIter->plan().clearFlags();
00502 
00503          isComplete = (groupIter->nPtr() == N); // Is this group complete?
00504 
00505          if (isComplete) {
00506 
00507             for (i = 0; i < Dimension; ++i) {
00508                if (gridFlags[i]) {
00509                   for (j = 0; j < 2; ++j) {
00510             
00511                      // Determine if Group may span boundary (i, j)
00512                      choose = false;
00513                      nIn = 0;
00514                      nOut = 0;
00515                      // Loop over atoms in group
00516                      for (k = 0; k < N; ++k) {
00517                         atomPtr = groupIter->atomPtr(k);
00518                         assert(atomPtr);
00519                         coordinate = atomPtr->position()[i];
00520                         if (atomPtr->isGhost()) {
00521                            if (j == 0) {
00522                               assert(inner(i, j) > bound(i, j));
00523                               if (coordinate < inner(i, j)) {
00524                                  ++nOut;
00525                               }
00526                               if (coordinate > outer(i, j)) {
00527                                  ++nIn;
00528                               }
00529                            } else { // if j = 1
00530                               assert(inner(i, j) < bound(i, j));
00531                               if (coordinate > inner(i, j)) {
00532                                  ++nOut;
00533                               }

00537                            }

00616    * Pack groups that contain atoms marked for exchange in this 
00617    * direction (direction i, j), and remove empty groups.
00618    *
00619    * Algorithm: Loop over groups. If the group contains one or

00626    {
00627       GroupIterator<N> groupIter;
00628       Atom* atomPtr;
00629       int k, nAtom;
00630       bool choose;
00631       emptyGroups_.clear();
00632 
00633       // Pack Groups
00634       buffer.beginSendBlock(Buffer::GROUP2 + N - 2);
00635       begin(groupIter);
00636       for ( ; groupIter.notEnd(); ++groupIter) {
00637          choose = false;
00638          nAtom = 0;
00639          for (k = 0; k < N; ++k) {
00640             atomPtr = groupIter->atomPtr(k);
00641             if (atomPtr) {
00642                if (atomPtr->plan().exchange(i, j)) {
00643                   choose = true;
00644                   groupIter->clearAtomPtr(k);
00645                } else {
00646                   ++nAtom;
00647                }
00648             }
00649          }
00650          if (nAtom == 0) {
00651             emptyGroups_.append(*groupIter);
00652          }
00653          if (choose) {
00654             groupIter->pack(buffer);
00655          }
00656       }
00657       buffer.endSendBlock();
00658 
00659       // Remove empty groups
00660       int nEmpty = emptyGroups_.size();
00661       for (int k = 0; k < nEmpty; ++k) {
00662          remove(&(emptyGroups_[k]));

00695    /*
00696    * Set ghost communication flags for all atoms in incomplete groups.
00697    *
00698    * Precondition: This is called by exchangeAtoms after exchanging atoms 
00699    * and groups between neighboring processors. At this point, there are
00700    * no ghosts atoms.
00701    *
00702    * Algorithm: Loop over all Group<N> objects in the group storage. 
00703    * For each group, check if the group is incomplete, implying that one or
00704    * more atoms in the group are owned by another processor. If the group 
00705    * is incomplete, loop over 6 transfer directions. For each direction,
00706    * if the group is marked for sending in that direction, set the ghost
00707    * ghost communication flag for transfer in that direction for every
