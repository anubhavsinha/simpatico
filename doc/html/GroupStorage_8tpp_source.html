00382    template <int N>
00383    #ifdef UTIL_MPI
00384    bool GroupStorage<N>::isValid(AtomStorage& atomStorage, 
00385                                  MPI::Intracomm& communicator,
00386                                  bool hasGhosts)
00387    #else
00388    bool GroupStorage<N>::isValid(AtomStorage& atomStorage, bool hasGhosts)
00389    #endif
00390    {
00391       int i;
00392       int atomId;
00393       int nAtom;  // Number of local atoms in particular group.
00394       int nGhost; // Number of local atoms in particular group.
00395       int nAtomGroup = 0; // Number of local atoms in all groups on processor
00396       Atom* atomPtr;
00397       ConstGroupIterator<N> groupIter;
00398 
00399       // Call simpler function that only checks storage data structures.
00400       isValid();
00401 
00402       // Loop over groups.
00403       const AtomMap& atomMap = atomStorage.map();
00404       for (begin(groupIter); groupIter.notEnd(); ++groupIter) {
00405          nAtom = 0;
00406          nGhost = 0;
00407          for (i = 0; i < N; ++i) {
00408             atomId  = groupIter->atomId(i);
00409             if (atomId < 0 || atomId >= atomStorage.totalAtomCapacity()) {
00410                UTIL_THROW("Invalid atom id in Group");
00411             }
00412             atomPtr = groupIter->atomPtr(i);
00413             if (atomPtr) {
00414                if (atomPtr != atomMap.find(atomId)) {
00415                   UTIL_THROW("Inconsistent non-null atom pointer in Group");
00416                }
00417                if (atomPtr->isGhost()) {
00418                   ++nGhost;
00419                } else {
00420                   ++nAtom;
00421                }
00422             } else {
00423                atomPtr = atomMap.find(atomId);
00424                if (atomPtr != 0) {
00425                   if (atomPtr->isGhost()) {
00426                      if (hasGhosts) {
00427                         UTIL_THROW("Missing ghost atom");
00428                      }
00429                   } else {
00430                      UTIL_THROW("Missing local atom");
00431                   }
00432                }
00433             }
00434          }
00435          if (nAtom == 0) {
00436             UTIL_THROW("Empty group");
00437          }
00438          if (hasGhosts && (nAtom + nGhost) < N) {
00439             UTIL_THROW("Incomplete group");
00440          }
00441          nAtomGroup += nAtom;
00442       }
00443 
00444       // Count number distinct groups.
00445       #ifdef UTIL_MPI
00446       unsetNTotal();
00447       computeNTotal(communicator);
00448       #endif
00449 
00450       #ifdef UTIL_MPI
00451       // Count & return number of local atoms in groups on all processors.
00452       int nAtomGroupTotal;
00453       const int source = 0;
00454       communicator.Reduce(&nAtomGroup, &nAtomGroupTotal, 1, 
00455                           MPI::INT, MPI::SUM, source);
00456       if (communicator.Get_rank() == source) {
00457          if (!nTotal_.isSet()) {
00458             UTIL_THROW("nTotal not set");
00459          }
00460          if (nAtomGroupTotal != N*nTotal()) {
00461             Log::file() << "nAtomGroupTotal = " << nAtomGroupTotal << std::endl;
00462             Log::file() << "nTotal*N        = " << N*nTotal() << std::endl;
00463             UTIL_THROW("Discrepancy in number of local atoms in Group objects");
00464          }
00465       }
00466       #endif
00467 
00468       return true;
00469    }

00478    /*
00479    * Identify groups that span boundaries.
00480    *
00481    * This function is called by exchangeAtoms, after computing plans for
00482    * exchanging atoms, based on their position, but before exchanging
00483    * atoms, and before clearing ghosts from any previous exchange.
00484    *
00485    * Algorithm: Loop over all Group<N> objects in the storage, identify
00486    * groups that span boundaries of the processor domain associated with
00487    * each of 6 transfer directions (3 Cartesian directions, and transfer
00488    * "down" (j=0) and "up" (j=1) in each direction). This requires information
00489    * about positions of ghost as well as local atoms. For each boundary of 
00490    * the domain, identify atoms whose positions are "inside" and "outside".
00491    * Count ghost atoms very near the boundary as both inside and outside,
00492    * for safety. If a group has atoms both inside and outside a domain 
00493    * boundary, it is marked for sending in the associated communication 
00494    * step. 
00495    *
00496    * After calculating a ghost communication plan for each group, clear 
00497    * the pointers to all ghost atoms in the group. The exchangeAtoms 
00498    * function will clear the actual ghost atoms from the AtomStorage.
00499    */
00500    template <int N> void 
00501    GroupStorage<N>::markSpanningGroups(FMatrix<double, Dimension, 2>& bound, 
00502                                        FMatrix<double, Dimension, 2>& inner, 
00503                                        FMatrix<double, Dimension, 2>& outer, 
00504                                        IntVector& gridFlags)
00505    {
00506       double coordinate;
00507       GroupIterator<N> groupIter;
00508       Atom* atomPtr;
00509       int nIn, nOut, i, j, k;
00510       bool isComplete;
00511       bool choose;
00512 
00513       // Loop over groups
00514       begin(groupIter);
00515       for ( ; groupIter.notEnd(); ++groupIter) {
00516          groupIter->plan().clearFlags();
00517 
00518          isComplete = (groupIter->nPtr() == N); // Is this group complete?
00519 
00520          if (isComplete) {
00521 
00522             for (i = 0; i < Dimension; ++i) {
00523                if (gridFlags[i]) {
00524                   for (j = 0; j < 2; ++j) {
00525             
00526                      // Determine if Group may span boundary (i, j)
00527                      choose = false;
00528                      nIn = 0;
00529                      nOut = 0;
00530                      // Loop over atoms in group
00531                      for (k = 0; k < N; ++k) {
00532                         atomPtr = groupIter->atomPtr(k);
00533                         assert(atomPtr);
00534                         coordinate = atomPtr->position()[i];
00535                         if (atomPtr->isGhost()) {
00536                            if (j == 0) {
00537                               assert(inner(i, j) > bound(i, j));
00538                               if (coordinate < inner(i, j)) {
00539                                  ++nOut;
00540                               }
00541                               if (coordinate > outer(i, j)) {
00542                                  ++nIn;
00543                               }
00544                            } else { // if j = 1
00545                               assert(inner(i, j) < bound(i, j));
00546                               if (coordinate > inner(i, j)) {
00547                                  ++nOut;
00548                               }
00549                               if (coordinate < outer(i, j)) {
00550                                  ++nIn;

00552                            }

00558                            }

00569                   } // end for j = 0, 1

00629    * Pack groups for exchange.
00630    *
00631    * Pack groups that contain atoms marked for exchange in this 
00632    * direction (direction i, j), and remove empty groups.
00633    *
00634    * Algorithm: Loop over groups. If the group contains one or
00635    * more atoms that are marked for exchange in direction i, j, 
00636    * pack the group for sending along. Remove empty groups in
00637    * a separate loop.
00638    */
00639    template <int N>
00640    void GroupStorage<N>::pack(int i, int j, Buffer& buffer)
00641    {
00642       GroupIterator<N> groupIter;
00643       Atom* atomPtr;
00644       int k, nAtom;
00645       bool choose;
00646       emptyGroups_.clear();
00647 
00648       // Pack Groups
00649       buffer.beginSendBlock(Buffer::GROUP2 + N - 2);
00650       begin(groupIter);
00651       for ( ; groupIter.notEnd(); ++groupIter) {
00652          choose = false;
00653          nAtom = 0;
00654          for (k = 0; k < N; ++k) {
00655             atomPtr = groupIter->atomPtr(k);
00656             if (atomPtr) {
00657                if (atomPtr->plan().exchange(i, j)) {
00658                   choose = true;
00659                   groupIter->clearAtomPtr(k);
00660                } else {
00661                   ++nAtom;
00662                }
00663             }
00664          }
00665          if (nAtom == 0) {
00666             emptyGroups_.append(*groupIter);
00667          }
00668          if (choose) {
00669             groupIter->pack(buffer);
00670          }
00671       }
00672       buffer.endSendBlock();
00673 
00674       // Remove empty groups
00675       int nEmpty = emptyGroups_.size();
00676       for (int k = 0; k < nEmpty; ++k) {
00677          remove(&(emptyGroups_[k]));
00678       }

00686    {

00688       Group<N>* oldGroupPtr;
00689       const AtomMap& atomMap = atomStorage.map();
00690       int groupId;
00691 
00692       buffer.beginRecvBlock();
00693       while (buffer.recvSize() > 0) {
00694          newGroupPtr = newPtr();
00695          newGroupPtr->unpack(buffer);
00696          groupId = newGroupPtr->id();
00697          oldGroupPtr = find(groupId);
00698          if (oldGroupPtr) {
00699             returnPtr();
00700             atomMap.findGroupLocalAtoms(*oldGroupPtr);
00701          } else {
00702             add();
00703             atomMap.findGroupLocalAtoms(*newGroupPtr);
00704          }

00712    * Set ghost communication flags for all atoms in incomplete groups.
00713    *
00714    * Precondition: This is called by exchangeAtoms after exchanging atoms 
00715    * and groups between neighboring processors. At this point, there are
00716    * no ghosts atoms.
00717    *
00718    * Algorithm: Loop over all Group<N> objects in the group storage. 
00719    * For each group, check if the group is incomplete, implying that one or
00720    * more atoms in the group are owned by another processor. If the group 
00721    * is incomplete, loop over 6 transfer directions. For each direction,
00722    * if the group is marked for sending in that direction, set the ghost
00723    * ghost communication flag for transfer in that direction for every 
00724    * local atom in the group. Also add each such atom to sendArray(i, j).
00725    *
00726    * Note: If a group is incomplete on this processor, and thus
00727    * contains atoms owned by other processors, the algorithm assumes
00728    * that the ghost communication flag for each atom will be set by
00729    * the processor that owns the atom.

00745          #ifdef UTIL_DEBUG

00776          #endif // ifdef DDMD_GROUP_STORAGE_DEBUG

00779          // If this group is incomplete, set ghost flags for atoms 
00780          nAtom = groupIter->nPtr();

00782             for (i = 0; i < Dimension; ++i) {
00783                if (gridFlags[i]) {

00796                         }
00797                      }
00798                   }
00799                }
00800             }
00801          }
