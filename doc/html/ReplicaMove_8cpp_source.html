152  * Perform replica exchange move.
153  */
154  bool ReplicaMove::move()
155  {
156  MPI::Request request[4];
157  MPI::Status status;
158  System::MoleculeIterator molIter;
159  Molecule::AtomIterator atomIter;
160  int iA;
161  int recvPt, sendPt;
162 
163  DArray<int> permutation;
164  permutation.allocate(nProcs_);
165 
166  // Gather all derivatives of the perturbation Hamiltonians and parameters on processor with rank 0
167  DArray<double> myDerivatives;
168  myDerivatives.allocate(nParameters_);
169  DArray<double> myParameters;
170  myParameters.allocate(nParameters_);
171 
172  for (int i=0; i< nParameters_; i++) {
173  myDerivatives[i] = system().perturbation().derivative(i);
174  myParameters[i] = system().perturbation().parameter(i);
175  }
176 
177  int size = 0;
178  size += memorySize(myDerivatives);
179  size += memorySize(myParameters);
180 
181  if (myId_ != 0) {
182 
183  MemoryOArchive sendCurrent;
184  sendCurrent.allocate(size);
185 
186  sendCurrent << myDerivatives;
187  sendCurrent << myParameters;
188 
189  sendCurrent.send(*communicatorPtr_, 0);
190  } else {
191  DArray< DArray<double> > allDerivatives;
192  DArray< DArray<double> > allParameters;
193  allDerivatives.allocate(nProcs_);
194  allParameters.allocate(nProcs_);
195 
196  allDerivatives[0].allocate(nParameters_);
197  allDerivatives[0] = myDerivatives;
198  allParameters[0].allocate(nParameters_);
199  allParameters[0] = myParameters;
200 
201  for (int i = 1; i<nProcs_; i++) {
202  MemoryIArchive recvPartner;
203  recvPartner.allocate(size);
204  recvPartner.recv(*communicatorPtr_, i);
205  allDerivatives[i].allocate(nParameters_);
206  allParameters[i].allocate(nParameters_);
207  recvPartner >> allDerivatives[i];
208  recvPartner >> allParameters[i];
209  }
210 
211  // Now we have the complete matrix U_ij = u_i(x_j), permutate nsampling steps according
212  // to acceptance criterium
213 
214  // start with identity permutation
215  for (int i = 0; i < nProcs_; i++)
216  permutation[i] = i;
217 
218  for (int n =0; n < nSampling_; n++) {
219  swapAttempt_++;
220  // choose a pair i,j, i!= j at random
221  int i = system().simulation().random().uniformInt(0,nProcs_);
222  int j = system().simulation().random().uniformInt(0,nProcs_-1);
223  if (i<=j) j++;
224 
225  // apply acceptance criterium
226  double weight = 0;
227  for (int k = 0; k < nParameters_; k++) {
228  double deltaDerivative = allDerivatives[i][k] - allDerivatives[j][k];
229  // the permutations operate on the states (the perturbation parameters)
230  weight += (allParameters[permutation[j]][k] - allParameters[permutation[i]][k])*deltaDerivative;
231  }
232  double exponential = exp(-weight);
233  int accept = system().simulation().random(). metropolis(exponential) ? 1 : 0;
234 
235  if (accept) {
236  swapAccept_++;
237  // swap states of pair i,j
238  int tmp = permutation[i];
239  permutation[i] = permutation[j];
240  permutation[j] = tmp;
241  }
242  }
243 
244  // send exchange partner information to all other processors
245  for (int i = 0; i < nProcs_; i++) {
246  if (i != 0)
247  communicatorPtr_->Send(&permutation[i], 1, MPI::INT, i, 0);
248  else
249  sendPt = permutation[i];
250 
251  if (permutation[i] != 0)
252  communicatorPtr_->Send(&i, 1, MPI::INT, permutation[i], 1);
253  else
254  recvPt = i;
255  }
256  }
257 
258 
259  if (myId_ != 0) {
260  // partner id to receive from
261  communicatorPtr_->Recv(&sendPt, 1, MPI::INT, 0, 0);
262  // partner id to send to
263  communicatorPtr_->Recv(&recvPt, 1, MPI::INT, 0, 1);
264  }
265 
266  if (recvPt == myId_ || sendPt == myId_) {
267  // no exchange necessary
268  outputFile_ << sendPt << std::endl;
269  return true;
270  }
271 
272  assert(recvPt != myId_ && sendPt != myId_);
273 
274  Vector myBoundary;
275  myBoundary = system().boundary().lengths();
276 
277  Vector ptBoundary;
278 
279  // Accomodate new boundary dimensions.
280  request[0] = communicatorPtr_->Irecv(&ptBoundary, 1,
281  MpiTraits<Vector>::type, recvPt, 1);
282 
283  // Send old boundary dimensions.
284  request[1] = communicatorPtr_->Isend(&myBoundary, 1,
285  MpiTraits<Vector>::type, sendPt, 1);
286 
287  request[0].Wait();
288  request[1].Wait();
289 
290  system().boundary().setOrthorhombic(ptBoundary);
291 
292  // Pack atomic positions and types.
293  iA = 0;
294  for (int iSpec=0; iSpec < system().simulation().nSpecies(); ++iSpec){
295  for (system().begin(iSpec, molIter); molIter.notEnd(); ++molIter){
296  for (molIter->begin(atomIter); atomIter.notEnd(); ++atomIter) {
297  myPositionPtr_[iA] = atomIter->position();
298  iA++;
299  }
300  }
301  }
302 
303  // Accomodate new configuration.
304  request[2] = communicatorPtr_->Irecv(ptPositionPtr_, iA,
305  MpiTraits<Vector>::type, recvPt, 2);
306 
307  // Send old configuration.
308  request[3] = communicatorPtr_->Isend(myPositionPtr_, iA,
309  MpiTraits<Vector>::type, sendPt, 2);
310 
311  request[2].Wait();
312  request[3].Wait();
313 
314  // Adopt the new atomic positions.
315  iA = 0;
316  for (int iSpec=0; iSpec < system().simulation().nSpecies(); ++iSpec){
317  for (system().begin(iSpec, molIter); molIter.notEnd(); ++molIter){
318  for (molIter->begin(atomIter); atomIter.notEnd(); ++atomIter){
319  atomIter->position() = ptPositionPtr_[iA];
320  ++iA;
321  }
322  }
323  }
324 
325 
326  // Notify component observers.
327  sendRecvPair partners;
328  partners[0] = sendPt;
329  partners[1] = recvPt;
330  Notifier<sendRecvPair>::notifyObservers(partners);
331 
332  // Log information about exchange partner to file
333  outputFile_ << sendPt << std::endl;
334 
335  return true;
336 
337  }
