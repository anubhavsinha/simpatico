137  /*
138  * Output final results to file.
139  */
140  void RadialDistribution::output(std::ostream& out)
141  {
142  double r, rho, prefactor, dV, hist, integral;
143  prefactor = 4.0*3.14159265359/3.0;
144  prefactor = prefactor*binWidth_*binWidth_*binWidth_;
145  integral = 0.0;
146  for (int i=0; i < nBin_; ++i) {
147  r = binWidth_*(double(i) + 0.5);
148  dV = prefactor*double(3*i*i + 3*i + 1);
149  hist = double(histogram_[i])/double(nSnapshot_);
150  rho = hist/(dV*norm_);
151  out << Dbl(r, 18, 8) << Dbl(rho, 18, 8);
152  if (outputIntegral_) {
153  integral += (hist - dV*norm_)/norm_;
154  out << Dbl(integral, 18, 8);
155  }
156  out << std::endl;
157  }
158  }
