157  /*
158  * Allocate send and recv buffers (private method).
159  *
160  * This method uses values of atomCapacity_ and ghostCapacity_ that
161  * must have been set previously. It is called by allocate(int, int)
162  * and readParameters() to do the actual allocation.
163  */
164  void Buffer::allocate()
165  {
166 
167  // Preconditions
168  if (atomCapacity_ <= 0) {
169  UTIL_THROW("atomCapacity_ must be positive");
170  }
171  if (ghostCapacity_ <= 0) {
172  UTIL_THROW("ghostCapacity_ must be positive");
173  }
174  if (isAllocated()) {
175  UTIL_THROW("Buffer cannot be re-allocated");
176  }
177 
178  // Capacity in bytes send and receive buffers. This is maximum of
179  // the buffer space required by the local atoms and ghost atoms.
180  int atomDataSize = atomCapacity_*atomSize();
181  int ghostDataSize = ghostCapacity_*ghostSize();
182  if (atomDataSize > ghostDataSize) {
183  dataCapacity_ = atomDataSize;
184  ghostCapacity_ = dataCapacity_/ghostSize();
185  } else {
186  dataCapacity_ = ghostDataSize;
187  atomCapacity_ = dataCapacity_/atomSize();
188  }
189 
190  // Leave space for a 4 byte header
191  bufferCapacity_ += dataCapacity_ + 4 * sizeof(int);
192 
193  // Allocate memory for the send buffer
194  sendBufferBegin_ = new char[bufferCapacity_];
195  if (sendBufferBegin_ == 0) {
196  UTIL_THROW("Error: memory not allocated for send buffer");
197  }
198  sendBufferEnd_ = sendBufferBegin_ + bufferCapacity_;
199 
200  // Allocate memory for the receive buffer
201  recvBufferBegin_ = new char[bufferCapacity_];
202  if (recvBufferBegin_ == 0) {
203  UTIL_THROW("Error: memory not allocated for recv buffer");
204  }
205  recvBufferEnd_ = recvBufferBegin_ + bufferCapacity_;
206 
207  recvPtr_ = recvBufferBegin_;
208  }

328  void Buffer::unpackAtom(Atom& atom)
329  {
330  if (recvType_ != (int)ATOM) {
331  UTIL_THROW("Receive type is not ATOM");

339  unpack(i);

345  unpack<Vector>(atom.position());
346  unpack<Vector>(atom.velocity());
347

349  unsigned int ui;
350  unpack(ui);

354  Mask& mask = atom.mask();
355  mask.clear();

357  unpack(size);
358  for (int j = 0; j < size; ++j) {
359  unpack(i);

536  * Send and receive buffer.
537  */
538  void Buffer::sendRecv(MPI::Intracomm& comm, int source, int dest)
539  {
540 
541  MPI::Request request[2];
542  int sendBytes = 0;
543  int myRank = comm.Get_rank();
544  int comm_size = comm.Get_size();
545 
546  // Preconditions
547  if (dest > comm_size - 1 || dest < 0) {
548  UTIL_THROW("Destination rank out of bounds");
549  }
550  if (source > comm_size - 1 || source < 0) {
551  UTIL_THROW("Source rank out of bounds");
552  }
553  if (dest == myRank) {
554  UTIL_THROW("Destination and my rank are identical");
555  }
556  if (source == myRank) {
557  UTIL_THROW("Source and my rank are identical");
558  }
559 
560  // Start nonblocking receive.
561  request[0] = comm.Irecv(recvBufferBegin_, bufferCapacity_ ,
562  MPI::CHAR, source, 5);
563 
564  // Start nonblocking send.
565  sendBytes = sendPtr_ - sendBufferBegin_;
566  request[1] = comm.Isend(sendBufferBegin_, sendBytes , MPI::CHAR, dest, 5);
567 
568  // Wait for completion of receive.
569  request[0].Wait();
570  recvPtr_ = recvBufferBegin_;
571 
572  // Wait for completion of send.
573  request[1].Wait();
574 
575  // Update statistics.
576  if (sendBytes > maxSendLocal_) {
577  maxSendLocal_ = sendBytes;
578  }
579  }
