00473    * Check validity of Species. Return true if valid, or throw an Exception.
00474    */
00475    bool Species::isValid() const
00476    {
00477 
00478       if (reservoir_.isAllocated()) {
00479  
00480          // Check validity of reservoir (throws exception if invalid)
00481          reservoir_.isValid();
00482 
00483          // Check atomTypeIds array
00484          if (!atomTypeIds_.isAllocated()) {
00485             UTIL_THROW("AtomTypeIds_ not allocated");
00486          }
00487          if (!isMutable()) {
00488             for (int i = 0; i < nAtom_; ++i) {
00489                if (atomTypeIds_[i] < 0) {
00490                   UTIL_THROW("Negative AtomTypeIds_ in non-mutable Species");
00491                }
00492             }
00493          }
00494 
00495          // Loop over all bonds (if any) in speciesBonds_ array
00496          int  atomId, bondId, atomId0, atomId1, j;
00497          bool hasBond;
00498          for (bondId = 0; bondId < nBond_; ++bondId) {
00499 
00500             atomId0 = speciesBond(bondId).atomId(0);
00501             atomId1 = speciesBond(bondId).atomId(1);
00502    
00503             // Check that atomIds are in valid range and unequal
00504             if (atomId0 < 0 || atomId0 >= nAtom_) {
00505                UTIL_THROW("Invalid atomId0 in speciesBonds_");
00506             }
00507             if (atomId1 < 0 || atomId1 >= nAtom_) {
00508                UTIL_THROW("Invalid atomId1 in speciesBonds_");
00509             }
00510             if (atomId0 == atomId1) {
00511                UTIL_THROW("Equal atom ids in a SpeciesBond");
00512             }
00513 
00514             // Check that bondId is in atombondIdArrays_ for atomId0
00515             hasBond = false;
00516             for (j = 0; j < atomBondIdArrays_[atomId0].size(); ++j) {
00517                if (atomBondIdArrays_[atomId0][j] == bondId) {
00518                   hasBond = true;
00519                   break;
00520                } 
00521             }
00522             if (!hasBond) {
00523                UTIL_THROW("BondId missing from atomBondIdArrays_");
00524             }
00525 
00526             // Check that bondId is in atombondIdArrays_ for atomId0
00527             hasBond = false;
00528             for (j = 0; j < atomBondIdArrays_[atomId1].size(); ++j) {
00529                if (atomBondIdArrays_[atomId1][j] == bondId) {
00530                   hasBond = true;
00531                   break;
00532                } 
00533             }
00534             if (!hasBond) {
00535                UTIL_THROW("BondId missing from atomBondIdArrays_");
00536             }
00537 
00538          }
00539 
00540          // Loop over atomBondIdArrays for all atoms
00541          int bondCounter = 0;
00542          for (atomId = 0; atomId < nAtom_; ++atomId) {
00543             for (j = 0; j < atomBondIdArrays_[atomId].size(); ++j) {
00544                bondId = atomBondIdArrays_[atomId][j];
00545 
00546                if (bondId < 0 || bondId >= nBond_) 
00547                   UTIL_THROW("Bond index out of range");
00548 
00549                atomId0 = speciesBond(bondId).atomId(0);
00550                atomId1 = speciesBond(bondId).atomId(1);
00551                if (atomId != atomId0 && atomId != atomId1) 
00552                   UTIL_THROW("Inconsistency in atomBondId and bond");
00553                ++bondCounter;
00554                
00555             }
00556          }
00557          if (bondCounter != 2*nBond_) 
00558             UTIL_THROW("Inconsistency in total number of bonds");
00559 
00560          #ifdef INTER_ANGLE
00561          // Loop over all angles (if any) in speciesAngles_ array
00562          int  angleId, id, id2, atomId2(-1);
00563          bool hasAngle;
00564          for (angleId = 0; angleId < nAngle_; ++angleId) {
00565 
00566             for (id = 0; id < 3; ++id) {
00567                atomId = speciesAngle(angleId).atomId(id);
00568 
00569                // Check that atomIds are in valid range and unequal
00570                if (atomId < 0 || atomId >= nAtom_) {
00571                   UTIL_THROW("Invalid atomId in speciesAngles_");
00572                }
00573                for (id2 = id + 1; id2 < 3; ++id2) {
00574                   atomId2 = speciesAngle(angleId).atomId(id2);
00575                   if (atomId2 == atomId) {
00576                       UTIL_THROW("Equal atom ids in a SpeciesAngle");
00577                   }
00578                }
00579 
00580                // Check that angleId is in atomAngleIdArrays_ for atomId
00581                hasAngle = false;
00582                for (j = 0; j < atomAngleIdArrays_[atomId].size(); ++j) {
00583                   if (atomAngleIdArrays_[atomId][j] == angleId) {
00584                      hasAngle = true;
00585                      break;
00586                   } 
00587                }
00588                if (!hasAngle) {
00589                   UTIL_THROW("AngleId missing from atomAngleIdArrays_");
00590                }
00591             }
00592 
00593          }
00594 
00595          // Loop over atomAngleIdArrays for all atoms
00596          int angleCounter = 0;
00597          bool hasAtom;
00598          for (atomId = 0; atomId < nAtom_; ++atomId) {
00599             for (j = 0; j < atomAngleIdArrays_[atomId].size(); ++j) {
00600                angleId = atomAngleIdArrays_[atomId][j];
00601                if (angleId < 0 || angleId >= nAngle_) 
00602                   UTIL_THROW("Angle index out of range");
00603 
00604                hasAtom = false;
00605                for (id = 0; id < 3; ++id) {
00606                   atomId2 = speciesAngle(angleId).atomId(id);
00607                   if (atomId2 == atomId) {
00608                      hasAtom = true;
00609                      break;
00610                   }
00611                }
00612                if (!hasAtom) {
00613                   UTIL_THROW("Inconsistency in atomAngleId and angle");
00614                }
00615                ++angleCounter;
00616             }
00617          }
00618          if (angleCounter != 3*nAngle_) 
00619             UTIL_THROW("Inconsistency in total number of angles");
00620          #endif
00621 
00622          #ifdef INTER_DIHEDRAL
