187  void BennettsMethod::sample(long iStep)
188  {
189  if (!isAtInterval(iStep)) return;
190 
191  int myPort, upperPort;
192  MPI::Request requestFermi[2];
193  MPI::Status status;
194 
195  // Exchange perturbation parameters and differences
196  if (myId_ != 0 && myId_ != nProcs_ - 1) {
197 
198  myPort = myId_%2;
199  upperPort = upperId_%2;
200 
201  for (int i = 0; i < nParameter_; ++i) {
202  myParam_[i] = system().perturbation().parameter(i);
203  lowerParam_[i] = system().perturbation().parameter(i,lowerId_);
204  upperParam_[i] = system().perturbation().parameter(i,upperId_);
205  }
206 
207  myArg_ = system().perturbation().difference(upperParam_);
208  lowerArg_ = system().perturbation().difference(lowerParam_);
209 
210  myArg_ -= shift_;
211  lowerArg_ += lowerShift_;
212 
213  myFermi_ = 1/(1 + exp(myArg_));
214  lowerFermi_ = 1/(1 + exp(lowerArg_));
215 
216  requestFermi[0] = communicatorPtr_->Irecv(&upperFermi_, 1, MPI::DOUBLE, upperId_,
217  TagFermi[upperPort]);
218 
219  requestFermi[1] = communicatorPtr_->Isend(&lowerFermi_, 1, MPI::DOUBLE, lowerId_,
220  TagFermi[myPort]);
221 
222  // Synchronizing
223  requestFermi[0].Wait();
224  requestFermi[1].Wait();
225 
226  upperAccumulator_.sample(upperFermi_);
227  myAccumulator_.sample(myFermi_);
228 
229  outputFile_ << Dbl(myFermi_) << " " << Dbl(upperFermi_) << " ";
230  outputFile_ << std::endl;
231 
232  } else if (myId_ == 0) {
233 
234  myPort = myId_%2;
235  upperPort = upperId_%2;
236 
237  for (int i = 0; i < nParameter_; ++i) {
238  myParam_[i] = system().perturbation().parameter(i);
239  upperParam_[i] = system().perturbation().parameter(i, upperId_);
240  }
241 
242  myArg_ = system().perturbation().difference(upperParam_);
243  myArg_ -= shift_;
244 
245  myFermi_ = 1/(1 + exp(myArg_));
246 
247  requestFermi[0] = communicatorPtr_->Irecv(&upperFermi_, 1, MPI::DOUBLE, upperId_,
248  TagFermi[upperPort]);
249 
250  // Synchronizing
251  requestFermi[0].Wait();
252 
253  upperAccumulator_.sample(upperFermi_);
254  myAccumulator_.sample(myFermi_);
255 
256  outputFile_ << Dbl(myFermi_) << " " << Dbl(upperFermi_) << " ";
257  outputFile_ << std::endl;
258 
259  } else if (myId_ == nProcs_ - 1) {
260 
261  myPort = myId_%2;
262 
263  for (int i = 0; i < nParameter_; ++i) {
264  myParam_[i] = system().perturbation().parameter(i);
265  lowerParam_[i] = system().perturbation().parameter(i, lowerId_);
266  }
