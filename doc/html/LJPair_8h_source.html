00235    * Calculate interaction energy for a pair, as function of squared distance.
00236    */
00237    inline double LJPair::energy(double rsq, int i, int j) const 
00238    {
00239       double r6i;
00240       if (rsq < cutoffSq_[i][j]) {
00241          if (rsq < 0.6*sigmaSq_[i][j]) {
00242              rsq = 0.6*sigmaSq_[i][j];
00243          }
00244          r6i = sigmaSq_[i][j]/rsq;
00245          r6i = r6i*r6i*r6i;
00246          return 4.0*epsilon_[i][j]*(r6i*r6i - r6i) + ljShift_[i][j];
00247       } else {
00248          return 0.0;
00249       }
00250    }
