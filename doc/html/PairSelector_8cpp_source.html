00036    bool PairSelector::match(const Atom& atom1, const Atom& atom2) const
00037    {
00038 
00039       // Check molecular identities
00040       if (pairType_ == INTRA && &atom1.molecule() != &atom2.molecule() ) {
00041          return false;
00042       } else
00043       if (pairType_ == INTER && &atom1.molecule() == &atom2.molecule() ) {
00044          return false;
00045       }
00046 
00047       // Exclude case where atom1 and atom2 are identical
00048       if (&atom1 == &atom2) {
00049          return false;
00050       }
00051 
00052       // Check atom types
00053       if (atom1TypeId_ >= 0 && atom1.typeId() != atom1TypeId_) {
00054          return false;
00055       }
00056       if (atom2TypeId_ >= 0 && atom2.typeId() != atom2TypeId_) {
00057          return false;
00058       }
00059 
00060       // Check for double counting
00061       if (avoidDoubleCounting_) {
00062          if (atom1TypeId_ < 0 && atom2TypeId_ < 0) {
00063             if (atom1.id() > atom2.id()) {
00064                return false;
00065             }       
00066          } else 
00067          if (atom1.typeId() == atom2.typeId()) {
00068             if (atom1.id() > atom2.id()) {
00069                return false;
00070             }       
00071          }
00072       }
00073      
00074       // If the pair has passed all previous tests, accept it. 
00075       return true;
00076 
00077    }
