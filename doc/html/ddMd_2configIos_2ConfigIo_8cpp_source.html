244  * Private method to read Group<N> objects.

246  template <int N>
247  int ConfigIo::readGroups(std::ifstream& file,
248  const char* sectionLabel,
249  const char* nGroupLabel,
250  GroupDistributor<N>& distributor)
251  {
252  int nGroup; // Total number of groups in file
253  if (domain().isMaster()) {
254  file >> Label(sectionLabel);
255  file >> Label(nGroupLabel) >> nGroup;
256  Group<N>* groupPtr;
257  int i, j, k;
258  distributor.setup();
259  for (i = 0; i < nGroup; ++i) {
260  groupPtr = distributor.newPtr();
261  file >> *groupPtr;
262  for (j = 0; j < 2; ++j) {
263  k = groupPtr->atomId(j);
264  }
265  distributor.add();
266  }
267  // Send any groups not sent previously.
268  distributor.send();
269  } else { // If I am not the master processor
270  // Receive all groups into BondStorage
271  distributor.receive();
272  }
273  return nGroup;
274  }

308  */
309  void ConfigIo::setAtomMasks()
310  {
311 
312  AtomIterator atomIter;
313  atomStorage().begin(atomIter);
314  for ( ; atomIter.notEnd(); ++atomIter) {
315  atomIter->mask().clear();
316  }
317 
318  int atomId0, atomId1;
319  Atom* atomPtr0;
320  Atom* atomPtr1;
321  GroupIterator<2> bondIter;
322  bondStorage().begin(bondIter);
323  for ( ; bondIter.notEnd(); ++bondIter) {
324  atomId0 = bondIter->atomId(0);
325  atomId1 = bondIter->atomId(1);
326  atomPtr0 = atomStorage().find(atomId0);
327  atomPtr1 = atomStorage().find(atomId1);
328  if (atomPtr0) {
329  atomPtr0->mask().append(atomId1);
330  }
331  if (atomPtr1) {
332  atomPtr1->mask().append(atomId0);
333  }
334  }
